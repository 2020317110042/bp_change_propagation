190   /**
191    * Entry point for running the configuration mapper from the command line.
192    *
193    * @param argv  command line arguments
194    */
195   public static void main(String[] argv) throws Exception {
196 
197     final Cmapper cmapper = new Cmapper();
198 
199     // Parse command line arguments, initializing the model
200     int j = -1;  // the index within args in which "-apromore_model" occurs
201     String user = null;
202     for (int i = 0; i < argv.length; i++) {
203       switch (argv[i]) {
204       case "-apromore_model":
205         if (i+3 >= argv.length) {
206           throw new IllegalArgumentException("-apromore_model without id/branch/version");
207         }
208         j = i;
209         i += 3;
210         break;
211 
212       case "-cmap":
213         if (++i >= argv.length) {
214           throw new IllegalArgumentException("-cmap without filename");
215         }
216         cmapper.setCmap(new FileCmap(new File(argv[i])));
217         break;
218 
219       case "-model":
220         if (++i >= argv.length) {
221           throw new IllegalArgumentException("-model without filename");
222         }
223         cmapper.setModel(new FileProcessModel(new File(argv[i])));
224         break;
225 
226       case "-qml":
227         if (++i >= argv.length) {
228           throw new IllegalArgumentException("-qml without filename");
229         }
230         cmapper.setQml(new FileQml(new File(argv[i])));
231         break;
232 
233       case "-user":
234         if (++i >= argv.length) {
235           throw new IllegalArgumentException("-user without user name");
236         }
237         user = argv[i];
238         break;
239 
240       default:
241         throw new IllegalArgumentException("Unknown parameter: " + argv[i]);
242       }
243     }
244 
245     // -apromore_model and -user can occur in either order, so we've deferred processing them until here
246     if (j != -1) {
247       cmapper.setModel(new ApromoreProcessModel(
248         Integer.valueOf(argv[j+1]),  // process ID
249         argv[j+2],                   // branch
250         argv[j+3],                   // version number
251         null,                        // Swing parent component
252         user                         // user name
253       ));
254     }
255 
256     SwingUtilities.invokeLater(new Runnable() {
257       public void run() {
258         try {
259           new Main(cmapper);
260         } catch (Exception e) {
261           e.printStackTrace();
262         }
263       }
264     });
265   }
