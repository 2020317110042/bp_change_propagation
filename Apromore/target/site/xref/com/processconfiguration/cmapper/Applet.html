57  /**
58   * Present the Cmapper as an applet.
59   */
60  public class Applet extends JApplet {
61  
62      private static ResourceBundle bundle = ResourceBundle.getBundle("com.processconfiguration.cmapper.Applet");
63  
64      public void init() {
65          super.init();
66  
67          try {
68              SwingUtilities.invokeAndWait(new Runnable() {
69  
70                  // Location in the WebDAV repository from which we'll be reading and writing the cmap
71                  private final Cmapper cmapper = new Cmapper();
72                  private Cmap cmap = null;
73                  private FileStoreService fileStore;
74  
75                  public void run() {
76                      try {
77  
78                          // Obtain the proxy for the WebDAV repository
79                          ApplicationContext applicationContext = new ClassPathXmlApplicationContext("classpath:/META-INF/spring/filestoreClientContext.xml");
80                          fileStore = (FileStoreService) applicationContext.getAutowireCapableBeanFactory().getBean("fileStoreClientExternal");
81  
82                          // Check cmap_url parameter
83                          try {
84                              System.err.println("Cmap_url=" + getParameter("cmap_url") + " uri=" + new URI(getParameter("cmap_url")));
85                              cmap = new DavCmap(new URI(getParameter("cmap_url")), fileStore);
86                              cmapper.setCmap(cmap);
87  
88                          } catch (Exception e) {
89                              e.printStackTrace();
90                              showStatus("Did not obtain existing Cmap: " + e);
91                              cmap = null;
92                          } 
93  
94                          // Check qml_url parameter
95                          try {
96                              System.err.println("Qml_url=" + getParameter("qml_url") + " uri=" + new URI(getParameter("qml_url")));
97                              cmapper.setQml(new DavQml(new URI(getParameter("qml_url")), fileStore));
98  
99                          } catch (Exception e) {
100                             e.printStackTrace();
101                             showStatus("Did not obtain existing questionnaire: " + e);
102                         } 
103 
104                         // Check apromore_model parameter
105                         final String model = getParameter("apromore_model");
106                         if (model != null) {
107                             try {
108                                 Pattern pattern = Pattern.compile("(\\S+)\\s+(\\S+)\\s+(\\S+)");
109                                 Matcher matcher = pattern.matcher(model);
110                                 if (!matcher.matches()) {
111                                     throw new Exception("Unable to parse apromore_model param: " + model);
112                                 }
113 
114                                 int    processID = Integer.valueOf(matcher.group(1));
115                                 String branch    = matcher.group(2);
116                                 String version   = matcher.group(3);
117                                 String user      = getParameter("user");
118 
119                                 cmapper.setModel(new ApromoreProcessModel(processID, branch, version, Applet.this, user));
120 
121                             } catch (Exception e) {
122                                 e.printStackTrace();
123                                 showStatus(bundle.getString("Unable_to_read_model") + model);
124                                 //log("Exception in apromore model");
125                             }
126                         }
127 
128                         final JPanel contentPane = new JPanel(new BorderLayout());
129                         setContentPane(contentPane);
130 
131                         contentPane.add(new CmapperView(cmapper), BorderLayout.CENTER);
132                         contentPane.add(createButtonPanel(cmapper, contentPane), BorderLayout.SOUTH);
133 
134                     } catch (Exception e) {
135                         e.printStackTrace();
136                     }
137                 }
138 
139                 /**
140                  * Create the panel of buttons that replace the menu options of the desktop version of the Cmapper.
141                  *
142                  * @param cmapper  the model manipulated by these controllers
143                  * @param contentPane
144                  * @return a panel containing "Load QML" and "Save C-map" buttons
145                  */
146                 private JPanel createButtonPanel(final Cmapper cmapper, final JPanel contentPane) {
147                     final JPanel buttonPanel = new JPanel(new FlowLayout());
148 
149                     // "Load QML" button downloads a questionnaire
150                     JButton loadButton = new JButton(new AbstractAction(bundle.getString("Load_qml")) {
151                         public void actionPerformed(ActionEvent event) {
152                             showStatus(bundle.getString("Load_qml"));
153                             try {
154                                 DavFileSystemView dfsv = new DavFileSystemView(fileStore);
155 
156                                 /* Jakarta Slide variation
157                                 URI uri = fileStore.getBaseURI();
158                                 String server = uri.getScheme() + "://" + uri.getRawAuthority();
159                                 String path   = uri.getPath() + "/";
160                                 System.err.println("WebdavSystemView server=" + server + " path=" + path);
161                                 FileSystemView dfsv = new WebdavSystemView(server, path, "admin", "password");
162                                 */
163 
164                                 System.err.println("File system view roots=" + dfsv.getRoots() + " default=" + dfsv.getDefaultDirectory());
165 
166                                 JFileChooser chooser = new JFileChooser(dfsv);
167                                 chooser.setDialogTitle(bundle.getString("Load_qml"));
168                                 chooser.setFileFilter(new FileNameExtensionFilter(bundle.getString("QML_questionnaire"), "qml"));
169                                 if (chooser.showOpenDialog(Applet.this) == JFileChooser.APPROVE_OPTION) {
170                                     // Load the selected QML file
171                                     System.err.println("File chooser selected " + chooser.getSelectedFile());
172                                     URI fileURI = chooser.getSelectedFile().toURI();
173                                     //URI fileURI = new URI(chooser.getSelectedFile().toString());
174                                     System.err.println("File chooser selection as URI " + fileURI);
175                                     URI selectedURI = new URI(fileURI.getRawPath());
176                                     System.err.println("File chooser selection as relative URI " + selectedURI);
177                                     cmapper.setQml(new DavQml(selectedURI, fileStore));
178                                     JOptionPane.showMessageDialog(Applet.this, "Loaded questionnaire " + chooser.getSelectedFile());
179                                 }
180                                 
181                             } catch (Exception e) {
182                                 e.printStackTrace();
183                                 JOptionPane.showMessageDialog(
184                                     null,
185                                     bundle.getString("Unable_to_read_qml") + e,
186                                     bundle.getString("Error"),
187                                     JOptionPane.ERROR_MESSAGE
188                                 );
189                             }
190                         }
191                     });
192                     loadButton.setToolTipText(bundle.getString("loadButton_tooltip"));
193                     buttonPanel.add(loadButton);
194 
195                     // "Show QML" button pops out a cheat sheet for the questionnaire facts
196                     final JButton showButton = new JButton(new AbstractAction(bundle.getString("Show_qml")) {
197                         QmlFrame qmlFrame = null;
198                         public void actionPerformed(ActionEvent event) {
199                             showStatus(bundle.getString("Show_qml"));
200                             if (qmlFrame != null) {
201                                 qmlFrame.dispose();
202                             }
203                             qmlFrame = new QmlFrame(cmapper.getQml());
204                             qmlFrame.setVisible(true);
205                         }
206                     });
207                     showButton.setEnabled(cmapper.isQmlSet());
208                     showButton.setToolTipText(bundle.getString("showButton_tooltip"));
209                     buttonPanel.add(showButton);
210 
211                     // "Show QML" button enabled/disabled whenever the cmapper model changes
212                     cmapper.addObserver(new Observer() {
213                         public void update(Observable observable, Object arg) {
214                             showButton.setEnabled(((Cmapper) observable).isQmlSet());
215                         }
216                     });
217 
218                     // "Save C-map" button uploads the cmap file via WebDAV
219                     JButton saveButton = new JButton(new AbstractAction(bundle.getString("Save_cmap")) {
220                         public void actionPerformed(ActionEvent event) {
221                             showStatus(bundle.getString("Save_cmap"));
222                             try {
223                                 JFileChooser chooser = new JFileChooser(new DavFileSystemView(fileStore));
224                                 //JFileChooser chooser = new JFileChooser(new WebdavSystemView("http://localhost:9000", "/filestore/dav/", "admin", "password"));  // Jakarta Slide variation
225                                 chooser.setDialogTitle(bundle.getString("Save_cmap"));
226                                 chooser.setFileFilter(new FileNameExtensionFilter(bundle.getString("Configuration_mapping"), "cmap"));
227                                 if (cmap != null) {
228                                     chooser.setSelectedFile(new File(cmap.getURI().getPath()));
229                                 }
230                                 if (chooser.showSaveDialog(Applet.this) == JFileChooser.APPROVE_OPTION) {
231                                     // Save the C-Map as the chosen filename
232                                     URI uri = new URI("/filestore/dav" + chooser.getSelectedFile().toURI().getRawPath());
233                                     //URI uri = new URI(chooser.getSelectedFile().toString().replaceAll(" ", "%20"));  // Jakarta Slide variation
234                                     cmap = new DavCmap(uri, fileStore);
235                                     cmapper.save(cmap);
236                                     cmapper.setCmap(cmap);
237                                     JOptionPane.showMessageDialog(Applet.this, "Saved configuration mapping " + uri);
238                                 }
239 
240                             } catch (Exception e) {
241                                 e.printStackTrace();
242                                 JOptionPane.showMessageDialog(
243                                     null,
244                                     bundle.getString("Unable_to_save_cmap") + e,
245                                     bundle.getString("Error"),
246                                     JOptionPane.ERROR_MESSAGE
247                                 );
248                             }
249                         }
250                     });
251                     saveButton.setToolTipText(bundle.getString("saveButton_tooltip"));
252                     buttonPanel.add(saveButton);
253 
254                     // "Link model" button writes the URL of this cmap into the C-BPMN
255                     JButton linkButton = new JButton(new AbstractAction("Link") {
256                         public void actionPerformed(ActionEvent event) {
257                             showStatus("Link");
258                             try {
259                                 cmapper.link();
260                             } catch (Exception e) {
261                                 e.printStackTrace();
262                                 JOptionPane.showMessageDialog(
263                                     null,
264                                     "Unable to link from model to cmap: " + e,
265                                     bundle.getString("Error"),
266                                     JOptionPane.ERROR_MESSAGE
267                                 );
268                             }
269                         }
270                     });
271                     linkButton.setToolTipText("Insert link from the BPMN to this configuration map");
272                     buttonPanel.add(linkButton);
273 
274                     return buttonPanel;
275                 }
276             });
277         } catch (Exception e) {
278             System.err.println("Failed to initialize applet, exception message: " + e.getMessage());
279             e.printStackTrace();
280         }
281     }
