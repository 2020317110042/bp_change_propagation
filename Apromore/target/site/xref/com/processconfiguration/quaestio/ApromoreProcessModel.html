110 	 * Update a C-BPMN process model on the Apromore manager service.

112          * This presents a {@link SaveDialog} window with which
113          * the user can invoke the {@link #commit} method.

117 	public void update(final BpmnDefinitions bpmn) throws Exception {
118 
119 		// Determine the default next version and branch
120 		Version newVersion = new Version(version.getMajor(), version.getMinor() + 1);
121 		String  newBranch  = branch;
122 
123                 // Present a UI to validate or edit the next version and branch
124 		if (saveDialog == null) {
125 			Container ancestor = SwingUtilities.getAncestorOfClass(Frame.class, parent);
126 			System.err.println("Quaestio save dialog modal parent is " + ancestor);
127 			saveDialog = new SaveDialog((Frame) ancestor);
128 			saveDialog.pack();
129 		}
130 		saveDialog.setModel(bpmn);
131 		saveDialog.setVersion(new Version(version.getMajor(), version.getMinor() + 1));
132 		saveDialog.setBranch(branch);
133 		saveDialog.setLocationRelativeTo(parent);
134                 saveDialog.setVisible(true);
135 	}

137 	/**
138 	 * Update a C-BPMN process model on the Apromore manager service.
139          *
140          * @param bpmn the updated process model
141 	 * @param newVersion  the version number of the update
142 	 * @param newBranch  the branch of the update
143 	 * @return whether the update was successful
144 	 */
145 	boolean commit(final BpmnDefinitions bpmn, final Version newVersion, final String newBranch) {
146 		try {
147 			// Serialize
148 			ByteArrayOutputStream baos = new ByteArrayOutputStream();
149 			bpmn.marshal(baos, true);
150 
151 			// Send to the server
152 			manager.updateProcess(
153 				0,			// session code
154 				user,			// user name
155 				NATIVE_TYPE,		// native type
156 				processId,		// process ID
157 				null,			// domain
158 				"non-null dummy",	// process name
159 				branch,			// original branch name
160 				branch,			// new branch name
161 				newVersion.toString(),	// version number
162 				version.toString(),	// original version number
163 				null,			// pre-version
164 				new ByteArrayInputStream(baos.toByteArray()),
165                 0 //propagation index
166 			);
167 
168 			// Updated process attributes become the current process attributes
169 			this.version = newVersion;
170 			this.branch  = newBranch;
171 
172 			return true;
173 
174 		} catch (Exception e) {
175 			String message = e.getMessage();
176 
177 			// See if we can give a better explanation
178 			if (message != null && message.startsWith("More than one result was returned from Query.getSingleResult()")) {
179 				message = "This version already exists on this branch.\n" +
180 				          "Select a different version or branch and retry.";
181 			} else {
182 				message = "Unable to save model.";
183 			}
184 
185 			JOptionPane.showMessageDialog(null, message, "Unable to save model", JOptionPane.ERROR_MESSAGE);
186 			System.err.println("Cause was " + e.getCause());
187 			e.printStackTrace();
188 
189 			return false;
190 		}
191 	}
