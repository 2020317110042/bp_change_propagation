326 	// this method returns 'hidden' if the inputs are 'activated' and 'blocked',
327 	// or 'blocked' if the inputs are 'activated' and 'hidden', otherwise
328 	// 'activated'
329 	private String getOtherTypeABH(String type1, String type2) {
330 		if ((type1.equals("activated") || type1.equals("blocked"))
331 				& (type2.equals("blocked") || type2.equals("activated")))
332 			return "hidden";
333 		else if ((type1.equals("activated") || type1.equals("hidden"))
334 				& (type2.equals("hidden") || type2.equals("activated")))
335 			return "blocked";
336 		else
337 			return "activated";
338 	}

344 		YAWLSpecificationFactsType s = cyawlModel.getSpecification().get(0);

347 			if (s.getDecomposition().get(index) != null) {

349 						index);

353 																// decomposition
354 																// with
355 																// attribute
356 																// isRootNet='true'
357 				}
358 			}
359 		}
360 		return null;
361 	}

363 	// this method updates the input C-YAWL model by performing the
364 	// configuration (it adds the <configuration> tag in a <task> tag with the
365 	// proper values)
366 	private static void updateCYAWLModel(String configType, String taskId,
367 			String flows, Object type) {
368 		ConfigurationType configuration;
369 		OutputPortConfigType sPort;
370 		InputPortConfigType jPort;
371 		ExternalNetElementType flow;
372 		List<OutputPortConfigType> sPorts;
373 		List<InputPortConfigType> jPorts;
374 		List<String> flowsList;
375 		boolean noConf = false, noSplit = false, noSPort = false, noJPort = false, noJoin = false, noRem = false, noNofi = false;
376 
377 		try {
378 			ExternalTaskFactsType task = getTask(taskId);// retrieve the proper
379 															// task, to which
380 															// the configuration
381 															// has to be applied
382 			if (task == null) {
383 				System.err.println("Task does not exist:".concat(taskId));
384 			} else {
385 
386 				if ((configuration = task.getConfiguration()) == null) {
387 					noConf = true;
388 					configuration = new ConfigurationType();
389 				}
390 
391 				if (configType.equals("split")) {
392 					SplitConfigType split;
393 					if ((split = configuration.getSplit()) == null) {
394 						noSplit = true;
395 						split = new SplitConfigType();
396 						try {
397 							OutputPortValueType temp;
398 							temp = task.getDefaultConfiguration().getSplit()
399 									.getValue();
400 							split.setValue(temp);
401 						} catch (NullPointerException e) {
402 							split.setValue(OutputPortValueType.ACTIVATED);// if
403 																			// no
404 																			// defaultConfiguration,
405 																			// then
406 																			// the
407 																			// "most common"
408 																			// value
409 																			// is
410 																			// set
411 																			// to
412 																			// 'activated'
413 						}
414 					}
415 
416 					flowsList = retrieveFlows(flows);
417 					if ((sPorts = split.getPort()).size() != 0) {// there exists
418 																	// at least
419 																	// a port
420 																	// already
421 
422 						if ((sPort = checkOutputPort(sPorts, flowsList)) == null) {// if
423 																					// ==
424 																					// null
425 																					// means
426 																					// the
427 																					// requested
428 																					// port
429 																					// doesn't
430 																					// exist
431 							noSPort = true;
432 							sPort = new OutputPortConfigType();
433 						}
434 					} else {
435 						noSPort = true;
436 						sPort = new OutputPortConfigType();
437 					}
438 
439 					// modify port according to the mapping
440 					sPort.setValue((OutputPortValueType) type);
441 					if (noSPort) {// set flowDestinations within the port
442 						for (String flowId : flowsList) {
443 							flow = new ExternalNetElementType();
444 							flow.setId(flowId);
445 							sPort.getFlowDestination().add(flow);
446 						}
447 						split.getPort().add(sPort);
448 					}
449 
450 					if (noSplit)
451 						configuration.setSplit(split);
452 				}
453 
454 				else if (configType.equals("join")) {
455 					JoinConfigType join;
456 					if ((join = configuration.getJoin()) == null) {
457 						noJoin = true;
458 						join = new JoinConfigType();
459 						try {
460 							InputPortValueType temp;
461 							temp = task.getDefaultConfiguration().getJoin()
462 									.getValue();
463 							join.setValue(temp);
464 						} catch (NullPointerException e) {
465 							join.setValue(InputPortValueType.ACTIVATED);// if no
466 																		// defaultConfiguration,
467 																		// then
468 																		// the
469 																		// "most common"
470 																		// value
471 																		// is
472 																		// set
473 																		// to
474 																		// 'activated'
475 						}
476 					}
477 
478 					flowsList = retrieveFlows(flows);
479 					if ((jPorts = join.getPort()).size() != 0) {// there exists
480 																// at least a
481 																// port already
482 
483 						if ((jPort = checkInputPort(jPorts, flowsList)) == null) {// if
484 																					// ==
485 																					// null
486 																					// means
487 																					// the
488 																					// requested
489 																					// port
490 																					// doesn't
491 																					// exist
492 							noJPort = true;
493 							jPort = new InputPortConfigType();
494 						}
495 					} else {
496 						noJPort = true;
497 						jPort = new InputPortConfigType();
498 					}
499 
500 					// modify port according to the mapping
501 					jPort.setValue((InputPortValueType) type);
502 					if (noJPort) {// set flowDestinations within the port
503 						for (String flowId : flowsList) {
504 							flow = new ExternalNetElementType();
505 							flow.setId(flowId);
506 							jPort.getFlowSource().add(flow);
507 						}
508 						join.getPort().add(jPort);
509 					}
510 
511 					if (noJoin)
512 						configuration.setJoin(join);
513 				}
514 
515 				else if (configType.equals("rem")) {
516 					RemConfigType rem;
517 					if ((rem = configuration.getRem()) == null) {
518 						noRem = true;
519 						rem = new RemConfigType();
520 					}
521 					rem.setValue((OutputPortValueType) type);
522 
523 					if (noRem)
524 						configuration.setRem(rem);
525 				}
526 
527 				else if (configType.equals("nofi")) {
528 					NofiConfigType nofi;
529 					if ((nofi = configuration.getNofi()) == null) {
530 						noNofi = true;
531 						nofi = new NofiConfigType();
532 					}
533 					org.yawlfoundation.yawlschema.NofiConfigType value = (org.yawlfoundation.yawlschema.NofiConfigType) type;
534 					nofi.setMinIncrease(value.getMinIncrease());
535 					nofi.setMaxDecrease(value.getMaxDecrease());
536 					nofi.setThresIncrease(value.getThresIncrease());
537 					nofi.setCreationMode(value.getCreationMode());
538 
539 					if (noNofi)
540 						configuration.setNofi(nofi);
541 				}
542 
543 				else
544 					System.err
545 							.println("Configuration type is none of 'split', 'join', 'rem', 'nofi'.");
546 
547 				if (noConf)
548 					task.setConfiguration(configuration);
549 			}
550 
551 		} catch (Exception e) {
552 			e.printStackTrace();
553 		}
554 	}

556 	private static List<String> retrieveFlows(String flows) {
557 		StringTokenizer flowsTokenize = new StringTokenizer(flows);
558 		ArrayList<String> flowsList = new ArrayList<String>();
559 		while (flowsTokenize.hasMoreTokens()) {
560 			flowsList.add(flowsTokenize.nextToken());
561 		}
562 		return flowsList;
563 	}

565 	// checks whether an output port is already contained in the <split> child
566 	// of the <configuration> tag of a given task
567 	private static OutputPortConfigType checkOutputPort(
568 			List<OutputPortConfigType> ports, List<String> flowsList) {
569 		int count;
570 		for (OutputPortConfigType currentPort : ports) {
571 			count = 0;
572 			if (currentPort.getFlowDestination().size() == flowsList.size()) {// first
573 																				// checks
574 																				// if
575 																				// the
576 																				// number
577 																				// of
578 																				// flows
579 																				// corresponds,
580 																				// to
581 																				// avoid
582 																				// situations
583 																				// such
584 																				// as
585 																				// port
586 																				// ABC
587 																				// that
588 																				// comprises
589 																				// AC
590 				for (ExternalNetElementType currentFlow : currentPort
591 						.getFlowDestination()) {
592 					for (String flowId : flowsList) {
593 						if (flowId.equals(currentFlow.getId())) {
594 							count++;
595 							break;
596 						}
597 					}
598 				}
599 				if (count == flowsList.size())
600 					return currentPort;
601 			}
602 		}
603 		return null;
604 	}

616 																			// the

618 																			// of
619 																			// flows
620 																			// corresponds,
621 																			// to
