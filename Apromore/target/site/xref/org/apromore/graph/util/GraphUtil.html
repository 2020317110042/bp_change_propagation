28  public final class GraphUtil {
29      
30      private static final Logger LOGGER = LoggerFactory.getLogger(GraphUtil.class);
31  
32      /* Only here so people can't instantiate this class */
33      private GraphUtil() { }

35      /**
36       * Finds out the type of this node.
37       * @param node returns the type of this node.
38       * @return the type of the node.
39       */
40      public static String getType(final INode node) {
41          String type = null;
42          if (node != null) {
43              if (node.getNodeType() != null) {
44                  if (isTask(node.getNodeType())) {
45                      type = GraphConstants.FUNCTION;
46                  } else if (isEvent(node.getNodeType())) {
47                      type = GraphConstants.EVENT;
48                  } else if (isMessage(node.getNodeType())) {
49                      type = GraphConstants.EVENT;
50                  } else if (isTimer(node.getNodeType())) {
51                      type = GraphConstants.EVENT;
52                  } else if (isJoin(node.getNodeType(), node.getName()) || isSplit(node.getNodeType(), node.getName()) ||
53                          isState(node.getNodeType(), node.getName())) {
54                      type = GraphConstants.CONNECTOR;
55                  } else {
56                      LOGGER.warn("Unable to determine Node Type, Type is UNKNOWN. " + node.getId());
57                  }
58              } else {
59                  LOGGER.warn("Unable to determine Node Type, Type is NULL (Could be a Pocket). " + node.getId());
60              }
61          } else {
62              LOGGER.warn("Unable to determine Node Type, Node is NULL. ");
63          }
64          return type;
65      }
66  
67      private static boolean isTask(NodeTypeEnum nodeType) {
68          return (nodeType.equals(NodeTypeEnum.TASK));
69      }
70  
71      private static boolean isEvent(NodeTypeEnum nodeType) {
72          return (nodeType.equals(NodeTypeEnum.EVENT));
73      }
74  
75      private static boolean isMessage(NodeTypeEnum nodeType) {
76          return (nodeType.equals(NodeTypeEnum.MESSAGE));
77      }
78  
79      private static boolean isTimer(NodeTypeEnum nodeType) {
80          return (nodeType.equals(NodeTypeEnum.TIMER));
81      }
82  
83      private static boolean isJoin(NodeTypeEnum nodeType, String nodeName) {
84          return (isOrJoin(nodeType, nodeName) || isXOrJoin(nodeType, nodeName) || isAndJoin(nodeType, nodeName));
85      }
86  
87      private static boolean isSplit(NodeTypeEnum nodeType, String nodeName) {
88          return (isOrSplit(nodeType, nodeName) || isXOrSplit(nodeType, nodeName) || isAndSplit(nodeType, nodeName));
89      }
90  
91      private static boolean isState(NodeTypeEnum nodeType, String nodeName) {
92          return nodeType.equals(NodeTypeEnum.STATE) || (nodeName != null && nodeName.equals("State"));
93      }
94  
95      private static boolean isOrJoin(NodeTypeEnum nodeType, String nodeName) {
96          return nodeType.equals(NodeTypeEnum.ORJOIN) || (nodeName != null && nodeName.equals("OrJoin"));
97      }
98  
99      private static boolean isXOrJoin(NodeTypeEnum nodeType, String nodeName) {
100         return nodeType.equals(NodeTypeEnum.XORJOIN) || (nodeName != null && nodeName.equals("XOrJoin"));
101     }
102 
103     private static boolean isAndJoin(NodeTypeEnum nodeType, String nodeName) {
104         return nodeType.equals(NodeTypeEnum.ANDJOIN) || (nodeName != null && nodeName.equals("AndJoin"));
105     }
106 
107     private static boolean isOrSplit(NodeTypeEnum nodeType, String nodeName) {
108         return nodeType.equals(NodeTypeEnum.ORSPLIT) || (nodeName != null && nodeName.equals("OrSplit"));
109     }
110 
111     private static boolean isXOrSplit(NodeTypeEnum nodeType, String nodeName) {
112         return nodeType.equals(NodeTypeEnum.XORSPLIT) || (nodeName != null && nodeName.equals("XOrSplit"));
113     }
114 
115     private static boolean isAndSplit(NodeTypeEnum nodeType, String nodeName) {
116         return nodeType.equals(NodeTypeEnum.ANDSPLIT) || (nodeName != null && nodeName.equals("AndSplit"));
117     }
