90      /**
91       * Builds a graph from the CPF XSD Format.
92       * @param cpf the cpf format from the canoniser.
93       * @return the JBpt CPF Graph representation
94       */
95      public Canonical convert(final CanonicalProcessType cpf) {
96          Canonical g = new Canonical();
97  
98          g.setName(cpf.getName());
99          g.setUri(cpf.getUri());
100         g.setVersion(cpf.getVersion());
101         g.setAuthor(cpf.getAuthor());
102         g.setCreationDate(cpf.getCreationDate());
103         g.setModifiedDate(cpf.getModificationDate());
104 
105         addAttributes(cpf, g);
106         addObjects(cpf, g);
107         addResources(cpf, g);
108         constructNodeAndEdges(cpf, g);
109 
110         return g;
111     }

114     /* populate the Graph with attributes. */
115     private void addAttributes(final CanonicalProcessType cpf, final Canonical g) {
116         List<TypeAttribute> ats = cpf.getAttribute();
117         for (TypeAttribute attr : ats) {
118             g.setProperty(attr.getName(), attr.getValue(), attr.getAny());
119         }
120     }

122     /* populate the Graph with the Objects. */
123     private void addObjects(final CanonicalProcessType cpf, final Canonical g) {
124         ICPFObject cpfObject;
125         for (NetType net : cpf.getNet()) {
126             for (ObjectType objectType : net.getObject()) {
127                 cpfObject = new CPFObject();
128                 cpfObject.setId(objectType.getId());
129                 cpfObject.setNetId(net.getId());
130                 cpfObject.setOriginalId(objectType.getOriginalID());
131                 if (objectType.getName() != null) {
132                     cpfObject.setName(objectType.getName());
133                 } else {
134                     cpfObject.setName(objectType.getId());
135                 }
136                 if (objectType.isConfigurable() != null) {
137                     cpfObject.setConfigurable(objectType.isConfigurable());
138                 }
139                 if (objectType instanceof SoftType) {
140                     cpfObject.setObjectType(ObjectTypeEnum.SOFT);
141                     cpfObject.setSoftType(((SoftType) objectType).getType());
142                 } else {
143                     cpfObject.setObjectType(ObjectTypeEnum.HARD);
144                 }
145 
146                 for (TypeAttribute type : objectType.getAttribute()) {
147                     cpfObject.setAttribute(type.getName(), type.getValue(), type.getAny());
148                 }
149 
150                 g.getObjects().add(cpfObject);
151             }
152         }
153     }

155     /* populate the Graph with the Resources. */
156     private void addResources(final CanonicalProcessType cpf, final Canonical g) {
157         ICPFResource cpfResource;
158         for (ResourceTypeType resourceType : cpf.getResourceType()) {
159             cpfResource = new CPFResource();
160             cpfResource.setId(resourceType.getId());
161             cpfResource.setOriginalId(resourceType.getOriginalID());
162             cpfResource.setSpecializationIds(resourceType.getSpecializationIds());
163             if (resourceType.getName() != null) {
164                 cpfResource.setName(resourceType.getName());
165             } else {
166                 cpfResource.setName(resourceType.getId());
167             }
168             if (resourceType.isConfigurable() != null) {
169                 cpfResource.setConfigurable(resourceType.isConfigurable());
170             }
171             if (resourceType instanceof HumanType) {
172                 cpfResource.setResourceType(ResourceTypeEnum.HUMAN);
173                 if (((HumanType) resourceType).getType() != null) {
174                     cpfResource.setHumanType(HumanTypeEnum.fromValue(((HumanType) resourceType).getType().value()));
175                 }
176             } else if (resourceType instanceof NonhumanType) {
177                 cpfResource.setResourceType(ResourceTypeEnum.NONHUMAN);
178                 if (((NonhumanType) resourceType).getType() != null) {
179                     cpfResource.setNonHumanType(NonHumanTypeEnum.fromValue(((NonhumanType) resourceType).getType().value()));
180                 }
181             }
182 
183             for (TypeAttribute type : resourceType.getAttribute()) {
184                 cpfResource.setAttribute(type.getName(), type.getValue(), type.getAny());
185             }
186 
187             g.getResources().add(cpfResource);
188         }

192     /* Add a node to the graph, could be of any of the types */
193     private void constructNodeAndEdges(final CanonicalProcessType cpf, final Canonical g) {
194         Map<String, CPFNode> flow = new HashMap<>();
195         for (NetType net : cpf.getNet()) {
196             flow.putAll(buildNodeListFromNet(net.getId(), net.getNode(), g));
197             buildEdges(net.getEdge(), g, flow);
198         }
199     }

202     /* Build the Node list for a single Net */
203     private Map<String, CPFNode> buildNodeListFromNet(final String netId, final List<NodeType> nodes, final Canonical g) {
204         Map<String, CPFNode> flow = new HashMap<>();
205         for (NodeType node : nodes) {
206             CPFNode output = new CPFNode();
207             output.setGraph(g);
208             populateNodeDetails(output, node, netId);
209 
210             if (node instanceof WorkType) {
211                 output = populateWorkDetails(output, (WorkType) node);
212                 if (node instanceof EventType) {
213                     output.setNodeType(NodeTypeEnum.EVENT);
214                     if (node instanceof MessageType) {
215                         output = populateMessageDetails(output, (MessageType) node);
216                     } else if (node instanceof TimerType) {
217                         output = populateTimerDetails(output, (TimerType) node);
218                     }
219                 } else if (node instanceof TaskType) {
220                     output = populateTaskDetails(output, (TaskType) node);
221                 } else {
222                     LOGGER.warn("Unknown Work Type in parsing CPF: " + node.getClass().toString());
223                 }
224             } else if (node instanceof StateType) {
225                 output.setNodeType(NodeTypeEnum.STATE);
226                 output.setName(NodeTypeEnum.STATE.value());
227             } else if (node instanceof SplitType) {
228                 if (node instanceof ORSplitType) {
229                     output.setNodeType(NodeTypeEnum.ORSPLIT);
230                     output.setName(NodeTypeEnum.ORSPLIT.value());
231                 } else if (node instanceof XORSplitType) {
232                     output.setNodeType(NodeTypeEnum.XORSPLIT);
233                     output.setName(NodeTypeEnum.XORSPLIT.value());
234                 } else if (node instanceof ANDSplitType) {
235                     output.setNodeType(NodeTypeEnum.ANDSPLIT);
236                     output.setName(NodeTypeEnum.ANDSPLIT.value());
237                 } else {
238                     LOGGER.warn("Unknown Split Type in parsing CPF: " + node.getClass().toString());
239                 }
240             } else if (node instanceof JoinType) {
241                 if (node instanceof ORJoinType) {
242                     output.setNodeType(NodeTypeEnum.ORJOIN);
243                     output.setName(NodeTypeEnum.ORJOIN.value());
244                 } else if (node instanceof XORJoinType) {
245                     output.setNodeType(NodeTypeEnum.XORJOIN);
246                     output.setName(NodeTypeEnum.XORJOIN.value());
247                 } else if (node instanceof ANDJoinType) {
248                     output.setNodeType(NodeTypeEnum.ANDJOIN);
249                     output.setName(NodeTypeEnum.ANDJOIN.value());
250                 } else {
251                     LOGGER.warn("Unknown Join Type in parsing CPF: " + node.getClass().toString());
252                 }
253             } else {
254                 LOGGER.warn("Unknown Node Type in parsing CPF: " + node.getClass().toString());
255             }
256 
257             flow.put(node.getId(), output);
258         }
259         return flow;
260     }

262     /* Populate the Node with the standard Node details. */
263     private void populateNodeDetails(CPFNode output, NodeType node, String netId) {
264         output.setId(node.getId());

271     /* Populate the Node with the Work Node details. */
272     private CPFNode populateWorkDetails(CPFNode output, WorkType node) {
273         if (node.isTeamWork() != null) {
274             output.setTeamWork(node.isTeamWork());
275         }
276         if (node.getAllocationStrategy() != null) {
277             output.setAllocation(AllocationStrategyEnum.fromValue(node.getAllocationStrategy().value()));
278         }
279 
280         if (node.getFilterByDataExpr() != null) {
281             CPFExpression resDataExpr = new CPFExpression();
282             resDataExpr.setDescription(node.getFilterByDataExpr().getDescription());
283             resDataExpr.setExpression(node.getFilterByDataExpr().getExpression());
284             resDataExpr.setLanguage(node.getFilterByDataExpr().getLanguage());
285             resDataExpr.setReturnType(node.getFilterByDataExpr().getReturnType());
286             output.setResourceDataExpr(resDataExpr);
287         }
288 
289         if (node.getFilterByRuntimeExpr() != null) {
290             CPFExpression resRunExpr = new CPFExpression();
291             resRunExpr.setDescription(node.getFilterByRuntimeExpr().getDescription());
292             resRunExpr.setExpression(node.getFilterByRuntimeExpr().getExpression());
293             resRunExpr.setLanguage(node.getFilterByRuntimeExpr().getLanguage());
294             resRunExpr.setReturnType(node.getFilterByRuntimeExpr().getReturnType());
295             output.setResourceDataExpr(resRunExpr);
296         }
297 
298         addCancelNodes(output, node);
299         addCancelEdges(output, node);
300         addInputExpression(output, node);
301         addOutputExpression(output, node);
302         addObjectReferences(output, node);
303         addResourceReferences(output, node);
304 
305         return output;
306     }

443     /* Builds the list of Edges for this Model */
444     private void buildEdges(final List<EdgeType> edgeTypes, final Canonical graph, final Map<String, CPFNode> nodes) {
445         CPFNode source;
446         CPFNode target;
447         CPFExpression expr;
448 
449         for (EdgeType edge : edgeTypes) {
450             source = nodes.get(edge.getSourceId());
451             target = nodes.get(edge.getTargetId());
452 
453             if (source != null && target != null) {
454                 CPFEdge cpfEdge = graph.addEdge(edge.getId(), source, target);
455 
456                 for (TypeAttribute attribute : edge.getAttribute()) {
457                     cpfEdge.addAttribute(attribute.getName(), attribute.getValue(), attribute.getAny());
458                 }
459 
460                 if (edge.getConditionExpr() != null) {
461                     expr = new CPFExpression();
462                     expr.setDescription(edge.getConditionExpr().getDescription());
463                     expr.setExpression(edge.getConditionExpr().getExpression());
464                     expr.setLanguage(edge.getConditionExpr().getLanguage());
465                     expr.setReturnType(edge.getConditionExpr().getReturnType());
466                     cpfEdge.setConditionExpr(expr);
467                 }
468 
469                 if (edge.isDefault()) {
470                     cpfEdge.setDefault(true);
471                 }
472 
473                 graph.setNodeProperty(source.getId(), GraphConstants.TYPE, GraphUtil.getType(source));
474                 graph.setNodeProperty(target.getId(), GraphConstants.TYPE, GraphUtil.getType(target));
475             } else {
476                 LOGGER.warn("Either the Source or Target Node for edge(" + edge.getId() + ") is NULL.");
477             }
478         }
