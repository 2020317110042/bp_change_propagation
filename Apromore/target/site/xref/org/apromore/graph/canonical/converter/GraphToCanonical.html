97      /**
98       * Builds a Canonical Format Type from the graph.
99       * @param graph the cpf format from the canoniser.
100      * @return the CPF Ty[pe from the Graph
101      */
102     public CanonicalProcessType convert(final Canonical graph) {
103         CanonicalProcessType c = new CanonicalProcessType();
104 
105         c.setName(graph.getName());
106         c.setUri(graph.getVersion());
107         c.setVersion(graph.getVersion());
108         c.setAuthor(graph.getAuthor());
109         c.setCreationDate(graph.getCreationDate());
110         c.setModificationDate(graph.getModifiedDate());
111 
112         addAttributesToCpf(graph, c);
113         addResourcesToCpf(graph, c);
114 
115         buildNets(graph, c);
116 
117         return c;
118     }

121     /* Build the Nets used in the Canonical Format from the graph. */
122     private void buildNets(final Canonical graph, final CanonicalProcessType c) {
123         NetType net;
124         Map<String, NetData> netData = buildNetNodeEdgeMap(graph);
125 
126         for (String netId : netData.keySet()) {
127             net = new NetType();
128             net.setId(netId);
129 
130             addObjectsToNet(net, graph, netData.get(netId).getNodes());
131             for (CPFNode node : netData.get(netId).getNodes()) {
132                 buildNode(net, node);
133             }
134             for (CPFEdge edge : netData.get(netId).getEdges()) {
135                 buildEdge(net, edge);
136             }
137 
138             c.getNet().add(net);
139         }
140     }

143     /* Builds a list of Nodes and Edges corresponding to a Net Id. */
144     private Map<String, NetData> buildNetNodeEdgeMap(Canonical graph) {
145         NetData netData;
146         Map<String, NetData> data = new HashMap<>();
147         for (CPFEdge edge : graph.getEdges()) {
148             if (!data.containsKey(edge.getSource().getNetId())) {
149                 netData = new NetData();
150             } else {
151                 netData = data.get(edge.getSource().getNetId());
152             }
153             netData.addEdge(edge);
154             netData.addNode(edge.getSource());
155             netData.addNode(edge.getTarget());
156             data.put(edge.getSource().getNetId(), netData);
157         }
158         return data;
159     }

161     /* Add the Attributes from the graph to the CPF. */
162     private void addAttributesToCpf(Canonical graph, CanonicalProcessType c) {
163         TypeAttribute typeA;
164         for (String propName : graph.getProperties().keySet()) {
165             typeA = new TypeAttribute();
166             typeA.setName(propName);
167             typeA.setValue(graph.getProperty(propName).getValue());
168             typeA.setAny(graph.getProperty(propName).getAny());
169             c.getAttribute().add(typeA);
170         }
171     }

173     /* Add the Resources from the graph to the CPF. */
174     private void addResourcesToCpf(Canonical graph, CanonicalProcessType c) {
175         for (ICPFResource cpfResource : graph.getResources()) {
176             if (cpfResource.getResourceType() != null) {
177                 if (cpfResource.getResourceType().equals(ResourceTypeEnum.HUMAN)) {
178                     c.getResourceType().add(constructHumanResource(cpfResource));
179                 } else if (cpfResource.getResourceType().equals(ResourceTypeEnum.NONHUMAN)) {
180                     c.getResourceType().add(constructNonHumanResource(cpfResource));
181                 }
182             } else {
183                 c.getResourceType().add(constructResource(cpfResource));
184             }
185         }
186     }

221     private void addObjectsToNet(NetType net, Canonical graph, Set<CPFNode> nodes) {
222         ICPFObject cpfObject;
223         for (CPFNode node : nodes) {
224             for (ICPFObjectReference cpfObjRef : node.getObjectReferences()) {
225                 if (!netContainsObject(net.getObject(), cpfObjRef.getObjectId())) {
226                     cpfObject = findObjectType(graph, cpfObjRef.getObjectId());
227                     if (cpfObject != null) {
228                         if (cpfObject.getObjectType() != null) {
229                             if (cpfObject.getObjectType().equals(ObjectTypeEnum.SOFT)) {
230                                 net.getObject().add(buildSoftTypeObject(cpfObject));
231                             } else if (cpfObject.getObjectType().equals(ObjectTypeEnum.HARD)) {
232                                 net.getObject().add(buildHardTypeObject(cpfObject));
233                             }
234                         } else {
235                             net.getObject().add(buildObject(cpfObject));
236                         }
237                     } else {
238                         LOGGER.warn("Graph contains Object references that don't have a corresponding object. " + cpfObjRef.getObjectId());
239                     }
240                 }
241             }
242         }
243     }

245     /* Does the Net Contain an object with this ID? */
246     private boolean netContainsObject(List<ObjectType> object, String objectId) {
247         boolean result = false;
248         for (ObjectType objType : object) {
249             if (objType.getId().equals(objectId)) {
250                 result = true;
251                 break;
252             }
253         }
254         return result;
255     }

257     /* Try and Find the Object Type using the ObjectId. */
258     private ICPFObject findObjectType(Canonical graph, final String objectId) {
259         ICPFObject objectType = null;
260         for (ICPFObject cpfObject : graph.getObjects()) {
261             if (cpfObject.getId().equals(objectId)) {
262                 objectType = cpfObject;
263                 break;
264             }
265         }
266         return objectType;
267     }

304     private List<TypeAttribute> buildAttributeList(final Map<String, IAttribute> attributes) {
305         TypeAttribute typAtt;
306         List<TypeAttribute> atts = new ArrayList<>();
307         for (Entry<String, IAttribute> e : attributes.entrySet()) {
308             typAtt = new TypeAttribute();
309             typAtt.setName(e.getKey());
310             typAtt.setValue(e.getValue().getValue());
311             typAtt.setAny(e.getValue().getAny());
312             atts.add(typAtt);
313         }
314         return atts;
315     }

317     /* Builds an Edge for this Net. */
318     private void buildEdge(NetType net, final CPFEdge edge) {
319         EdgeType edgeType = new EdgeType();
320         if (edge.getOriginalId() == null) {
321             edgeType.setId(edge.getId());
322         } else {
323             edgeType.setId(edge.getOriginalId());
324         }
325         edgeType.setOriginalID(edge.getOriginalId());
326         edgeType.setDefault(edge.isDefault());
327         if (edge.getConditionExpr() != null) {
328             ConditionExpressionType conditionExpr = new ConditionExpressionType();
329             conditionExpr.setDescription(edge.getConditionExpr().getDescription());
330             conditionExpr.setExpression(edge.getConditionExpr().getExpression());
331             conditionExpr.setLanguage(edge.getConditionExpr().getLanguage());
332             conditionExpr.setReturnType(edge.getConditionExpr().getReturnType());
333             edgeType.setConditionExpr(conditionExpr);
334         }
335         if (edge.getSource() != null) {
336             edgeType.setSourceId(edge.getSource().getId());
337         } else {
338             LOGGER.warn("Edge(" + edge.getId() + ") found with NULL source.");
339         }
340         if (edge.getTarget() != null) {
341             edgeType.setTargetId(edge.getTarget().getId());
342         } else {
343             LOGGER.warn("Edge(" + edge.getId() + ") found with NULL target.");
344         }
345 
346         updateAttributes(edgeType, edge);
347 
348         net.getEdge().add(edgeType);
349     }

351     private void updateAttributes(EdgeType output, CPFEdge edge) {
352         TypeAttribute typeAttribute;
353         for (Entry<String, IAttribute> attribute : edge.getAttributes().entrySet()) {
354             typeAttribute = new TypeAttribute();
355             typeAttribute.setName(attribute.getKey());
356             typeAttribute.setValue(attribute.getValue().getValue());
357             typeAttribute.setAny(attribute.getValue().getAny());
358             output.getAttribute().add(typeAttribute);
359         }
360     }

362     /* Builds a Node for this Net. */
363     private void buildNode(NetType net, final CPFNode node) {
364         if (node.getNodeType() != null ) {
365             if (node.getNodeType().equals(NodeTypeEnum.MESSAGE)) {
366                 net.getNode().add(constructMessageType(node));
367             } else if (node.getNodeType().equals(NodeTypeEnum.TIMER)) {
368                 net.getNode().add(constructTimerType(node));
369             } else if (node.getNodeType().equals(NodeTypeEnum.TASK)) {
370                 net.getNode().add(constructTaskType(node));
371             } else if (node.getNodeType().equals(NodeTypeEnum.EVENT)) {
372                 net.getNode().add(constructEventType(node));
373             } else if (node.getNodeType().equals(NodeTypeEnum.STATE)) {
374                 net.getNode().add(constructStateType(node));
375             } else if (node.getNodeType().equals(NodeTypeEnum.ORJOIN)) {
376                 net.getNode().add(constructORJoinType(node));
377             } else if (node.getNodeType().equals(NodeTypeEnum.XORJOIN)) {
378                 net.getNode().add(constructXORJoinType(node));
379             } else if (node.getNodeType().equals(NodeTypeEnum.ANDJOIN)) {
380                 net.getNode().add(constructANDJoinType(node));
381             } else if (node.getNodeType().equals(NodeTypeEnum.ORSPLIT)) {
382                 net.getNode().add(constructORSplitType(node));
383             } else if (node.getNodeType().equals(NodeTypeEnum.XORSPLIT)) {
384                 net.getNode().add(constructXORSplitType(node));
385             } else if (node.getNodeType().equals(NodeTypeEnum.ANDSPLIT)) {
386                 net.getNode().add(constructANDSplitType(node));
387             } else {
388                 LOGGER.warn("Unknown Node type: " + node.getNodeType());
389             }
390         } else {
391             LOGGER.warn("Unknown Node type: " + node.getNodeType());
392         }
393     }

395     /* Construct the Message Node Type. */
396     private NodeType constructMessageType(final CPFNode node) {
397         MessageType type = new MessageType();
398         updateWorkNodeData(type, node);
399 
400         if (node.getDirection() != null) {
401             type.setDirection(DirectionEnum.valueOf(node.getDirection().toString()));
402         }
403 
404         return type;
405     }

407     /* Construct the Timer Node Type. */
408     private NodeType constructTimerType(final CPFNode node) {
409         TimerType type = new TimerType();
410         updateWorkNodeData(type, node);
411 
412         if (node.getTimeDate() != null) {
413             XMLGregorianCalendar date;
414             try {
415                 date = DatatypeFactory.newInstance().newXMLGregorianCalendar(node.getTimeDate());
416             } catch (DatatypeConfigurationException e) {
417                 date = null;
418             }
419             type.setTimeDate(date);
420         }
421         if (node.getTimeDuration() != null && !node.getTimeDuration().equals("")) {
422             Duration duration;
423             try {
424                 duration = DatatypeFactory.newInstance().newDuration(node.getTimeDuration());
425             } catch (DatatypeConfigurationException e) {
426                 duration = null;
427             }
428             type.setTimeDuration(duration);
429         }
430         if (node.getTimeExpression() != null) {
431             TimerExpressionType timeExpr = new TimerExpressionType();
432             timeExpr.setDescription(node.getTimeExpression().getDescription());
433             timeExpr.setExpression(node.getTimeExpression().getExpression());
434             timeExpr.setLanguage(node.getTimeExpression().getLanguage());
435             timeExpr.setReturnType(node.getTimeExpression().getReturnType());
436             type.setTimeExpression(timeExpr);
437         }
438 
439         return type;
440     }

507     private void updateNodeData(NodeType output, final CPFNode node) {
508         output.setId(node.getId());
509         output.setOriginalID(node.getOriginalId());
510         output.setName(node.getName());
511         if (node.isConfigurable()) {
512             output.setConfigurable(node.isConfigurable());
513         }
514 
515         updateAttributes(output, node);
516     }

518     private void updateWorkNodeData(WorkType output, final CPFNode node) {
519         updateNodeData(output, node);
520         if (node.getAllocation() != null) {
521             output.setAllocationStrategy(AllocationStrategyEnum.fromValue(node.getAllocation().value()));
522         }
523         if (node.getResourceDataExpr() != null) {
524             ResourceDataFilterExpressionType resDataExpr = new ResourceDataFilterExpressionType();
525             resDataExpr.setDescription(node.getResourceDataExpr().getDescription());
526             resDataExpr.setExpression(node.getResourceDataExpr().getExpression());
527             resDataExpr.setLanguage(node.getResourceDataExpr().getLanguage());
528             resDataExpr.setReturnType(node.getResourceDataExpr().getReturnType());
529             output.setFilterByDataExpr(resDataExpr);
530         }
531         if (node.getResourceRuntimeExpr() != null) {
532             ResourceRuntimeFilterExpressionType resRunExpr = new ResourceRuntimeFilterExpressionType();
533             resRunExpr.setDescription(node.getResourceRuntimeExpr().getDescription());
534             resRunExpr.setExpression(node.getResourceRuntimeExpr().getExpression());
535             resRunExpr.setLanguage(node.getResourceRuntimeExpr().getLanguage());
536             resRunExpr.setReturnType(node.getResourceRuntimeExpr().getReturnType());
537             output.setFilterByRuntimeExpr(resRunExpr);
538         }
539 
540         updateCancelNodes(output, node);
541         updateCancelEdges(output, node);
542         updateInputExpressions(output, node);
543         updateOutputExpressions(output, node);
544         updateObjectReferences(output, node);
545         updateResourceReferences(output, node);
546     }

599             objectRefType.setObjectId(objectReference.getObjectId());
600             objectRefType.setConsumed(objectReference.isConsumed());

603                 objectRefType.setType(InputOutputType.fromValue(objectReference.getObjectRefType().value()));
604             }
605             output.getObjectRef().add(objectRefType);
606         }
607     }
