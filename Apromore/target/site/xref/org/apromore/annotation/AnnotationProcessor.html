20  package org.apromore.annotation;
21  
22  import org.apromore.anf.AnnotationsType;
23  import org.apromore.annotation.exception.AnnotationProcessorException;
24  import org.apromore.cpf.CanonicalProcessType;
25  import org.apromore.plugin.ParameterAwarePlugin;
26  import org.apromore.plugin.PluginResult;
27  
28  /**
29   * Interface to an Apromore Annotations post Processor. Each Annotation is build as a OSGi plugin and has to implement this interface.
30   *
31   * @author <a href="mailto:cam.james@gmail.com">Cameron James</a>
32   */
33  public interface AnnotationProcessor extends ParameterAwarePlugin {
34  
35      /**
36       * Type of the source native format which this annotation processor can handle. For example "EPML 2.0" or "YAWL 2.2"
37       * @return the type of the source process format
38       */
39      String getProcessFormatProcessor();
40  
41      /**
42       * Converts the data in native format to the canonical format and its annotation format.
43       * @param canonisedFormat the source canonical format.
44       * @param annotationFormat the the old annotation format.
45       * @return a PluginResult with information about this operation
46       * @throws AnnotationProcessorException in case of an Exception during conversion
47       */
48      PluginResult processAnnotation(CanonicalProcessType canonisedFormat, AnnotationsType annotationFormat)
49              throws AnnotationProcessorException;
50  
51  }
