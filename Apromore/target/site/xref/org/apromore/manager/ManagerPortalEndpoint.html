178     public JAXBElement<MergeProcessesOutputMsgType> mergeProcesses(@RequestPayload final JAXBElement<MergeProcessesInputMsgType> req) {
179         LOGGER.trace("Executing operation mergeProcesses");
180         MergeProcessesInputMsgType payload = req.getValue();
181         MergeProcessesOutputMsgType res = new MergeProcessesOutputMsgType();
182         ResultType result = new ResultType();
183         res.setResult(result);
184         try {
185             // Build data to send to toolbox
186             String algo = payload.getAlgorithm();
187             String processName = payload.getProcessName();
188             String version = payload.getVersionName();
189             String domain = payload.getDomain();
190             Integer folderId = payload.getFolderId();
191             String username = payload.getUsername();
192             boolean makePublic = payload.isMakePublic();
193             ParametersType parameters = new ParametersType();
194             for (ParameterType p : payload.getParameters().getParameter()) {
195                 ParameterType param = new ParameterType();
196                 param.setName(p.getName());
197                 param.setValue(p.getValue());
198                 parameters.getParameter().add(param);
199             }
200             // processes
201             ProcessVersionIdsType ids = new ProcessVersionIdsType();
202             for (ProcessVersionIdType t : payload.getProcessVersionIds().getProcessVersionId()) {
203                 ProcessVersionIdType id = new ProcessVersionIdType();
204                 id.setProcessId(t.getProcessId());
205                 id.setBranchName(t.getBranchName());
206                 id.setVersionNumber(t.getVersionNumber());
207                 ids.getProcessVersionId().add(id);
208             }
209             ProcessModelVersion pmv = merSrv.mergeProcesses(processName, version, domain, username, algo, folderId, parameters, ids, makePublic);
210             ProcessSummaryType process = uiHelper.createProcessSummary(pmv.getProcessBranch().getProcess(), pmv.getProcessBranch(), pmv,
211                     "", domain, pmv.getCreateDate(), pmv.getLastUpdateDate(), username, makePublic);
212 
213             res.setProcessSummary(process);
214             result.setCode(0);
215             result.setMessage("");
216         } catch (Exception ex) {
217             LOGGER.error("", ex);
218             result.setCode(-1);
219             result.setMessage(ex.getMessage());
220         }
221         return WS_OBJECT_FACTORY.createMergeProcessesResponse(res);
222     }
