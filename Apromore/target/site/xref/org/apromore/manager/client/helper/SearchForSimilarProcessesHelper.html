33  public class SearchForSimilarProcessesHelper {
34  
35      public static final String GREEDY_ALGORITHM = "Greedy";
36  
37      /**
38       * Creates the Parameter Type needed by the client.
39       *
40       * @param method the algorithm used by the service.
41       * @param modelThreshold the Model Threshold
42       * @param labelThreshold the Label Threshold
43       * @param contextThreshold the Context Threshold
44       * @param skipnWeight the Skip weight
45       * @param subnWeight the Sub N weight
46       * @param skipeWeight the Skip E weight
47       * @return the ParameterType used in the WebService object
48       */
49      public static ParametersType setParams(String method, double modelThreshold, double labelThreshold, double contextThreshold,
50              double skipnWeight, double subnWeight, double skipeWeight) {
51          ParametersType params = new ParametersType();
52  
53          params.getParameter().add(addParam("modelthreshold", modelThreshold));
54          params.getParameter().add(addParam("labelthreshold", labelThreshold));
55          params.getParameter().add(addParam("contextthreshold", contextThreshold));
56  
57          if (GREEDY_ALGORITHM.equals(method)) {
58              params.getParameter().add(addParam("skipnweight", skipnWeight));
59              params.getParameter().add(addParam("subnweight", subnWeight));
60              params.getParameter().add(addParam("skipeweight", skipeWeight));
61          }
62  
63          return params;
64      }
65  
66  
67      /* Used to create a parameter object. */
68      private static ParameterType addParam(String name, double value) {
69          ParameterType p = new ParameterType();
70          p.setName(name);
71          p.setValue(value);
72          return p;
73      }
