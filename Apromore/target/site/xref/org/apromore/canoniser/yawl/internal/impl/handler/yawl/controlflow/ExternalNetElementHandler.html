79      /**
80       * Simple connection between two already converted element. Source and Target have to be SESE, as we are not going to introduce any kind of
81       * routing node.
82       *
83       * @param sourceNode
84       *            of CPF
85       * @param targetNode
86       *            of CPF
87       * @return a CPF edge
88       * @throws CanoniserException
89       */
90      protected EdgeType createSimpleEdge(final NodeType sourceNode, final NodeType targetNode) throws CanoniserException {
91          final EdgeType edge = CPF_FACTORY.createEdgeType();
92          // Assign unique ID that is reproducible by using sources and targets ID
93          edge.setId(generateEdgeId(sourceNode, targetNode));
94          edge.setSourceId(sourceNode.getId());
95          edge.setTargetId(targetNode.getId());
96          getConvertedParent().getEdge().add(edge);
97          if (sourceNode.getOriginalID() != null && targetNode.getOriginalID() != null) {
98              createGraphicsForFlow(sourceNode.getOriginalID(), targetNode.getOriginalID());
99          } else {
100             createGraphicsForFlow(sourceNode.getId(), targetNode.getId());
101         }
102         return edge;
103     }

128     private void convertEdge(final FlowsIntoType flowsInto, final EdgeType edge) {
129         if (flowsInto.getIsDefaultFlow() != null) {
130             edge.setDefault(true);
131         } else {
132             edge.setDefault(false);
133         }
134         if (flowsInto.getPredicate() != null) {
135             ConditionExpressionType expressionType = CPF_FACTORY.createConditionExpressionType();
136             expressionType.setLanguage(CPFSchema.EXPRESSION_LANGUAGE_XPATH);
137             expressionType.setExpression(flowsInto.getPredicate().getValue());
138             edge.setConditionExpr(expressionType);
139         }
140     }

195     protected void connectToSuccessors(final NodeType node, final List<FlowsIntoType> flowsIntoList) throws CanoniserException {
196         for (final FlowsIntoType flowsInto : flowsIntoList) {
197             // Check if the next element has already been converted (in case of cycles) and get its introduced predecessor node.
198             final NodeType introducedNode = getContext().getIntroducedPredecessor(flowsInto.getNextElementRef());
199             if (introducedNode != null) {
200                 EdgeType edge = createSimpleEdge(node, introducedNode);
201                 convertEdge(flowsInto, edge);
202             } else {
203                 // Add ourself to the waiting queue for the next element
204                 getContext().getIncomingQueue(flowsInto.getNextElementRef()).add(new PredecessorAdapater(node, flowsInto));
205             }
206         }
207     }

209     protected void connectFromPredecessors(final ExternalNetElementType netElement, final NodeType node) throws CanoniserException {
210         // Remember that we introduced a "new" predecessor
211         getContext().addIntroducedPredecessor(netElement, node);
212         // Connect all items of our incoming queue with our "new" predecessor.
213         // Please note in cyclic nets there could be some elements that are not yet in our incoming queue.
214         // These elements will take care of adding themselves!
215         final Iterator<PredecessorAdapater> incomingIterator = getContext().getIncomingQueue(netElement).iterator();
216         while (incomingIterator.hasNext()) {
217             PredecessorAdapater nextItem = incomingIterator.next();
218             EdgeType edge = createSimpleEdge(nextItem.getNode(), node);
219             convertEdge(nextItem.getFlowsInto(), edge);
220             // We are joining so no need for predicates
221             incomingIterator.remove();
222         }
223     }

233     protected void createGraphics(final ExternalNetElementType element) throws CanoniserException {
234         final GraphicsType graphics = ANF_FACTORY.createGraphicsType();
235         graphics.setCpfId(generateUUID(CONTROLFLOW_ID_PREFIX, element.getId()));
236         graphics.setId(generateUUID());
237 
238         // Convert Vertex to ANF
239         convertVertexLayout(getContext().getLayoutVertexForElement(element.getId()), graphics);
240 
241         // Add YAWL specific extensions
242         final Collection<LayoutDecoratorFactsType> layoutDecorators = getContext().getLayoutDecoratorForElement(element.getId());
243         if (layoutDecorators != null) {
244             for (final LayoutDecoratorFactsType decorator : layoutDecorators) {
245                 graphics.getAny().add(ExtensionUtils.marshalYAWLFragment(ExtensionUtils.DECORATOR, decorator, LayoutDecoratorFactsType.class));
246             }
247         }
248         if (getContext().getLayoutLabelForElement(element.getId()) != null) {
249             graphics.getAny().add(
250                     ExtensionUtils.marshalYAWLFragment(ExtensionUtils.LABEL, getContext().getLayoutLabelForElement(element.getId()), LayoutLabelFactsType.class));
251         }
252 
253         getContext().getAnnotationResult().getAnnotation().add(graphics);
254     }

256     /**
257      * Create the graphical annotation for a YAWL edge.
258      *
259      * @param sourceId
260      *            of the YAWL source
261      * @param targetId
262      *            of the YAWL target
263      * @throws CanoniserException
264      */
265     protected void createGraphicsForFlow(final String sourceId, final String targetId) throws CanoniserException {
266         final GraphicsType graphics = ANF_FACTORY.createGraphicsType();
267         graphics.setCpfId(generateUUID(CONTROLFLOW_ID_PREFIX, getContext().buildEdgeId(sourceId, targetId)));
268         graphics.setId(generateUUID());
269 
270         final LayoutFlowFactsType flowLayout = getContext().getLayoutFlow(getContext().buildEdgeId(sourceId, targetId));
271 
272         // Convert and add YAWL specific extension
273         if (flowLayout != null) {
274             graphics.getAny().add(
275                     ExtensionUtils.marshalYAWLFragment(ExtensionUtils.FLOW, getContext().getLayoutFlow(getContext().buildEdgeId(sourceId, targetId)),
276                             LayoutFlowFactsType.class));
277             graphics.setLine(convertFlowLineStyle(flowLayout));
278             try {
279                 graphics.getPosition().addAll(convertFlowPositions(flowLayout));
280                 if (graphics.getPosition().isEmpty()) {
281                     // Calculate start and end position
282                     LayoutVertexFactsType sourceVertex = getContext().getLayoutVertexForElement(sourceId);
283                     LayoutVertexFactsType targetVertex = getContext().getLayoutVertexForElement(targetId);
284                     if (sourceVertex != null && targetVertex != null) {
285                         try {
286                             graphics.getPosition().add(calculateFlowPosition(sourceVertex, flowLayout.getPorts().getOut()));
287                             graphics.getPosition().add(calculateFlowPosition(targetVertex, flowLayout.getPorts().getIn()));
288                         } catch (ParseException e) {
289                             throw new CanoniserException("Could not calculate default position for flow from " + sourceId + " to " + targetId, e);
290                         }
291                     }
292                 }
293 
294             } catch (final ParseException e) {
295                 throw new CanoniserException("Could not convert layout of flow from " + sourceId + " to " + targetId, e);
296             }
297         }
298 
299         getContext().getAnnotationResult().getAnnotation().add(graphics);
300     }

302     private PositionType calculateFlowPosition(final LayoutVertexFactsType vertex, final BigInteger port) throws ParseException {
303         PositionType position = ANF_FACTORY.createPositionType();
304         LayoutRectangleType rect = getRectangleFromVertex(vertex);
305         //TODO decide based on port
306         position.setX(convertToBigDecimal(rect.getX()));
307         position.setY(convertToBigDecimal(rect.getY()));
308         return position;

311     private LayoutRectangleType getRectangleFromVertex(final LayoutVertexFactsType vertex) {
312         final LayoutAttributesFactsType attr = vertex.getAttributes();
313         for (final JAXBElement<?> element : attr.getAutosizeOrBackgroundColorOrBendable()) {
314             final Object elementValue = element.getValue();
315             if (elementValue instanceof LayoutRectangleType) {
316                 return (LayoutRectangleType) elementValue;
317             }
318         }
319         return null;
320     }

322     private Collection<PositionType> convertFlowPositions(final LayoutFlowFactsType flowLayout) throws ParseException {
323         final ArrayList<PositionType> positions = new ArrayList<PositionType>();
324         for (final JAXBElement<?> obj : flowLayout.getAttributes().getAutosizeOrBackgroundColorOrBendable()) {
325             if (obj.getValue() instanceof LayoutPointsType) {
326                 final LayoutPointsType points = (LayoutPointsType) obj.getValue();
327                 for (final LayoutPointType point : points.getValue()) {
328                     final PositionType position = ANF_FACTORY.createPositionType();
329                     position.setX(convertToBigDecimal(point.getX()));
330                     position.setY(convertToBigDecimal(point.getY()));
331                     positions.add(position);
332                 }
333             }
334         }
335         return positions;
336     }

338     private LineType convertFlowLineStyle(final LayoutFlowFactsType flowLayout) {
339         final LineType lineType = ANF_FACTORY.createLineType();
340         lineType.setWidth(new BigDecimal(DEFAULT_LINE_SIZE));
341         for (final JAXBElement<?> obj : flowLayout.getAttributes().getAutosizeOrBackgroundColorOrBendable()) {
342             if (obj.getName().getLocalPart().equalsIgnoreCase(LINE_STYLE)) {
343                 lineType.setWidth(new BigDecimal((BigInteger) obj.getValue()));
344             }
345         }
346         return lineType;
347     }

349     protected void convertVertexLayout(final LayoutVertexFactsType vertex, final GraphicsType graphics) throws CanoniserException {
350         if (vertex != null) {
351             final FillType fill = ANF_FACTORY.createFillType();
352 
353             // Convert all Attributes
354             final LayoutAttributesFactsType attr = vertex.getAttributes();
355             for (final JAXBElement<?> element : attr.getAutosizeOrBackgroundColorOrBendable()) {
356                 final Object elementValue = element.getValue();
357                 if (elementValue instanceof LayoutRectangleType) {
358                     try {
359                         convertLayoutRectangleAttribute((LayoutRectangleType) elementValue, graphics);
360                     } catch (final ParseException e) {
361                         throw new CanoniserException("Could not convert layout of element " + vertex.getId(), e);
362                     }
363                 }
364                 if (element.getName().getLocalPart().equals(BACKGROUND_COLOR) && element.getValue() != null) {
365                     final BigInteger color = (BigInteger) element.getValue();
366                     fill.setColor(ConversionUtils.convertColorToString(color.intValue()));
367                 }
368             }
369 
370             fill.setImage(vertex.getIconpath());
371             graphics.setFill(fill);
372         }
373     }

375     private void convertLayoutRectangleAttribute(final LayoutRectangleType rect, final GraphicsType graphics) throws ParseException {
376         final PositionType position = ANF_FACTORY.createPositionType();
377         position.setX(convertToBigDecimal(rect.getX()));
378         position.setY(convertToBigDecimal(rect.getY()));
379         graphics.getPosition().add(position);
380         final SizeType size = ANF_FACTORY.createSizeType();
381         size.setHeight(convertToBigDecimal(rect.getH()));
382         size.setWidth(convertToBigDecimal(rect.getW()));
383         graphics.setSize(size);
384     }
