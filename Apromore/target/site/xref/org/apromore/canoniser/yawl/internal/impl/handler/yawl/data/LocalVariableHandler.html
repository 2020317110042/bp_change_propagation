33  public class LocalVariableHandler extends VariableHandler<VariableFactsType> {
34  
35      /* (non-Javadoc)
36       * @see org.apromore.canoniser.yawl.internal.impl.handler.ConversionHandler#convert()
37       */
38      @Override
39      public void convert() throws CanoniserException {
40          SoftType softType = createSoftType(getObject().getName(), getObject().getType());
41          ExtensionUtils.addToExtensions(ExtensionUtils.marshalYAWLFragment(ExtensionUtils.LOCAL_VARIABLE, getObject(), VariableFactsType.class), softType);
42      }
