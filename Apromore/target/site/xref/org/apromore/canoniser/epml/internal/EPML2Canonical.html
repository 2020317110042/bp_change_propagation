132     /**
133      * The constructor receives the header then does the canonization process in order to allow the user to retrieve the produced process again into
134      * the canonical format. The user also will be able to retrieve the annotation element which stores the annotation data for the canonized modelass
135      * isolated from the process flow.
136      * 
137      * @param epml the header for an EPML (EPC Markup Language) which is file format for EPC diagrams.
138      * @throws org.apromore.canoniser.exception.CanoniserException
139      * @since 1.0
140      */

142         main(epml);

150     void main(TypeEPML epml) throws CanoniserException {
151         epml = removeFakes(epml);
152 
153         TypeAttribute att = new TypeAttribute();
154         att.setName("InitialFormat");
155         att.setValue("EPML 2.0");
156         cproc.getAttribute().add(att);
157 
158         if (epml.getDirectory() != null && epml.getDirectory().size() > 0) {
159             for (int i = 0; i < epml.getDirectory().size(); i++) {
160                 for (TExtensibleElements epc : epml.getDirectory().get(i).getEpcOrDirectory()) {
161                     if (epc instanceof TypeEPC) {
162                         TypeEPC typeEPC = (TypeEPC) epc;
163                         cproc.setName(typeEPC.getName());
164                         cproc.setUri(typeEPC.getEpcId().toString());
165                         annotations.setUri(cproc.getUri());
166                         // TODO what version to add here?
167                         cproc.setVersion("1.0");
168                         NetType net = new NetType();
169                         net.setName(typeEPC.getName());
170                         translateEpc(net, typeEPC);
171                         id_map.put(typeEPC.getEpcId(), String.valueOf(ids));
172                         net.setId(String.valueOf(ids++));
173                         cproc.getNet().add(net);
174                     }
175                 }
176                 for (TaskType task : subnet_list) {
177                     task.setSubnetId(id_map.get(new BigInteger(task.getSubnetId())));
178                 }
179                 subnet_list.clear();
180             }
181         } else {
182             // The EPML element doesn't have any directory
183             //throw new CanoniserException("EPML file is missing a 'directory' invalid!");
184             for (TypeEPC epc : epml.getEpcs()) {
185                 NetType net = new NetType();
186                 translateEpc(net, epc);
187                 id_map.put(epc.getEpcId(), String.valueOf(ids));
188                 net.setId(String.valueOf(ids++));
189                 cproc.getNet().add(net);
190             }
191             for (TaskType task : subnet_list) {
192                 task.setSubnetId(id_map.get(new BigInteger(task.getSubnetId())));
193             }
194             subnet_list.clear();
195         }
196 
197         // If there were any configuration variants, catalogue them
198         if (!variantMap.isEmpty()) {
199             Variants pcVariants = new Variants();
200             pcVariants.getVariant().addAll(variantMap.values());
201             TypeAttribute cpfAttribute = new TypeAttribute();
202             cpfAttribute.setName("bpmn_cpf/extensions");
203             cpfAttribute.setAny(pcVariants);
204             cproc.getAttribute().add(cpfAttribute);
205         }
206     }

208     /**
209      * This method for removing the fake functions and events in case the model has them.
210      * 
211      * @param epml
212      *            the header for an EPML
213      * @return epml the header for the EPML modelass after modification
214      * @throws CanoniserException
215      * @since 1.0
216      */
217     private TypeEPML removeFakes(final TypeEPML epml) throws CanoniserException {
218         List<TEpcElement> remove_list = new LinkedList<>();
219         List<TypeArc> arc_remove_list = new LinkedList<>();
220 
221         if (epml.getDirectory() != null && epml.getDirectory().size() > 0) {
222             for (int i = 0; i < epml.getDirectory().size(); i++) {
223                 for (TExtensibleElements epc : epml.getDirectory().get(i).getEpcOrDirectory()) {
224                     if (epc instanceof TypeEPC) {
225                         for (Object element : ((TypeEPC) epc).getEventAndFunctionAndRole()) {
226                             if (element instanceof TypeFunction || element instanceof TypeEvent) {
227                                 QName typeRef = new QName("typeRef");
228                                 String str = ((TEpcElement) element).getOtherAttributes().get(typeRef);
229                                 if (str != null && str.equals("fake")) {
230                                     remove_list.add((TEpcElement) element);
231                                 }
232                             }
233                         }
234                         for (TEpcElement element : remove_list) {
235                             for (Object arc : ((TypeEPC) epc).getEventAndFunctionAndRole()) {
236                                 if (arc instanceof TypeArc) {
237                                     if (((TypeArc) arc).getFlow() != null) {
238                                         if (((TypeArc) arc).getFlow().getSource().equals(element.getId())) {
239                                             for (Object arc2 : ((TypeEPC) epc).getEventAndFunctionAndRole()) {
240                                                 if (arc2 instanceof TypeArc) {
241                                                     if (((TypeArc) arc2).getFlow().getTarget().equals(element.getId())) {
242                                                         ((TypeArc) arc2).getFlow().setTarget(((TypeArc) arc).getFlow().getTarget());
243                                                     }
244                                                 }
245                                             }
246                                             arc_remove_list.add((TypeArc) arc);
247                                         }
248                                     }
249                                 }
250                             }
251                             for (TypeArc arc : arc_remove_list) {
252                                 ((TypeEPC) epc).getEventAndFunctionAndRole().remove(arc);
253                             }
254                             ((TypeEPC) epc).getEventAndFunctionAndRole().remove(element);
255                         }
256                     }
257                 }
258             }
259         } else {
260             //throw new CanoniserException("Invalid EPC no 'directory'!");
261             for (TypeEPC epc : epml.getEpcs()) {
262                 for (Object element : epc.getEventAndFunctionAndRole()) {
263                     if (element instanceof TypeFunction || element instanceof TypeEvent) {
264                         QName typeRef = new QName("typeRef");
265                         if (((TEpcElement) element).getOtherAttributes().get(typeRef).equals("fake")) {
266                             remove_list.add((TEpcElement) element);
267                         }
268                     }
269                 }
270                 for (TEpcElement element : remove_list) {
271                     for (Object arc : epc.getEventAndFunctionAndRole()) {
272                         if (arc instanceof TypeArc) {
273                             if (((TypeArc) arc).getFlow() != null) {
274                                 if (((TypeArc) arc).getFlow().getSource().equals(element.getId())) {
275                                     for (Object arc2 : epc.getEventAndFunctionAndRole()) {
276                                         if (arc2 instanceof TypeArc) {
277                                             if (((TypeArc) arc2).getFlow().getTarget().equals(element.getId())) {
278                                                 ((TypeArc) arc2).getFlow().setTarget(((TypeArc) arc).getFlow().getTarget());
279                                             }
280                                         }
281                                     }
282                                     arc_remove_list.add((TypeArc) arc);
283                                 }
284                             }
285                         }
286                     }
287                     for (TypeArc arc : arc_remove_list) {
288                         epc.getEventAndFunctionAndRole().remove(arc);
289                     }
290                     epc.getEventAndFunctionAndRole().remove(element);
291                 }
292             }
293         }
294 
295         return epml;
296     }

298     @SuppressWarnings("unchecked")
299     private void translateEpc(final NetType net, final TypeEPC epc) throws CanoniserException {
300         Map<String, String> role_names = new HashMap<>();
301 
302         for (Object obj : epc.getEventAndFunctionAndRole()) {
303             if (obj instanceof JAXBElement) {
304                 JAXBElement<?> element = (JAXBElement<?>) obj;
305                 if (element.getValue() instanceof TypeEvent) {
306                     translateEvent(net, (TypeEvent) element.getValue());
307                     addNodeAnnotations(element.getValue());
308                 } else if (element.getValue() instanceof TypeFunction) {
309                     translateFunction(net, ((JAXBElement<TypeFunction>) obj).getValue());
310                     addNodeAnnotations(element.getValue());
311                 } else if (element.getValue() instanceof TypeAND) {
312                     id_map.put(((TEpcElement) element.getValue()).getId(), String.valueOf(ids));
313                     addNodeAnnotations(element.getValue());
314                     ((TEpcElement) element.getValue()).setId(BigInteger.valueOf(ids++));
315                     and_list.add((TypeAND) element.getValue());
316                 } else if (element.getValue() instanceof TypeOR) {
317                     id_map.put(((TEpcElement) element.getValue()).getId(), String.valueOf(ids));
318                     addNodeAnnotations(element.getValue());
319                     ((TEpcElement) element.getValue()).setId(BigInteger.valueOf(ids++));
320                     or_list.add((TypeOR) element.getValue());
321                 } else if (element.getValue() instanceof TypeXOR) {
322                     id_map.put(((TEpcElement) element.getValue()).getId(), String.valueOf(ids));
323                     addNodeAnnotations(element.getValue());
324                     ((TEpcElement) element.getValue()).setId(BigInteger.valueOf(ids++));
325                     xor_list.add((TypeXOR) element.getValue());
326                 } else if (element.getValue() instanceof TypeRole) {
327                     if (!role_names.containsKey(((TypeRole) element.getValue()).getName())) {
328                         translateRole((TypeRole) element.getValue());
329                         addNodeAnnotations(element.getValue());
330                         role_names.put(((TypeRole) element.getValue()).getName(), String.valueOf(ids - 1));
331                     } else {
332                         id_map.put(((TypeRole) element.getValue()).getId(), role_names.get(((TypeRole) element.getValue()).getName()));
333                     }
334                 } else if (element.getValue() instanceof TypeObject) {
335                     translateObject(net, (TypeObject) element.getValue());
336                     addNodeAnnotations(element.getValue());
337                 } else if (element.getValue() instanceof TypeRANGE) {
338                     range_ids.add(((TypeRANGE) element.getValue()).getId());
339                 } else if (element.getValue() instanceof TypeProcessInterface) {
340                     translatePI(net, (TypeProcessInterface) element.getValue());
341                     addNodeAnnotations(element.getValue());
342                 }
343             }
344         }
345 
346         // Convert Arcs after all Elements are converted, because we need the IDs of all Elements
347         for (Object obj : epc.getEventAndFunctionAndRole()) {
348             if (obj instanceof JAXBElement) {
349                 JAXBElement<?> element = (JAXBElement<?>) obj;
350                 if (element.getValue() instanceof TypeArc) {
351                     TypeArc arc = (TypeArc) element.getValue();
352                     if (arc.getFlow() != null) {
353                         if (range_ids.contains(arc.getFlow().getSource()) || range_ids.contains(arc.getFlow().getTarget())) {
354                             range_flow.add(arc);
355                         } else {
356                             translateArc(net, arc);
357                         }
358                         addEdgeAnnotation(arc);
359                     } else if (arc.getRelation() != null) {
360                         if (range_ids.contains(arc.getRelation().getSource()) || range_ids.contains(arc.getRelation().getTarget())) {
361                             range_relation.add(arc);
362                         } else {
363                             translateArc(net, arc);
364                         }
365                         addEdgeAnnotation(arc);
366                     }
367                 }
368             }
369         }
370 
371         for (TypeArc arc : range_flow) {
372             if (range_ids.contains(arc.getFlow().getSource())) {
373                 for (TypeArc arc2 : range_flow) {
374                     if (range_ids.contains(arc2.getFlow().getTarget())) {
375                         arc.getFlow().setSource(arc2.getFlow().getSource());
376                         translateArc(net, arc);
377                         addEdgeAnnotation(arc);
378                     }
379                 }
380             }
381         }
382 
383         for (TypeArc arc : range_relation) {
384             if (range_ids.contains(arc.getRelation().getSource())) {
385                 for (TypeArc arc2 : range_relation) {
386                     if (range_ids.contains(arc2.getRelation().getTarget())) {
387                         arc.getRelation().setSource(arc2.getRelation().getSource());
388                         translateArc(net, arc);
389                         addEdgeAnnotation(arc);
390                     }
391                 }
392             }
393         }
394 
395         int counter;
396         for (TypeAND and : and_list) {
397             counter = 0;
398             BigInteger n = and.getId();
399             for (String s : flow_source_id_list) {
400                 if (n.toString().equals(s)) {
401                     counter++;
402                 }
403             }
404             if (counter <= 1) {
405                 ANDJoinType andJ = new ANDJoinType();
406                 andJ.setId(String.valueOf(and.getId()));
407                 andJ.setName(and.getName());
408 
409                 for (de.epml.TypeAttribute epmlAttribute: and.getAttribute()) {
410                     if ("configurationAnnotation".equals(epmlAttribute.getTypeRef())) {
411                         addConfigurationAnnotationAttribute2(epmlAttribute, andJ.getAttribute());
412                     } else {
413                         addUnrecognizedAttribute(epmlAttribute, andJ.getAttribute());
414                     }
415                 }
416 
417                 net.getNode().add(andJ);
418             } else {
419                 ANDSplitType andS = new ANDSplitType();
420                 andS.setId(String.valueOf(and.getId()));
421                 andS.setName(and.getName());
422 
423                 for (de.epml.TypeAttribute epmlAttribute: and.getAttribute()) {
424                     if ("configurationAnnotation".equals(epmlAttribute.getTypeRef())) {
425                         addConfigurationAnnotationAttribute2(epmlAttribute, andS.getAttribute());
426                     } else {
427                         addUnrecognizedAttribute(epmlAttribute, andS.getAttribute());
428                     }
429                 }
430 
431                 net.getNode().add(andS);
432             }
433         }
434         and_list.clear();
435 
436         for (TypeOR or : or_list) {
437             counter = 0;
438             BigInteger n = or.getId();
439             for (String s : flow_source_id_list) {
440                 if (n.toString().equals(s)) {
441                     counter++;
442                 }
443             }
444             if (counter <= 1) {
445                 ORJoinType orJ = new ORJoinType();
446                 orJ.setId(String.valueOf(or.getId()));
447                 orJ.setName(or.getName());
448 
449                 for (de.epml.TypeAttribute epmlAttribute: or.getAttribute()) {
450                     if ("configurationAnnotation".equals(epmlAttribute.getTypeRef())) {
451                         addConfigurationAnnotationAttribute2(epmlAttribute, orJ.getAttribute());
452                     } else {
453                         addUnrecognizedAttribute(epmlAttribute, orJ.getAttribute());
454                     }
455                 }
456 
457                 net.getNode().add(orJ);
458             } else {
459                 ORSplitType orS = new ORSplitType();
460                 orS.setId(String.valueOf(or.getId()));
461                 orS.setName(or.getName());
462 
463                 for (de.epml.TypeAttribute epmlAttribute: or.getAttribute()) {
464                     if ("configurationAnnotation".equals(epmlAttribute.getTypeRef())) {
465                         addConfigurationAnnotationAttribute2(epmlAttribute, orS.getAttribute());
466                     } else {
467                         addUnrecognizedAttribute(epmlAttribute, orS.getAttribute());
468                     }
469                 }
470 
471                 net.getNode().add(orS);
472                 //processUnrequiredEvents(net, or.getId()); // after creating the split node ,, delete the event
473             }
474         }
475         or_list.clear();
476 
477         for (TypeXOR xor : xor_list) {
478             counter = 0;
479             BigInteger n = xor.getId();
480             for (String s : flow_source_id_list) {
481                 if (n.toString().equals(s)) {
482                     counter++;
483                 }
484             }
485 
486             if (counter <= 1) {
487                 XORJoinType xorJ = new XORJoinType();
488                 xorJ.setId(String.valueOf(xor.getId()));
489                 xorJ.setName(xor.getName());
490 
491                 if(xor.getConfigurableConnector()!=null) {
492                     xorJ.setConfigurable(true);
493                 }
494 
495                 for (de.epml.TypeAttribute epmlAttribute: xor.getAttribute()) {
496                     if ("configurationAnnotation".equals(epmlAttribute.getTypeRef())) {
497                         addConfigurationAnnotationAttribute2(epmlAttribute, xorJ.getAttribute());
498                     } else {
499                         addUnrecognizedAttribute(epmlAttribute, xorJ.getAttribute());
500                     }
501                 }
502 
503                 net.getNode().add(xorJ);
504             } else {
505                 XORSplitType xorS = new XORSplitType();
506                 xorS.setId(String.valueOf(xor.getId()));
507                 xorS.setName(xor.getName());
508                 if(xor.getConfigurableConnector()!=null) {
509                     xorS.setConfigurable(true);
510 
511                 }
512 
513                 for (de.epml.TypeAttribute epmlAttribute: xor.getAttribute()) {
514                     if ("configurationAnnotation".equals(epmlAttribute.getTypeRef())) {
515                         addConfigurationAnnotationAttribute2(epmlAttribute, xorS.getAttribute());
516                     } else {
517                         addUnrecognizedAttribute(epmlAttribute, xorS.getAttribute());
518                     }
519                 }
520 
521                 net.getNode().add(xorS);
522                 //processUnrequiredEvents(net, xor.getId());
523             }
524         }
525         xor_list.clear();
526     }

528     private void addEdgeAnnotation(final TypeArc arc) throws CanoniserException {
529         LineType line = new LineType();
530         GraphicsType graph = new GraphicsType();
531         FontType font = new FontType();
532 
533         if (arc.getGraphics() != null) {
534             if (id_map.get(arc.getId()) == null && range_relation.contains(arc)) {
535                 Logger.getAnonymousLogger().info("Arc " + arc.getId() + " discarded");
536                 return;
537             }
538             graph.setCpfId(id_map.get(arc.getId()));
539             graph.setId(graph.getCpfId());
540             if (arc.getGraphics().size() > 0) {
541                 if (arc.getGraphics().get(0) != null) {
542                     if (arc.getGraphics().get(0).getFont() != null) {
543                         font.setColor(arc.getGraphics().get(0).getFont().getColor());
544                         font.setDecoration(arc.getGraphics().get(0).getFont().getDecoration());
545                         font.setFamily(arc.getGraphics().get(0).getFont().getFamily());
546                         font.setHorizontalAlign(arc.getGraphics().get(0).getFont().getHorizontalAlign());
547                         font.setRotation(arc.getGraphics().get(0).getFont().getRotation());
548                         font.setSize(arc.getGraphics().get(0).getFont().getSize());
549                         font.setStyle(arc.getGraphics().get(0).getFont().getStyle());
550                         font.setVerticalAlign(arc.getGraphics().get(0).getFont().getVerticalAlign());
551                         font.setWeight(arc.getGraphics().get(0).getFont().getWeight());
552                         graph.setFont(font);
553                     }
554                     if (arc.getGraphics().get(0).getLine() != null) {
555                         line.setColor(arc.getGraphics().get(0).getLine().getColor());
556                         line.setShape(arc.getGraphics().get(0).getLine().getShape());
557                         line.setStyle(arc.getGraphics().get(0).getLine().getStyle());
558                         line.setWidth(arc.getGraphics().get(0).getLine().getWidth());
559                         graph.setLine(line);
560                     }
561 
562                     for (TypeMove2 mov2 : arc.getGraphics().get(0).getPosition()) {
563                         PositionType pos = new PositionType();
564                         pos.setX(mov2.getX());
565                         pos.setY(mov2.getY());
566                         graph.getPosition().add(pos);
567                     }
568                     annotations.getAnnotation().add(graph);
569                 }
570             }
571         }
572     }

574     private void addNodeAnnotations(final Object obj) {
575         GraphicsType graphT = new GraphicsType();
576         LineType line = new LineType();
577         FillType fill = new FillType();
578         PositionType pos = new PositionType();
579         SizeType size = new SizeType();
580         FontType font = new FontType();
581         String cpfId;
582 
583         TEpcElement element = (TEpcElement) obj;
584         cpfId = id_map.get(element.getId());
585 
586         if (element.getGraphics() != null) {
587             if (element.getGraphics().getFill() != null) {
588                 fill.setColor(element.getGraphics().getFill().getColor());
589                 fill.setGradientColor(element.getGraphics().getFill().getGradientColor());
590                 fill.setGradientRotation(element.getGraphics().getFill().getGradientRotation());
591                 fill.setImage(element.getGraphics().getFill().getImage());
592                 graphT.setFill(fill);
593             }
594 
595             if (element.getGraphics().getPosition() != null) {
596                 size.setHeight(element.getGraphics().getPosition().getHeight());
597                 size.setWidth(element.getGraphics().getPosition().getWidth());
598                 graphT.setSize(size);
599                 pos.setX(element.getGraphics().getPosition().getX());
600                 pos.setY(element.getGraphics().getPosition().getY());
601                 graphT.getPosition().add(pos);
602             }
603 
604             if (element.getGraphics().getLine() != null) {
605                 line.setColor(element.getGraphics().getLine().getColor());

633     private void processUnrequiredEvents(final NetType net, final BigInteger id) throws CanoniserException {
634         List<EdgeType> edge_remove_list = new LinkedList<>();
635         List<NodeType> node_remove_list = new LinkedList<>();
636         String event_id;
637         boolean found = false;
638         for (EdgeType edge : net.getEdge()) {
639             if (edge.getSourceId() != null) {
640                 // Test if Edge is from a XOR-split to an Event
641                 if (edge.getSourceId().equals(id.toString()) && event_ids.contains(edge.getTargetId())) {
642                     // Test if the target Event is an Exit Event
643                     if (!isExitNode(edge.getTargetId(), net)) {
644                         event_id = edge.getTargetId();
645                         for (EdgeType edge2 : net.getEdge()) {
646                             if (edge2.getSourceId() != null && edge2.getSourceId().equals(event_id)) {
647                                 edge.setTargetId(edge2.getTargetId());
648                                 edge_remove_list.add(edge2);
649                                 found = true;
650                             }
651                         }
652                         // Delete the unrequired event and set its name as a condition for the edge
653                         for (NodeType node : net.getNode()) {
654                             if (node.getId().equals(event_id)) {
655                                 if (found) {
656                                     if (node.getName() != null) {
657                                         edge.setConditionExpr(convertStringExpression(node.getName()));   
658                                     }
659                                     node_remove_list.add(node);
660                                 } else {
661                                     if (node.getName() != null) {
662                                         edge.setConditionExpr(convertStringExpression(node.getName()));   
663                                     }                                    
664                                     node.setName("");
665                                 }
666                             }
667                         }
668                     } else {
669                         NodeType node = findNodeById(edge.getTargetId(), net);
670                         if (node.getName() != null) {
671                             edge.setConditionExpr(convertStringExpression(node.getName()));
672                         }
673                     }
674                 }
675             }
676         }
677 
678         for (EdgeType edge : edge_remove_list) {
679             net.getEdge().remove(edge);
680         }
681         edge_remove_list.clear();
682         for (NodeType node : node_remove_list) {
683             net.getNode().remove(node);
684         }
685         node_remove_list.clear();
686 
687     }

766     /* For an EPML split or join, each semicolon-terminated variant in the "configurationAnnotation" is a colon-separated variant:gatewayType pair. */
767     private void addConfigurationAnnotationAttribute2(de.epml.TypeAttribute epmlAttribute, List<org.apromore.cpf.TypeAttribute> cpfAttributeList) {
768         ConfigurationAnnotation pcConfigurationAnnotation = new ConfigurationAnnotation();
769         String[] value = epmlAttribute.getValue().split(";");
770         for (int i = 0; i < value.length; i++) {
771             if (!value[i].isEmpty()) {
772                 String[] field = value[i].split(":", 2);
773                 ConfigurationAnnotation.Configuration pcConfiguration = new ConfigurationAnnotation.Configuration();
774                 pcConfiguration.setVariantRef(findVariant(field[0]));
775                 switch (field[1]) {
776                 case "and": pcConfiguration.setType(TGatewayType.PARALLEL);             break;
777                 case "or":  pcConfiguration.setType(TGatewayType.INCLUSIVE);            break;
778                 case "xor": pcConfiguration.setType(TGatewayType.DATA_BASED_EXCLUSIVE); break;
779                 }
780                 pcConfigurationAnnotation.getConfiguration().add(pcConfiguration);
781             }
782         }
783         org.apromore.cpf.TypeAttribute cpfAttribute = new org.apromore.cpf.TypeAttribute();
784         cpfAttribute.setName("bpmn_cpf/extensions");
785         cpfAttribute.setAny(pcConfigurationAnnotation);
786         cpfAttributeList.add(cpfAttribute);
787     }
