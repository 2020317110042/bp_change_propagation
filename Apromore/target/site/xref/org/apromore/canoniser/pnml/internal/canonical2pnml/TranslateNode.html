56      public void translateTask(TaskType task) {
57          TransitionToolspecificType trantool = new TransitionToolspecificType();
58          TransitionToolspecificType ttt = new TransitionToolspecificType();
59          TransitionType tran = new TransitionType();
60          boolean isSub = false;
61          data.put_id_map(task.getId(), String.valueOf(ids));
62          tran.setId(String.valueOf(ids++));
63          tran.setGraphics(newGraphicsNodeType(dummyPosition(), transitionDefaultDimension()));
64  
65          if (task.getName() != null) {
66              NodeNameType test = new NodeNameType();
67              test.setText(task.getName());
68              tran.setName(test);
69          }
70  
71          if (task.getSubnetId() != null) {
72              isSub = true;
73              data.setSubnet(tran);
74          }
75  
76          if (data.get_triggermap().containsKey(tran.getName().getText())) {
77              trantool.setTool("WoPeD");
78              trantool.setVersion("1.0");
79              trantool.setTrigger((TransitionToolspecificType.Trigger) data.get_triggermap_value(tran.getName().getText()));
80              if (isSub) {
81                  trantool.setSubprocess(true);
82              }
83              tran.getToolspecific().add(trantool);
84          }
85  
86          if (task.getResourceTypeRef().size() > 0) {
87              ttt.setTool("WoPeD");
88              ttt.setVersion("1.0");
89              TriggerType tt = new TriggerType();
90              GraphicsSimpleType gt = new GraphicsSimpleType();
91              DimensionType dt = new DimensionType();
92              dt.setX(BigDecimal.valueOf(Long.valueOf(24)));
93              dt.setY(BigDecimal.valueOf(Long.valueOf(22)));
94              gt.setDimension(dt);
95              tt.setId("");
96              tt.setType(200);
97              tt.setGraphics(gt);
98              data.put_triggermap(task.getName(), tt);
99              ttt.setTrigger((TransitionToolspecificType.Trigger) tt);
100             for (Object resource : task.getResourceTypeRef()) {
101                 if (resource instanceof ResourceTypeRefType) {
102                     if (((ResourceTypeRefType) resource).getResourceTypeId() != null) {
103                         if (data.get_resourcemap().containsKey(
104                                 String.valueOf(((ResourceTypeRefType) resource).getResourceTypeId()))) {
105                             ResourceTypeType res = data
106                                     .get_resourcemap_value(String
107                                             .valueOf(((ResourceTypeRefType) resource).getResourceTypeId()));
108                             String[] split = res.getName().split("-");
109                             split[0] = split[0].substring(0);
110                             split[1] = split[1].substring(0);
111 
112                             OrganizationUnitType ui = new OrganizationUnitType();
113                             ui.setName(split[0]);
114                             RoleType ro = new RoleType();
115                             ro.setName(split[1]);
116                             TransitionResourceType tres = new TransitionResourceType();
117                             tres.setOrganizationalUnitName(ui);
118                             tres.setRoleName(ro);
119                             GraphicsSimpleType gtres = new GraphicsSimpleType();
120                             DimensionType dtres = new DimensionType();
121                             dtres.setX(BigDecimal.valueOf(Long.valueOf(60)));
122                             dtres.setY(BigDecimal.valueOf(Long.valueOf(22)));
123                             gtres.setDimension(dt);
124                             tres.setGraphics(gtres);
125                             ttt.setTransitionResource((TransitionToolspecificType.TransitionResource) tres);
126                             data.put_resourcepositionmap(task.getName(), tres);
127                             if (!data.getunit().contains(split[0])) {
128                                 data.addunit(split[0]);
129                             }
130                             if (!data.getroles().contains(split[1])) {
131                                 data.addroles(split[1]);
132                             }
133                         }
134                     }
135                 }
136             }
137 
138             if (isSub) {
139                 trantool.setSubprocess(true);
140             }
141             tran.getToolspecific().add(ttt);
142         }
143 
144         if (trantool.getTool() == null && ttt.getTool() == null && isSub) {
145             TransitionToolspecificType sub = new TransitionToolspecificType();
146             sub.setTool("WoPeD");
147             sub.setVersion("1.0");
148             sub.setSubprocess(true);
149             tran.getToolspecific().add(sub);
150         }
151 
152         data.getNet().getTransition().add(tran);
153         data.put_pnmlRefMap(tran.getId(), tran);
154 
155         data.put_originalid_map(BigInteger.valueOf(Long.valueOf(tran.getId())), task.getOriginalID());
156 
157         if (data.isCpfTaskPnmlTransition()) {
158 
159             // Tell TranslateArc where to attach incoming and outgoing arcs
160             data.getStartNodeMap().put(task, tran);
161             data.getEndNodeMap().put(task, tran);
162             
163         } else {
164             // The transition we've created only starts the task; we also need
165             // to add a place to represent that the task is running and a second
166             // transition to represent the task ending.
167             //
168             // [tran] --> (running) --> [end]
169 
170             // Create the place for the state in which the task is running
171             PlaceType running = new PlaceType();
172             running.setId(String.valueOf(ids++));
173             running.setGraphics(newGraphicsNodeType(dummyPosition(), placeDefaultDimension()));
174             data.getNet().getPlace().add(running);
175 
176             // Create the arc from the start transition to the running place
177             ArcType startToRunning = new ArcType();
178             startToRunning.setId(String.valueOf(ids++));
179             startToRunning.setSource(tran);
180             startToRunning.setTarget(running);
181             data.getNet().getArc().add(startToRunning);
182 
183             // Create the transition that stops the task running
184             TransitionType end = new TransitionType();
185             end.setId(String.valueOf(ids++));
186             end.setName(running.getName());
187             end.setGraphics(newGraphicsNodeType(dummyPosition(), transitionDefaultDimension()));
188             data.getNet().getTransition().add(end);
189 
190             // Create the arc from the running place to the end transition
191             ArcType runningToEnd = new ArcType();
192             runningToEnd.setId(String.valueOf(ids++));
193             runningToEnd.setSource(running);
194             runningToEnd.setTarget(end);
195             data.getNet().getArc().add(runningToEnd);
196 
197             // Name the nodes (English naming convention)
198             String taskName = task.getName();
199             if (taskName != null) {
200                 // Start transition named "<Task> start"
201                 NodeNameType name = new NodeNameType();
202                 name.setText(taskName + "_start");
203                 tran.setName(name);
204 
205                 // Running place named "<Task>"
206                 name = new NodeNameType();
207                 name.setText(taskName);
208                 running.setName(name);
209 
210                 // End transition named "<Task> end"
211                 name = new NodeNameType();
212                 name.setText(taskName + "_end");
213                 end.setName(name);
214             }
215 
216             // Tell TranslateArc where to attach incoming and outgoing arcs
217             data.getStartNodeMap().put(task, tran);
218             data.getRunningPlaceMap().put(task, running);
219             data.getEndNodeMap().put(task, end);
220         }
221     }

270     public void translateEvent(NodeType node) {
271         if (node instanceof MessageType || node instanceof TimerType) {
272             TransitionType tran = new TransitionType();
273             data.put_id_map(node.getId(), String.valueOf(ids));
274             tran.setId(String.valueOf(ids++));
275             tran.setGraphics(newGraphicsNodeType(dummyPosition(), transitionDefaultDimension()));
276 
277             if (node.getName() != null) {
278                 NodeNameType test = new NodeNameType();
279                 test.setText(node.getName());
280                 tran.setName(test);
281             }
282 
283             TransitionToolspecificType ttt = new TransitionToolspecificType();
284             ttt.setTool("WoPeD");
285             ttt.setVersion("1.0");
286             TransitionToolspecificType.Trigger tt = new TransitionToolspecificType.Trigger();
287             GraphicsSimpleType gt = new GraphicsSimpleType();
288             DimensionType dt = new DimensionType();
289             dt.setX(BigDecimal.valueOf(Long.valueOf(24)));
290             dt.setY(BigDecimal.valueOf(Long.valueOf(22)));
291             gt.setDimension(dt);
292             tt.setId("");
293             if (node instanceof MessageType) {
294                 tt.setType(201);
295             } else {
296                 assert node instanceof TimerType;
297                 tt.setType(202);
298             }
299             tt.setGraphics(gt);
300             data.put_triggermap(node.getName(), tt);
301             ttt.setTrigger(tt);
302             tran.getToolspecific().add(ttt);
303 
304             // Create a place to wait for the timer to expire
305             String waitingPlaceNameText = null;
306             if (node instanceof MessageType) {
307                 waitingPlaceNameText = "msg_wait";
308             } else {
309                 assert node instanceof TimerType;
310                 waitingPlaceNameText = "time_wait";
311             }
312             assert waitingPlaceNameText != null;
313             if (node.getName() != null) {
314                 waitingPlaceNameText = node.getName() + "_" + waitingPlaceNameText;
315             }
316 
317             NodeNameType waitingPlaceName = new NodeNameType();
318             waitingPlaceName.setText(waitingPlaceNameText);
319 
320             PlaceType waitingPlace = new PlaceType();
321             waitingPlace.setId(String.valueOf(ids++));
322             waitingPlace.setName(waitingPlaceName);
323             waitingPlace.setGraphics(newGraphicsNodeType(dummyPosition(), placeDefaultDimension()));
324             data.getNet().getPlace().add(waitingPlace);
325 
326             // Create arc from waitingPlace to tran
327             ArcType arc = new ArcType();
328             arc.setId(String.valueOf(ids++));
329             arc.setSource(waitingPlace);
330             arc.setTarget(tran);
331             data.getNet().getArc().add(arc);
332 
333             data.getNet().getTransition().add(tran);
334             data.put_pnmlRefMap(tran.getId(), tran);
335             data.getStartNodeMap().put(node, waitingPlace);
336             data.getRunningPlaceMap().put(node, waitingPlace);
337             data.getEndNodeMap().put(node, tran);
338             data.put_originalid_map(BigInteger.valueOf(Long.valueOf(tran.getId())), node.getOriginalID());
339 
340         } else {
341             PlaceType place = new PlaceType();
342             data.put_id_map(node.getId(), String.valueOf(ids));
343             place.setId(String.valueOf(ids++));
344             place.setGraphics(newGraphicsNodeType(dummyPosition(), placeDefaultDimension()));
345 
346             if (node.getName() != null) {
347                 NodeNameType test = new NodeNameType();
348                 test.setText(node.getName());
349                 place.setName(test);
350             }
351 
352             data.getNet().getPlace().add(place);
353             data.put_pnmlRefMap(place.getId(), place);
354 
355             data.getStartNodeMap().put(node, place);
356             data.getRunningPlaceMap().put(node, place);
357             data.getEndNodeMap().put(node, place);
358 
359             data.put_originalid_map(BigInteger.valueOf(Long.valueOf(place.getId())), node.getOriginalID());
360         }
361     }
