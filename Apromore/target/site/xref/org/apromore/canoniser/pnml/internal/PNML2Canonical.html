113     void main(final PnmlType pnml,
114               final boolean  isCpfTaskPnmlTransition,
115               final boolean  isCpfEdgePnmlPlace) throws CanoniserException {
116 
117         CanonicalProcessType cpf = CpfObjectFactory.getInstance().createCanonicalProcessType();
118         cpf.setName("dummy");
119         cpf.setUri("");
120         cpf.setVersion("1.0");
121         data.setCanonicalProcess(cpf);
122 
123         if (pnml.getNet() != null && pnml.getNet().size() > 0) {
124             for (org.apromore.pnml.NetType pnet : pnml.getNet()) {
125                 NetType net = CpfObjectFactory.getInstance().createNetType();
126                 data.setNet(net);
127                 tpn.setValues(data, ids);
128                 tpn.translatePetriNet(pnet);
129                 ids = tpn.getIds();
130                 data.put_id_map(pnet.getId(), String.valueOf(ids));
131                 data.setRootId(ids);
132                 data.getNet().setId(String.valueOf(ids++));
133                 data.getCanonicalProcess().getNet().add(data.getNet());
134             }
135         }
136     }
