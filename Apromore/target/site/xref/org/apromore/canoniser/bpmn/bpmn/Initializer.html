153     /**
154      * Populates {@link #cancelNodeIdMap} so that {@link #findAttachedTaskId} will work, and removes {@link CpfANDSplitType}s
155      * which are to be replaced by boundary events.
156      */
157     private void processBoundaryEvents() {
158 
159         // Populate cancelNodeIds
160         cpf.accept(new TraversingVisitor(new DepthFirstTraverserImpl(), new BaseVisitor() {
161             @Override public void visit(final TaskType task) {
162                 for (CancellationRefType cancellationRef : task.getCancelNodeId()) {
163                     assert !cancelNodeIdMap.containsKey(cancellationRef.getRefId());
164                     cancelNodeIdMap.put(cancellationRef.getRefId(), task.getId());
165                 }
166             }
167         }));
168 
169         // Identify AND splits which can be converted into boundary events
170         final Map<CpfANDSplitType, CpfTaskType> splits = new HashMap<CpfANDSplitType, CpfTaskType>();
171         cpf.accept(new TraversingVisitor(new DepthFirstTraverserImpl(), new BaseVisitor() {
172             @Override public void visit(final ANDSplitType split) {
173                 CpfANDSplitType cpfSplit = (CpfANDSplitType) split;
174                 CpfTaskType task = null;
175 
176                 // Check that there's exactly one task and at least one event targeted by the gateway
177                 int taskCount  = 0;
178                 int eventCount = 0;
179                 for (CpfEdgeType edge : cpfSplit.getOutgoingEdges()) {
180                     CpfNodeType node = (CpfNodeType) cpf.getElement(edge.getTargetId());
181 
182                     if (node instanceof CpfEventType) {
183                         eventCount++;
184                     } else if (node instanceof CpfTaskType) {
185                         task = (CpfTaskType) node;
186                         taskCount++;
187                     }
188                 }
189 
190                 // Check that the task cancels all the events
191                 if (eventCount > 0 && taskCount == 1) {
192                     assert task != null;
193 
194                     // Determine the set of candidate boundary events
195                     Set<String> cancelledEdgeIdSet = new HashSet<String>();
196                     for (CpfEdgeType edge : cpfSplit.getOutgoingEdges()) {
197                         CpfNodeType node = (CpfNodeType) cpf.getElement(edge.getTargetId());
198                         if (node instanceof CpfEventType) {
199                             cancelledEdgeIdSet.add(edge.getId());
200                         }
201                     }
202 
203                     // Determine the task's cancellation set
204                     Set<String> taskCancelledIdSet = new HashSet<String>();
205                     for (CancellationRefType cancellationRef : task.getCancelNodeId()) {
206                         taskCancelledIdSet.add(cancellationRef.getRefId());
207                     }
208 
209                     // If the task cancels all the candidate boundary events, attempt rewriting
210                     if (taskCancelledIdSet.containsAll(cancelledEdgeIdSet)) {
211                         splits.put(cpfSplit, task);
212                         java.util.logging.Logger.getAnonymousLogger().info(cpfSplit.getId() + " can be made into a boundary event");
213                     }
214                 }
215             }
216         }));
217 
218         // Remove the identified AND splits
219         for (CpfANDSplitType split : splits.keySet()) {
220 
221             // Retarget the AND split's incoming edge to the attached task
222             for (CpfEdgeType in : split.getIncomingEdges()) {
223                 CpfTaskType task = splits.get(split);
224                 in.setTargetId(task.getId());
225                 task.getIncomingEdges().add(in);
226             }
227 
228             // Remove the AND split and its outgoing edges
229             CpfNetType parent = findParent(split);
230             parent.getNode().remove(split);
231             uncancel(split, parent.getNode());
232 
233             parent.getEdge().removeAll(split.getOutgoingEdges());
234             for (CpfEdgeType edge : split.getOutgoingEdges()) {
235                 CpfNodeType node = (CpfNodeType) cpf.getElement(edge.getTargetId());
236                 node.getIncomingEdges().remove(edge);
237                 uncancel(edge, parent.getNode());
238             }
239         }
240     }

242     private void uncancel(final CpfEdgeType edge, final List<NodeType> nodes) {
243         for (NodeType node2 : nodes) {
244             if (node2 instanceof WorkType) {
245                 Iterator<CancellationRefType> i = ((WorkType) node2).getCancelEdgeId().iterator();
246                 while (i.hasNext()) {
247                     CancellationRefType cancellation = i.next();
248                     if (edge.getId().equals(cancellation.getRefId())) {
249                         i.remove();
250                     }
251                 }   
252             }
253         }
254     }
255 
256     private void uncancel(final CpfNodeType node, final List<NodeType> nodes) {
257         for (NodeType node2 : nodes) {
258             if (node2 instanceof WorkType) {
259                 Iterator<CancellationRefType> i = ((WorkType) node2).getCancelNodeId().iterator();
260                 while (i.hasNext()) {
261                     CancellationRefType cancellation = i.next();
262                     if (node.getId().equals(cancellation.getRefId())) {
263                         i.remove();
264                     }
265                 }   
266             }
267         }
268     }

271      * Find a {@link NetType} given its identifier.
272      *
273      * @param id  the identifier attribute of the sought net
274      * @return the net in <code>cpf</code> with the identifier <code>id</code>
275      * @throws CanoniserException if <code>id</code> doesn't identify a net in <code>cpf</code>
276      */
277     public CpfNetType findNet(final String id) throws CanoniserException {
278 
279         for (final NetType net : cpf.getNet()) {
280             if (id.equals(net.getId())) {
281                 return (CpfNetType) net;
282             }
283         }
284 
285         // Failed to find the desired name
286         throw new CanoniserException("CPF model has no net with id " + id);
287     }

428     void populateFlowNode(final TFlowNode flowNode, final CpfNodeType cpfNode) throws CanoniserException {
429         populateFlowElement(flowNode, cpfNode);
430 
431         // Handle incoming and outgoing
432         if (false) {  // TODO - allow this to be configured by the BPMN canoniser PluginRequest
433             defer(new Initialization() {
434                 public void initialize() throws CanoniserException {
435                     for (final EdgeType edge : cpfNode.getIncomingEdges()) {
436                         flowNode.getIncoming().add(new QName(targetNamespace, findElement(edge.getId()).getId()));
437                     }
438                     for (final EdgeType edge : cpfNode.getOutgoingEdges()) {
439                         flowNode.getOutgoing().add(new QName(targetNamespace, findElement(edge.getId()).getId()));
440                     }
441                 }
442             });
443         }
444 
445         // Some flow nodes may have a default sequence flow
446         int defaultEdgeCount = 0;
447         for (final EdgeType edge : cpfNode.getOutgoingEdges()) {
448             if (edge.isDefault()) {
449 
450                // No element should have more than one default edge
451                if (++defaultEdgeCount > 1) {
452                    throw new CanoniserException("CPF node " + cpfNode.getId() + " has " + defaultEdgeCount + " default edges");
453                 }
454 
455                 // The edge will be flagged as default after all elements have been created (it might not exist yet)
456                 defer(new Initialization() {
457                     public void initialize() throws CanoniserException {
458                         assert edge.getSourceId() != null;
459 
460                         TBaseElement  defaultingElement = idMap.get(edge.getSourceId());
461                         TSequenceFlow defaultFlow       = (TSequenceFlow) idMap.get(edge.getId());
462 
463                         assert defaultingElement != null : "Could not find BPMN element corresponding to source of default CPF edge " + edge.getId();
464                         assert defaultFlow       != null : "Could not find BPMN flow corresponding to default CPF edge " + edge.getId();
465 
466                         if (defaultingElement instanceof TActivity) {
467                             ((TActivity) defaultingElement).setDefault(defaultFlow);
468                         } else if (defaultingElement instanceof TComplexGateway) {
469                             ((TComplexGateway) defaultingElement).setDefault(defaultFlow);
470                         } else if (defaultingElement instanceof TExclusiveGateway) {
471                             ((TExclusiveGateway) defaultingElement).setDefault(defaultFlow);
472                         } else if (defaultingElement instanceof TInclusiveGateway) {
473                             ((TInclusiveGateway) defaultingElement).setDefault(defaultFlow);
474                         } else if (defaultingElement instanceof TParallelGateway) {
475                             try {
476                                 rewriteExplicitSplitImplicitly((TParallelGateway) defaultingElement);
477                             } catch (CanoniserException e) {
478                                 throw new CanoniserException("Default flow " + defaultFlow.getId() + " from gateway " + defaultingElement.getId() +
479                                                              " can't be rewritten", e);
480                             }
481                         } else {
482                             throw new CanoniserException("Could not set default sequence flow for " + defaultingElement.getId());
483                         }
484                     }
485                 });
486             }
487         }
488     }

490     /**
491      * Rewrite a diverging parallel gateway implicitly.
492      *
493      * This can only be called once all the BPMN elements have been created and have indexed identifiers.
494      * In other words, it should be called by the {@link #defer} method.
495      *
496      * @param gateway  a diverging parallel gateway
497      * @throws CanoniserException if <code>gateway</code> doesn't have exactly one incoming sequence flow
498      */
499     private void rewriteExplicitSplitImplicitly(final TParallelGateway gateway) throws CanoniserException {
500 
501         // Check that this really is a diverging gateway
502         if (gateway.getIncoming().size() != 1) {
503             throw new CanoniserException(gateway.getId() + " is not diverging, incoming flows are " + gateway.getIncoming());
504         }
505 
506         // Remove the gateway and its incoming sequence flow
507         BpmnSequenceFlow incomingFlow = (BpmnSequenceFlow) bpmn.findElement(gateway.getIncoming().get(0));
508 
509         throw new CanoniserException("Rewriting parallel gateway " + gateway.getId() + " implicitly not yet implemented");
510     }

512     void populateActivity(final TActivity activity, final CpfNodeType cpfNode) throws CanoniserException {
513         populateFlowNode(activity, cpfNode);
514 
515         // Create data associations
516         assert cpfNode instanceof WorkType;  // TODO - add a CpfWorkType interface matching WorkType
517         for (ObjectRefType objectRef : ((WorkType) cpfNode).getObjectRef()) {
518             CpfObjectRefType cpfObjectRef = (CpfObjectRefType) objectRef;
519 
520             switch (cpfObjectRef.getType()) {
521             case INPUT:  activity.getDataInputAssociation().add(new BpmnDataInputAssociation(cpfObjectRef, activity, this));   break;
522             case OUTPUT: activity.getDataOutputAssociation().add(new BpmnDataOutputAssociation(cpfObjectRef, activity, this)); break;
523             default:     assert false : "CPF ObjectRef " + cpfObjectRef.getId() + " has unsupported type " + cpfObjectRef.getType();
524             }
525         }
526     }

536     private void populateEvent(final TEvent                                        event,
537                                final List<JAXBElement<? extends TEventDefinition>> eventDefinitionList,
538                                final CpfEventType                                  cpfEvent) throws CanoniserException {
539 
540         populateFlowNode(event, cpfEvent);
541 
542         if (cpfEvent.isCompensation()) {
543             TCompensateEventDefinition ced = new TCompensateEventDefinition();
544             ced.setActivityRef(cpfEvent.getCompensationActivityRef());
545             eventDefinitionList.add(factory.createCompensateEventDefinition(ced));
546         }
547 
548         if (cpfEvent.isError()) {
549             TErrorEventDefinition eed = new TErrorEventDefinition();
550             eed.setErrorRef(cpfEvent.getErrorRef());
551             eventDefinitionList.add(factory.createErrorEventDefinition(eed));
552         }
553 
554         if (cpfEvent instanceof CpfMessageType) {
555             CpfMessageType cpfMessage = (CpfMessageType) cpfEvent;
556 
557             /*
558             switch (cpfMessage.getDirection) {
559             case INCOMING:
560             case OUTGOING:
561             }
562             */
563 
564             TMessageEventDefinition med = new TMessageEventDefinition();
565             //med.setMessageRef((QName) ...);
566             eventDefinitionList.add(factory.createMessageEventDefinition(med));
567         }
568 
569         if (cpfEvent.isSignalCatcher()) {
570             TSignalEventDefinition sed = new TSignalEventDefinition();
571             sed.setSignalRef(cpfEvent.getSignalCaughtRef());
572             eventDefinitionList.add(factory.createSignalEventDefinition(sed));
573         }
574 
575         if (cpfEvent.isSignalThrower()) {
576             TSignalEventDefinition sed = new TSignalEventDefinition();
577             sed.setSignalRef(cpfEvent.getSignalThrownRef());
578             eventDefinitionList.add(factory.createSignalEventDefinition(sed));
579         }
580 
581         if (cpfEvent instanceof CpfTimerType) {
582             CpfTimerType cpfTimer = (CpfTimerType) cpfEvent;
583 
584             if (cpfTimer.getTimeDate() != null) {
585                 TTimerEventDefinition ted = new TTimerEventDefinition();
586                 TFormalExpression fe = new TFormalExpression();
587                 try {

719     /**
720      * Look up or create a variant by name.
721      *
722      * This method generates the IDs for the variants and installs the <code>pc:variants</code> catalogue element.
723      *
724      * @param name  the variant name
725      * @return the variant instance, never <code>null</code>
726      */
727     private Variants.Variant findVariant(String name) {
728         if (variantMap.containsKey(name)) {
729             return variantMap.get(name);
730         } else {
731             if (variants == null) {
732                 variants = new Variants();
733 
734                 // Add the pc:variants extension element to the first Process in the BPMN document
735                 defer(new Initialization() {
736                     public void initialize() throws CanoniserException {
737                         for (JAXBElement<? extends TRootElement> jaxbRoot: bpmn.getRootElement()) {
738                             if (jaxbRoot.getValue() instanceof TProcess) {
739                                 TProcess process = (TProcess) jaxbRoot.getValue();
740                                 TExtensionElements extensionElements = process.getExtensionElements();
741                                 if (extensionElements == null) {
742                                     extensionElements = new TExtensionElements();
743                                     process.setExtensionElements(extensionElements);
744                                 }
745                                 assert extensionElements != null;
746                                 extensionElements.getAny().add(variants);
747                                 return;  // only add the catalog once
748                             }
749                         }
750                         throw new CanoniserException("Unable to install pc:variants");
751                     }
752                 });
753             }
754             assert variants != null;
755 
756             Variants.Variant variant = new Variants.Variant();
757             variant.setId("vid-" + variants.getVariant().size());
758             variant.setName(name);
759 
760             variants.getVariant().add(variant);
761             variantMap.put(name, variant);
762 
763             return variant;
764 	}

767     /**
768      * Translate a CPF attribute list to the BPMN BaseElement's extensionElements.
769      *
770      * @param attributes  the attribute list of the source CPF element
771      * @param baseElement  the destination BPMN element
772      */
773     private void populateBaseElementExtensionElements(final TBaseElement baseElement, final Attributed cpfElement) {
774 
775         final String name = BPMN_CPF_NS + "/" + EXTENSION_ELEMENTS;
776 
777         List<TypeAttribute> attributes = new ArrayList<TypeAttribute>();
778         for (TypeAttribute attr : cpfElement.getAttribute()) {
779             if (name.equals(attr.getName()) || "epml_cpf/extensions".equals(attr.getName())) {
780                 attributes.add(attr);
781             }
782         }
783 
784         boolean isConfigurable = (cpfElement instanceof NodeType         && Boolean.TRUE.equals(((NodeType) cpfElement).isConfigurable()))   ||
785                                  (cpfElement instanceof ObjectType       && Boolean.TRUE.equals(((ObjectType) cpfElement).isConfigurable())) ||
786                                  (cpfElement instanceof ResourceTypeType && Boolean.TRUE.equals(((ResourceTypeType) cpfElement).isConfigurable()));
787 
788         if (attributes.size() > 0 || isConfigurable) {
789             TExtensionElements extensionElements = baseElement.getExtensionElements();
790             if (extensionElements == null) {
791                 extensionElements = factory.createTExtensionElements();
792             }
793             assert extensionElements != null;
794 
795             for (TypeAttribute attribute : attributes) {
796                 if (attribute.getAny() == null) {
797                     String value = attribute.getValue();
798                     if (value != null && value.startsWith("annotation;")) {
799                         String[] values = value.split(";");
800 			assert "annotation".equals(values[0]);
801 
802                         if (cpfElement instanceof EdgeType) {
803                             EdgeType edge = (EdgeType) cpfElement;
804                             Object sourceObject = cpf.getElement(edge.getSourceId());
805                             Object targetObject = cpf.getElement(edge.getTargetId());
806                             if (sourceObject instanceof SplitType || targetObject instanceof JoinType) {
807                                 ConfigurationAnnotation configurationAnnotation = new ConfigurationAnnotation();
808                                 for (int i = 1; i<values.length; i++) {
809                                     ConfigurationAnnotation.Configuration configuration = new ConfigurationAnnotation.Configuration();
810                                     String[] fields = values[i].split(":", 2);
811                                     configuration.setVariantRef(findVariant(fields[0]));
812                                     configurationAnnotation.getConfiguration().add(configuration);
813 			        }
814                                 extensionElements.getAny().add(configurationAnnotation);
815                             }
816                         }
817                         else if (cpfElement instanceof RoutingType) {
818                             final RoutingType routing = (RoutingType) cpfElement;
819                             ConfigurationAnnotation configurationAnnotation = new ConfigurationAnnotation();
820                             for (int i = 1; i<values.length; i++) {
821                                 final ConfigurationAnnotation.Configuration configuration = new ConfigurationAnnotation.Configuration();
822                                 String[] fields = values[i].split(":", 2);
823                                 configuration.setVariantRef(findVariant(fields[0]));
824 
825                                 // type attribute
826                                 switch (fields[1]) {
827                                 case "and": configuration.setType(TGatewayType.PARALLEL);  break;
828                                 case "or":  configuration.setType(TGatewayType.INCLUSIVE);  break;
829                                 case "xor": configuration.setType(TGatewayType.DATA_BASED_EXCLUSIVE);  break;
830                                 default: throw new RuntimeException("Unsupported gateway type: " + fields[1]);
831                                 }
832 
833                                 configurationAnnotation.getConfiguration().add(configuration);
834 			    }
835                             extensionElements.getAny().add(configurationAnnotation);
836                             extensionElements.getAny().add(new Configurable());  // mark that gateway as configurable model
837                         }
838                         else if (cpfElement instanceof WorkType) {
839                             ConfigurationAnnotation configurationAnnotation = new ConfigurationAnnotation();
840                             for (int i = 1; i<values.length; i++) {
841                                 ConfigurationAnnotation.Configuration configuration = new ConfigurationAnnotation.Configuration();
842                                 String[] fields = values[i].split(":", 2);
843                                 configuration.setVariantRef(findVariant(fields[0]));
844                                 configuration.setName(fields[1]);
845                                 configurationAnnotation.getConfiguration().add(configuration);
846 			    }
847                             extensionElements.getAny().add(configurationAnnotation);
848                         }
849                         else {
850                             java.util.logging.Logger.getAnonymousLogger().warning("Unable to interpret annotation attribute of CPF node " + cpfElement);
851                         }
852 		    }
853                 }
854                 else if (attribute.getAny() instanceof Element) {
855                     Element element = (Element) attribute.getAny();
856                     NodeList nodes = element.getChildNodes();
857                     for (int i = 0; i < nodes.getLength(); i++) {
858                         if (nodes.item(i) instanceof Element) {
859                             extensionElements.getAny().add(nodes.item(i));
860                         }
861                     }
862                 }
863                 else {  // the presumption here is that anything that didn't get turned into a DOM Element must be from the cpf: or pc: namespaces
864                     extensionElements.getAny().add(attribute.getAny());
865                 }
866             }
867 
868             if (isConfigurable) {
869                 com.processconfiguration.Configurable pcConfigurable = new com.processconfiguration.ObjectFactory().createConfigurable(); 
870                 extensionElements.getAny().add(pcConfigurable);
871             }
872 
873             baseElement.setExtensionElements(extensionElements);
874         }
