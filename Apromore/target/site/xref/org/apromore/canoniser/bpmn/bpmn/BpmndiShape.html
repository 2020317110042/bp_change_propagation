43  /**
44   * BPMNDI Shape element with canonisation methods.
45   *
46   * @author <a href="mailto:simon.raboczi@uqconnect.edu.au">Simon Raboczi</a>
47   */
48  public class BpmndiShape extends BPMNShape {
49  
50      private static final double SCALE = 1;
51  
52      /** No-arg constructor. */
53      public BpmndiShape() { }
54  
55      /**
56       * Construct a BPMNDI Shape corresponding to an ANF Graphics annotation.
57       * @param graphics  an ANF graphics annotation, never <code>null</code>
58       * @param initializer  BPMN document construction state
59       * @throws CanoniserException if the shape can't be constructed
60       */
61      public BpmndiShape(final GraphicsType graphics, final Initializer initializer) throws CanoniserException {
62          initializer.populateDiagramElement(this, graphics);
63  
64          // Validate the graphics parameter: requires a bounding box, defined by a top-left position and a size (width and height)
65          if (graphics.getPosition().size() != 1) {
66              throw new CanoniserException("ANF Graphics annotation " + graphics.getId() + " for CPF shape " +
67                      graphics.getCpfId() + " should have just one origin position");
68          }
69          if (graphics.getSize() == null) {
70              throw new CanoniserException("ANF Graphics annotation " + graphics.getId() + " for CPF shape " +
71                      graphics.getCpfId() + " should specify a size");
72          }
73  
74          // Validate the cpfId: must reference a BPMN element that has a bounding box
75          TBaseElement bpmnElement = initializer.findElement(graphics.getCpfId());
76          if (bpmnElement instanceof TProcess) {
77              java.util.logging.Logger.getAnonymousLogger().warning(
78                      "ANF graphics " + graphics.getId() + " references CPF element " + graphics.getCpfId() + " which corresponds to a BPMN process " +
79                              bpmnElement.getId()
80              );
81          } else if (!(bpmnElement instanceof TDataObject     ||
82                  bpmnElement instanceof TDataObjectReference ||
83                  bpmnElement instanceof TDataStoreReference  ||
84                  bpmnElement instanceof TFlowNode            ||
85                  bpmnElement instanceof TLane                ||
86                  bpmnElement instanceof TParticipant         ||
87                  bpmnElement instanceof TProcess)) {  // TODO - decide whether TProcess really is legitimate to have a BPMNShape
88  
89              throw new CanoniserException(graphics.getCpfId() + " isn't a BPMN element with a Shape");
90          }
91  
92          // Handle @bpmnElement
93          setBpmnElement(new QName(initializer.getTargetNamespace(),
94                  initializer.findElement(graphics.getCpfId()).getId()));
95  
96          // Handle @isExpanded
97          if (bpmnElement instanceof TSubProcess) {
98              setIsExpanded(true);  // TODO - this should dereference graphics.getCpfId() to determine a proper value
99          }
100 
101         // add the ANF position and size as a BPMNDI bounds
102         Bounds bounds = new Bounds();
103         bounds.setHeight(SCALE * graphics.getSize().getHeight().doubleValue());
104         bounds.setWidth(SCALE * graphics.getSize().getWidth().doubleValue());
105         bounds.setX(SCALE * graphics.getPosition().get(0).getX().doubleValue());
106         bounds.setY(SCALE * graphics.getPosition().get(0).getY().doubleValue());
107         setBounds(bounds);
108     }
109 }
