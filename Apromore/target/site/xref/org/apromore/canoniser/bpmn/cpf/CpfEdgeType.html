35  /**
36   * CPF 0.6 edge with convenience methods.
37   *
38   * @author <a href="mailto:simon.raboczi@uqconnect.edu.au">Simon Raboczi</a>
39   * @since 0.4
40   */
41  public class CpfEdgeType extends EdgeType implements Attributed {
42  
43      private static final Logger LOGGER = LoggerFactory.getLogger(CpfEdgeType.class);
44  
45      /** {@link TypeAttribute} name for the name of the original BPMN sequence flow. */
46      private static final String  NAME = "name";
47  
48      /** This edge's source node. */
49      protected NodeType source;
50  
51      /** This edge's target node. */
52      protected NodeType target;
53  
54      // Constructors
55  
56      /** No-arg constructor. */
57      public CpfEdgeType() { }
58  
59      /**
60       * Construct a CPF Edge from a BPMN Sequence Flow.
61       *
62       * @param sequenceFlow  a BPMN Sequence Flow
63       * @param initializer  global construction state
64       * @throws CanoniserException if the <code>sequenceFlow</code> has more than one condition expression
65       */
66      public CpfEdgeType(final TSequenceFlow sequenceFlow, final Initializer initializer) throws CanoniserException {
67          initializer.populateFlowElement(this, sequenceFlow);
68  
69          // Handle conditionExpression
70          if (sequenceFlow.getConditionExpression() != null) {
71  
72              // We don't handle multiple conditions
73              if (sequenceFlow.getConditionExpression().getContent().size() != 1) {
74                  throw new CanoniserException("BPMN sequence flow " + sequenceFlow.getId() + " has " +
75                                               sequenceFlow.getConditionExpression().getContent().size() +
76                                               " conditions, which the canoniser doesn't implement");
77              }
78  
79              ConditionExpressionType conditionExpr = new ConditionExpressionType();
80              conditionExpr.setExpression(sequenceFlow.getConditionExpression().getContent().get(0).toString());
81              setConditionExpr(conditionExpr);
82          }
83  
84          // Handle @name
85          setName(sequenceFlow.getName());
86  
87          initializer.defer(new Initialization() {
88              public void initialize() throws CanoniserException {
89                  // handle source
90                  if (LOGGER.isTraceEnabled()) {
91                      LOGGER.trace("Flow Sequence source: " + sequenceFlow.getSourceRef());
92                  }
93                  CpfNodeType sourceRef = (CpfNodeType) initializer.findElement(sequenceFlow.getSourceRef());
94                  setSourceId(sourceRef.getId());
95                  sourceRef.getOutgoingEdges().add(CpfEdgeType.this);
96  
97                  // handle target
98                  if (LOGGER.isTraceEnabled()) {
99                      LOGGER.trace("Flow Sequence target: " + sequenceFlow.getTargetRef());
100                 }
101                 CpfNodeType targetRef = (CpfNodeType) initializer.findElement(sequenceFlow.getTargetRef());
102                 setTargetId(targetRef.getId());
103                 targetRef.getIncomingEdges().add(CpfEdgeType.this);
104             }
105         });
106     }
107 
108     // Accessors
109 
110     /** @return this edge's original BPMN name */
111     public String getName() {
112         return ExtensionUtils.getString(getAttribute(), NAME);
113     }
114 
115     /** @param name the name of the original BPMN sequence flow */
116     public void setName(final String name) {
117         ExtensionUtils.setString(getAttribute(), NAME, name);
118     }
119 
120     /** @return this edge's source node */
121     public NodeType getSourceRef() {
122         return source;
123     }
124 
125     /** @param node  the new source node */
126     public void setSourceRef(final NodeType node) {
127         source = node;
128     }
129 
130     /** @param node  the new target node */
131     public void setTargetRef(final NodeType node) {
132         target = node;
133     }
134 
135     /** @return this edge's target node */
136     public NodeType getTargetRef() {
137         return target;
138     }
