111      *

116     /** Initialize {@link #fixNamespacesXslt}. */
117     private String[] getFixNamespacesXslt() {
118         try {
119             ByteArrayOutputStream baos = new ByteArrayOutputStream();
120             TransformerFactory.newInstance().newTransformer().transform(
121                 new StreamSource(getClass().getClassLoader().getResourceAsStream("xsd/fix-namespaces.xsl")),
122                 new StreamResult(baos)
123             );
124             String[] s = baos.toString().split("genid:TARGET");
125             if (s.length != 2) {
126                 throw new RuntimeException("genid:TARGET does not occur exactly once within fix-namespaces.xsl");
127             }
128             return s;
129             
130         } catch (TransformerException e) {
131             throw new RuntimeException("Couldn't parse fix-namespaces.xslt", e);
132         }
133     }

154     /**
155      * Construct a BPMN model from a canonical model.
156      * In other words, de-canonise a CPF/ANF model into a BPMN one.
157      *
158      * @param cpf  a canonical process model
159      * @param anf  annotations for the canonical process model
160      * @throws CanoniserException if unable to generate BPMN from the given CPF and ANF arguments
161      */
162     private BpmnDefinitions(final CpfCanonicalProcessType cpf, final AnnotationsType anf) throws CanoniserException {
163         // We can get by without an ANF parameter, but we definitely need a CPF
164         if (cpf == null) {
165             throw new CanoniserException("Cannot create BPMN from null CPF");
166         }
167 
168         Initializer initializer = new Initializer(this, cpf, "http://www.apromore.org/bpmn/" + UUID.randomUUID() + "#");
169 
170         // Set attributes of the document root
171         setExporter(APROMORE_URI);
172         setExporterVersion(APROMORE_VERSION);
173         setExpressionLanguage(CPFSchema.EXPRESSION_LANGUAGE_XPATH);  // This is the default, so specifying it is redundant
174         setId(null);
175         setName(cpf.getName());
176         setTargetNamespace(initializer.getTargetNamespace());
177         setTypeLanguage(CPFSchema.TYPE_LANGUAGE_XSD);  // This is the default, so specifying it is redundant
178 
179         /* TODO - add as extension attributes
180         String author = cpf.getAuthor();
181         String creationDate = cpf.getCreationDate();
182         String modificationDate = cpf.getModificationDate();
183         */
184 
185         // Assume there will be pools, all of which belong to a single collaboration
186         TCollaboration collaboration = initializer.getFactory().createTCollaboration();
187         JAXBElement<TCollaboration> wrapperCollaboration = initializer.getFactory().createCollaboration(collaboration);
188 
189         // Workaround for the Apromore-core's failure to generate the CPF rootIds attribute
190         final List<String> rootIds = cpf.getRootIds();
191         final List<String> subnetIds = new ArrayList<String>();
192         if (rootIds.size() == 0) {
193             cpf.accept(new org.apromore.cpf.TraversingVisitor(new org.apromore.cpf.DepthFirstTraverserImpl(), new org.apromore.cpf.BaseVisitor() {
194                 @Override public void visit(final NetType net) { rootIds.add(net.getId()); }
195                 @Override public void visit(final TaskType task) { subnetIds.add(task.getSubnetId()); }  // null check not required
196             }));
197             rootIds.removeAll(subnetIds);
198             initializer.warn("Using reconstructed root net list: " + rootIds);
199         }
200 
201         // Translate CPF Nets as BPMN Processes TODO: Fix
202         for (final NetType net : cpf.getNet()) {
203             // Only root elements are decanonised here; subnets are dealt with by recursion
204             if (!rootIds.contains(net.getId())) {
205                 continue;
206             }
207 
208             getRootElement().add(initializer.getFactory().createProcess(new BpmnProcess((CpfNetType) net, initializer, collaboration)));
209 
210             /*
211             // If we haven't added the collaboration yet and this process is a pool, add the collaboration
212             if (!getRootElement().contains(wrapperCollaboration)) {
213                 getRootElement().add(wrapperCollaboration);
214             }
215             */
216         }
217 
218         // pc:variants occurring on the document element of the CPF need to be moved to one of the root elements of the BPMN
219         for (org.apromore.cpf.TypeAttribute attribute: cpf.getAttribute()) {
220             outer: switch (attribute.getName()) {
221             case "bpmn_cpf/extensions":
222                 for (JAXBElement<? extends TRootElement> jaxbRoot: getRootElement()) {
223                     TRootElement root = jaxbRoot.getValue();
224                     TExtensionElements extensionElements = root.getExtensionElements();
225                     if (extensionElements == null) {
226                         extensionElements = new BpmnObjectFactory().createTExtensionElements();
227                         root.setExtensionElements(extensionElements);
228                     }
229                     assert extensionElements != null;
230                     assert extensionElements == root.getExtensionElements();
231 
232                     extensionElements.getAny().add(attribute.getAny());
233 
234                     break outer;
235                 }
236                 throw new CanoniserException("Couldn't find a root element to place the BPMN variants attribute");
237 
238             case "BranchID":
239             case "BranchName":
240             case "RootFragmentId":
241             case "InitialFormat":
242             case "IntialFormat":
243             case "PMVID":
244             case "ProcessName":
245             case "VersionNumber":
246                 // recognized, but ignored
247                 break;
248 
249             default:
250                 throw new CanoniserException("Unsupported extension attribute in CPF: " + attribute.getName() + " value=" + attribute.getValue() + " any=" + attribute.getAny());
251             }
252         }
253 
254         // Make sure all the deferred fields did eventually get filled in
255         initializer.close();
256 
257         // Translate any ANF annotations into a BPMNDI diagram element
258         if (anf != null) {
259             getBPMNDiagram().add(new BpmndiDiagram(anf, initializer));
260         }
261     }

265     /**
266      * Construct a BPMN model from a canonical model.
267      * In other words, de-canonise a CPF/ANF model into a BPMN one.
268      *
269      * @param cpf  a canonical process model
270      * @param anf  annotations for the canonical process model
271      * @return a new BPMN process model corresponding to <code>cpf</code>
272      * @throws CanoniserException if unable to generate BPMN from the given CPF and ANF arguments
273      */
274     public static BpmnDefinitions newInstance(final CpfCanonicalProcessType cpf, final AnnotationsType anf) throws CanoniserException {
275 
276         // Use JAXB to construct the de-canonised BPMN document
277         BpmnDefinitions definitions = new BpmnDefinitions(cpf, anf);
278 
279         // Use XSLT to work around JAXB's inability to correctly generate IDREFS like bpmn:flowNodeRefs
280         try {
281             return definitions.correctFlowNodeRefs(definitions, new BpmnObjectFactory());
282         } catch (JAXBException | TransformerException e) {
283             throw new CanoniserException("Unable to correct JAXB-misgenerated bpmn:flowNodeRef elements", e);
284         }
285     }

287     /**
288      * Construct an instance from a stream.
289      *
290      * @param in  a BPMN-formatted stream
291      * @param validate  whether to perform schema validation
292      * @throws JAXBException if the stream can't be parsed as BPMN
293      * @return the parsed instance
294      */
295     @SuppressWarnings("unchecked")
296     public static BpmnDefinitions newInstance(final InputStream in, final Boolean validate) throws JAXBException {
297         assert in != null;
298         Unmarshaller unmarshaller = /*BPMN_CONTEXT*/ newContext().createUnmarshaller();
299         BpmnIDResolver resolver = new BpmnIDResolver();
300         BpmnUnmarshallerListener listener = new BpmnUnmarshallerListener(resolver);
301         unmarshaller.setListener(listener);
302         unmarshaller.setProperty(ID_RESOLVER, resolver);
303         unmarshaller.setProperty(OBJECT_FACTORY, new Object[]{new BpmnObjectFactory(), new BpmndiObjectFactory()});
304         if (validate) {
305             unmarshaller.setSchema(getBpmnSchema() /*BPMN_SCHEMA*/);
306         }
307         BpmnDefinitions result = ((JAXBElement<BpmnDefinitions>) unmarshaller.unmarshal(new StreamSource(in))).getValue();
308         result.idMap.putAll(listener.getIdMap());
309         return result;
310     }

338     /**
339      * Serialize this instance to a steam.
340      *
341      * Default deserialization is pretty-printed but not schema-validated.
342      *
343      * @param out  the stream for writing
344      * @param validate  whether to perform schema validation
345      * @throws JAXBException if the steam can't be written to
346      */
347     public void marshal(final OutputStream out, final Boolean validate) throws JAXBException {
348       try {
349         // Create an empty DOM
350         DOMResult intermediateResult = new DOMResult();
351 
352         // Marshal from JAXB to DOM
353         Marshaller marshaller = /*BPMN_CONTEXT*/ newContext().createMarshaller();
354         marshaller.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, true);
355         if (validate) {
356             marshaller.setSchema(getBpmnSchema() /*BPMN_SCHEMA*/);
357         }
358         marshaller.marshal(new BpmnObjectFactory().createDefinitions(this), intermediateResult);
359 
360         // Apply the XSLT transformation, from DOM to the output stream
361         TransformerFactory transformerFactory = TransformerFactory.newInstance();
362         Transformer transformer = transformerFactory.newTransformer(
363             new StreamSource(new java.io.StringBufferInputStream(fixNamespacesXslt[0] + getTargetNamespace() + fixNamespacesXslt[1]))
364         );
365         DOMSource finalSource = new DOMSource(intermediateResult.getNode());
366         StreamResult finalResult = new StreamResult(out);
367         transformer.transform(finalSource, finalResult);
368       } catch (TransformerException e) { throw new JAXBException("Dodgy wrapped exception", e); }  // TODO - create transformer elsewhere
369     }

371     /**
372      * Workaround for incorrect marshalling of {@link TLane#getFlowNodeRef} by JAXB.
373      *
374      * A flow node reference on a lane ought to be serialized as
375      * <pre>
376      * &lt;lane&gt;
377      *   &lt;flowNodeRef&gt;id-123&lt;/flowNodeRef&gt;
378      * &lt;/lane&gt;
379      * </pre>
380      * but instead they end up serialized as
381      * <pre>
382      * &lt;lane&gt;
383      *   &lt;task id="id-123"/&gt;
384      * &lt;/lane&gt;
385      * </pre>
386      * This method applies an XSLT transform to correct things.
387      *
388      * @param definitions  the buggy JAXB document
389      * @param factory  source of elements for the result document
390      * @throws JAXBException if <code>definitions</code> can't be marshalled to XML or unmarshalled back
391      * @throws TransformerException  if the XSLT transformation fails
392      * @return corrected JAXB document
393      */
394     // TODO - change the return type and the factory parameter to be Definitions and ObjectFactory, and move to bpmn-schema
395     public BpmnDefinitions correctFlowNodeRefs(final BpmnDefinitions definitions,
396                                                final BpmnObjectFactory factory) throws JAXBException, TransformerException {
397 
398         JAXBContext context = JAXBContext.newInstance(factory.getClass(),
399                                                       com.processconfiguration.ObjectFactory.class,
400                                                       org.omg.spec.bpmn._20100524.di.ObjectFactory.class,
401                                                       org.omg.spec.bpmn._20100524.model.ObjectFactory.class,
402                                                       org.omg.spec.dd._20100524.dc.ObjectFactory.class,
403                                                       org.omg.spec.dd._20100524.di.ObjectFactory.class,
404                                                       com.signavio.ObjectFactory.class);
405 
406         // Marshal the BPMN into a DOM tree
407         DOMResult intermediateResult = new DOMResult();
408         Marshaller marshaller = context.createMarshaller();
409         marshaller.marshal(factory.createDefinitions(definitions), intermediateResult);
410 
411         // Apply the XSLT transformation, generating a new DOM tree
412         TransformerFactory transformerFactory = TransformerFactory.newInstance();
413         Transformer transformer = transformerFactory.newTransformer(
414             new StreamSource(getClass().getClassLoader().getResourceAsStream("xsd/fix-flowNodeRef.xsl"))
415         );
416         DOMSource finalSource = new DOMSource(intermediateResult.getNode());
417         DOMResult finalResult = new DOMResult();
418         transformer.transform(finalSource, finalResult);
419 
420         // Unmarshal back to JAXB
421         Object def2 = context.createUnmarshaller().unmarshal(finalResult.getNode());
422         return ((JAXBElement<BpmnDefinitions>) def2).getValue();
423     }

425     /**
426      * Replace every instance of multiple incoming sequence flows to an activity with an exclusive data-based gateway,
427      * every instance of multiple outgoing sequence flows from an activity with a parallel gateway, and every mixed gateway
428      * as a converging gateway flowing to a diverging gateway.
429      *
430      * This is required as pre-processing for canonisation, since CPF Work elements must have no more than one incoming edge and
431      * no more than one outgoing edge.
432      *
433      * @throws CanoniserException if the rewriting can't be performed
434      */
435     public void rewriteImplicitGatewaysExplicitly() throws CanoniserException {
436         for (JAXBElement<? extends TRootElement> rootElement : getRootElement()) {
437             if (rootElement.getValue() instanceof TProcess) {
438                 TProcess process = (TProcess) rootElement.getValue();
439                 rewriteFlowElements(process.getFlowElement());
440             }
441         }
442     }
443 
444     private void rewriteFlowElements(List<JAXBElement<? extends TFlowElement>> jfel) throws CanoniserException {
445         for (JAXBElement<? extends TFlowElement> flowElement : new ArrayList<JAXBElement<? extends TFlowElement>>(jfel)) {
446             if (flowElement.getValue() instanceof TFlowNode && !(flowElement.getValue() instanceof TGateway)) {
447                 TFlowNode flowNode = (TFlowNode) flowElement.getValue();
448 
449                 if (flowNode.getIncoming().size() > 1) {
450 
451                     // Create a converging XOR gateway
452                     BpmnObjectFactory factory = new BpmnObjectFactory();
453                     TExclusiveGateway xor = factory.createTExclusiveGateway();
454                     xor.setId(flowNode.getId() + "_implicit_join");
455                     xor.setGatewayDirection(TGatewayDirection.CONVERGING);
456                     jfel.add(factory.createExclusiveGateway(xor));
457 
458                     // Retarget the incoming flows to the XOR gateway
459                     for (QName incoming: flowNode.getIncoming()) {
460                         BpmnSequenceFlow incomingFlow = (BpmnSequenceFlow) findElement(incoming);
461                         incomingFlow.setTargetRef(xor);
462                     }
463 
464                     xor.getIncoming().addAll(flowNode.getIncoming());
465                     flowNode.getIncoming().clear();
466 
467                     // Create a flow from the XOR gateway to the flow node
468                     BpmnSequenceFlow flow = factory.createTSequenceFlow();
469                     flow.setId(flowNode.getId() + "_implicit_join_edge");  // TODO - should use IdFactory to ward against clashes
470                     flow.setSourceRef(xor);
471                     flow.setTargetRef(flowNode);
472                     jfel.add(factory.createSequenceFlow(flow));
473 
474                     QName flowQName = new QName(targetNamespace, flow.getId());
475                     flowNode.getIncoming().add(flowQName);
476                     xor.getOutgoing().add(flowQName);
477                 }
478 
479                 if (flowNode.getOutgoing().size() > 1) {
480 
481                     // Create a diverging AND gateway
482                     BpmnObjectFactory factory = new BpmnObjectFactory();
483                     TParallelGateway and = factory.createTParallelGateway();
484                     and.setId(flowNode.getId() + "_implicit_split");
485                     and.setGatewayDirection(TGatewayDirection.DIVERGING);
486                     jfel.add(factory.createParallelGateway(and));
487 
488                     // Change the source of the outgoing flows to be from the AND gateway
489                     for (QName outgoing: flowNode.getOutgoing()) {
490                         BpmnSequenceFlow outgoingFlow = (BpmnSequenceFlow) findElement(outgoing);
491                         outgoingFlow.setSourceRef(and);
492                     }
493 
494                     and.getOutgoing().addAll(flowNode.getOutgoing());
495                     flowNode.getOutgoing().clear();
496 
497                     // Create a flow to the AND gateway from the flow node
498                     BpmnSequenceFlow flow = factory.createTSequenceFlow();
499                     flow.setId(flowNode.getId() + "_implicit_split_edge");  // TODO - should use IdFactory to ward against clashes
500                     flow.setSourceRef(flowNode);
501                     flow.setTargetRef(and);
502                     jfel.add(factory.createSequenceFlow(flow));
503 
504                     QName flowQName = new QName(targetNamespace, flow.getId());
505                     flowNode.getOutgoing().add(flowQName);
506                     and.getIncoming().add(flowQName);
507                 }
508 
509                 // Recursively rewrite any subprocesses
510                 if (flowNode instanceof TSubProcess) {
511                     rewriteFlowElements(((TSubProcess) flowNode).getFlowElement());
512                 }
513             }
514 
515             // Rewrite mixed gateways
516             if (flowElement.getValue() instanceof TGateway) {
517                 TGateway gateway = (TGateway) flowElement.getValue();
518 
519                 if (gateway.getIncoming().size() > 1 && gateway.getOutgoing().size() > 1) {
520 
521                     // Create a diverging XOR gateway
522                     BpmnObjectFactory factory = new BpmnObjectFactory();
523                     TGateway newGateway;
524                     if (gateway instanceof TComplexGateway) {
525                         newGateway = factory.createTComplexGateway();
526                         jfel.add(factory.createComplexGateway((TComplexGateway) newGateway));
527                     } else if (gateway instanceof TEventBasedGateway) {
528                         newGateway = factory.createTEventBasedGateway();
529                         jfel.add(factory.createEventBasedGateway((TEventBasedGateway) newGateway));
530                     } else if (gateway instanceof TExclusiveGateway) {
531                         newGateway = factory.createTExclusiveGateway();
532                         jfel.add(factory.createExclusiveGateway((TExclusiveGateway) newGateway));
533                     } else if (gateway instanceof TInclusiveGateway) {
534                         newGateway = factory.createTInclusiveGateway();
535                         jfel.add(factory.createInclusiveGateway((TInclusiveGateway) newGateway));
536                     } else if (gateway instanceof TParallelGateway) {
537                         newGateway = factory.createTParallelGateway();
538                         jfel.add(factory.createParallelGateway((TParallelGateway) newGateway));
539                     } else {
540                         throw new CanoniserException("Mixed gateway " + gateway.getId() + " of type " + gateway.getClass() + " unsupported");
541                     }
542                     assert newGateway != null;
543                     newGateway.setId(gateway.getId() + "_mixed_split");
544                     newGateway.setGatewayDirection(TGatewayDirection.DIVERGING);
545 
546                     // Re-source the outgoing flows from the diverging gateway
547                     for (QName outgoing: gateway.getOutgoing()) {
548                         BpmnSequenceFlow outgoingFlow = (BpmnSequenceFlow) findElement(outgoing);
549                         outgoingFlow.setSourceRef(newGateway);
550                     }
551                     gateway.setGatewayDirection(TGatewayDirection.CONVERGING);
552                     gateway.getOutgoing().clear();
553 
554                     // Create a flow from the original (now converging) gateway to the new diverging gateway
555                     BpmnSequenceFlow flow = factory.createTSequenceFlow();
556                     flow.setId(gateway.getId() + "_mixed_edge");
557                     flow.setSourceRef(gateway);
558                     flow.setTargetRef(newGateway);
559                     jfel.add(factory.createSequenceFlow(flow));
560 
561                     QName flowQName = new QName(targetNamespace, flow.getId());
562                     newGateway.getIncoming().add(flowQName);
563                     gateway.getOutgoing().add(flowQName);
564                 }
565             }
566         }
567     }

569     /**
570      * @param id  the name of a BPMN element
571      * @throws CanoniserException if <code>id</code> isn't a BPMN element from the current document
572      */
573     public TBaseElement findElement(final QName id) throws CanoniserException {
574         if (!targetNamespace.equals(id.getNamespaceURI())) {
575             if ("http://www.omg.org/spec/BPMN/20100524/MODEL".equals(id.getNamespaceURI())) {
576                 java.util.logging.Logger.getAnonymousLogger().warning(
577                     id + " is a QName reference to the BPMN namespace rather than the document's target namespace.  " +
578                     "Treating as {" + targetNamespace + "}" + id.getLocalPart() + " instead.  " +
579                     "Please fix the original document.");
580             } else {
581                 throw new CanoniserException(id + " is not local to this BPMN model");
582             }
583         }
584 
585         return idMap.get(id.getLocalPart());
586     }
