79      /**
80       * Construct a CPF document corresponding to a BPMN document.
81       *
82       * The resulting CPF requires its <code>uri</code> property to be set in order to be schema-legal.
83       *
84       * @param definitions  a BPMN document
85       * @throws CanoniserException  if the CPF document can't be constructed
86       */
87      public CpfCanonicalProcessType(final BpmnDefinitions definitions) throws CanoniserException {
88  
89          final Initializer initializer = new Initializer(this, definitions, elementMap);
90  
91          // Populate attributes
92          setName(requiredName(definitions.getName()));
93          setVersion(CPFSchema.CPF_VERSION);
94  
95          // CPF Nodes aren't allowed more than one incoming edge or more than one outgoing edge
96          definitions.rewriteImplicitGatewaysExplicitly();
97  
98          // Traverse the BPMN document, creating CPF elements corresponding to each BPMN element
99          for (JAXBElement<? extends TRootElement> jere : definitions.getRootElement()) {
100             TRootElement rootElement = jere.getValue();
101             if (rootElement instanceof TCollaboration) {
102                 TCollaboration collaboration = (TCollaboration) rootElement;
103 
104                 // Participants
105                 for (final TParticipant participant : collaboration.getParticipant()) {
106                     if (participant.getProcessRef() == null) {  // black box pool
107                         getResourceType().add(new CpfResourceTypeTypeImpl(participant, initializer));
108                     } else {
109                         TProcess process = (TProcess) initializer.findBpmnElement(participant.getProcessRef());
110                         if (process == null) {
111                             throw new CanoniserException("Participant " + participant.getId() + " missing process " + participant.getProcessRef());
112                         }
113                         initializer.warn("Canonisation ignores pool " + participant.getId() + " for process " + participant.getProcessRef());
114                     }
115                 }
116 
117                 // Message flows
118                 for (final TMessageFlow messageFlow : collaboration.getMessageFlow()) {
119                     messageFlow.getName();
120                     initializer.defer(new Initialization() {
121                         public void initialize() throws CanoniserException {
122                             TBaseElement source = initializer.findBpmnElement(messageFlow.getSourceRef());
123                             if (source == null) {
124                                 throw new CanoniserException("Message flow " + messageFlow.getId() + " missing source " + messageFlow.getSourceRef());
125                             }
126 
127                             TBaseElement target = initializer.findBpmnElement(messageFlow.getTargetRef());
128                             if (target == null) {
129                                 throw new CanoniserException("Message flow " + messageFlow.getId() + " missing target " + messageFlow.getTargetRef());
130                             }
131 
132                             if (messageFlow.getMessageRef() != null) {
133                                 TMessage message = (TMessage) initializer.findBpmnElement(messageFlow.getMessageRef());
134                                 if (message == null) {
135                                     throw new CanoniserException("Message flow " + messageFlow.getId() + " missing message ref " +
136                                                                  messageFlow.getMessageRef());
137                                 }
138                             }
139                         }
140                     });
141                 }
142 
143             } else if (rootElement instanceof TProcess) {  // Each top-level BPMN Process becomes a CPF Net in the rootIDs list
144                 new CpfNetType(new ProcessWrapper((TProcess) rootElement), null, initializer, this);
145             } else {
146                 initializer.warn("Canonisation ignores " + rootElement.getId() + " of type " + rootElement.getClass().getCanonicalName());
147             }
148         }
149 
150         // Execute deferred operations
151         initializer.close();
152 
153         // Assemble the list of boundary events
154         final Set<CpfTaskType> tasks = new HashSet<CpfTaskType>();
155         accept(new TraversingVisitor(new DepthFirstTraverserImpl(), new BaseVisitor() {
156             @Override public void visit(final TaskType task) {
157                 CpfTaskType cpfTask = (CpfTaskType) task;
158                 if (cpfTask.getBoundaryEvents().size() > 0) {
159                     tasks.add(cpfTask);
160                 }
161             }
162         }));
163 
164         // Rewrite the boundary events
165         for (CpfTaskType cpfTask : tasks) {
166             rewriteTaskWithBoundaryEvents(cpfTask, initializer);
167         }
168     }

170     /**
171      * Construct an instance from a CPF-formatted stream.
172      *
173      * @param in  a CPF-formatted stream
174      * @param validate  whether to perform schema validation while parsing
175      * @return JAXB object model of the parsed stream
176      * @throws JAXBException if the stream can't be unmarshalled as CPF
177      * @throws SAXException if the CPF schema can't be parsed
178      */
179     public static CpfCanonicalProcessType newInstance(final InputStream in, final Boolean validate) throws JAXBException, SAXException {
180         Unmarshaller unmarshaller = JAXBContext.newInstance(org.apromore.cpf.ObjectFactory.class, com.processconfiguration.ObjectFactory.class)
181                                                .createUnmarshaller();
182         CpfUnmarshallerListener listener = new CpfUnmarshallerListener();
183         unmarshaller.setListener(listener);
184         unmarshaller.setProperty(ID_RESOLVER, new CpfIDResolver());
185         unmarshaller.setProperty(OBJECT_FACTORY, new ObjectFactory());
186         if (validate) {
187             unmarshaller.setSchema(CPFSchema.getCPFSchema());
188         }
189         CpfCanonicalProcessType result = ((JAXBElement<CpfCanonicalProcessType>) unmarshaller.unmarshal(new StreamSource(in))).getValue();
190         result.elementMap.clear();
191         result.elementMap.putAll(listener.getElementMap());
192         return result;
193     }

196      * Convert a {@link CanonicalProcessType} to a {@link CpfCanonicalProcessType}.
197      *
198      * @param cpf  any CPF document
199      * @return an instrumented version of the given <code>cpf</code>
200      * @throws JAXBException if the remarshalling fails
201      * @throws SAXException if the CPF schema can't be parsed
202      */
203     public static CpfCanonicalProcessType remarshal(final CanonicalProcessType cpf) throws JAXBException, SAXException {
204         if (cpf instanceof CpfCanonicalProcessType) {
205             return (CpfCanonicalProcessType) cpf;
206         } else {
207             ByteArrayOutputStream out = new ByteArrayOutputStream();
208             CPFSchema.marshalCanonicalFormat(out, cpf, false);
209             ByteArrayInputStream in = new ByteArrayInputStream(out.toByteArray());
210             return newInstance(in, false);
211         }
212     }

241     /**
242      * Insert an AND split routing in front of a task, connected to its boundary events.
243      *
244      * The task and any interrupting boundary events will cancel each other.
245      *
246      * @param cpfTask  a CPF task corresponding to a BPMN activity with boundary events
247      * @param initializer  document construction state
248      */
249     private void rewriteTaskWithBoundaryEvents(final CpfTaskType task, final Initializer initializer) {
250         assert task.getIncomingEdges().size() == 1 : task.getId() + " doesn't have a single incoming edge";
251 
252         CpfNetType parent = initializer.findParent(task);
253 
254         // What other elements cancel the task?  They must also cancel the elements we create for rewriting.
255         Set<WorkType> cancelling = new HashSet<WorkType>();
256         for (NodeType node : parent.getNode()) {
257             if (node instanceof WorkType) {
258                 WorkType work = (WorkType) node;
259                 for (CancellationRefType cancellationRef : work.getCancelNodeId()) {
260                     if (task.getId().equals(cancellationRef.getRefId())) {
261                          cancelling.add(work);
262                     }
263                 }
264             }
265         }
266 
267         for (CpfEdgeType incomingEdge : new ArrayList<CpfEdgeType>(task.getIncomingEdges())) {
268 
269             // Create AND split
270             CpfANDSplitType andSplit = new CpfANDSplitType();
271             andSplit.setId(initializer.newId(task.getId() + "_boundary_routing"));
272             elementMap.put(andSplit.getId(), andSplit);
273             parent.getNode().add(andSplit);
274             cancel(andSplit, cancelling);
275 
276             // Reconnect the incoming edge to the AND split
277             incomingEdge.setTargetId(andSplit.getId());
278             ((CpfNodeType) elementMap.get(incomingEdge.getTargetId())).getIncomingEdges().remove(incomingEdge);
279             andSplit.getIncomingEdges().add(incomingEdge);
280 
281             // Create a new edge from the AND split to the task
282             CpfEdgeType edge = new CpfEdgeType();
283             edge.setId(initializer.newId(task.getId() + "_boundary_edge"));
284             elementMap.put(edge.getId(), edge);
285             parent.getEdge().add(edge);
286             cancel(edge, cancelling);
287 
288             edge.setSourceId(andSplit.getId());
289             andSplit.getOutgoingEdges().add(edge);
290 
291             edge.setTargetId(task.getId());
292             task.getIncomingEdges().add(edge);
293 
294             for (CpfEventType event : task.getBoundaryEvents()) {
295 
296                 // Create a new edge from the AND split to the event
297                 edge = new CpfEdgeType();
298                 edge.setId(initializer.newId(event.getId() + "_boundary_edge"));
299                 elementMap.put(edge.getId(), edge);
300                 parent.getEdge().add(edge);
301                 cancel(edge, cancelling);
302 
303                 edge.setSourceId(andSplit.getId());
304                 andSplit.getOutgoingEdges().add(edge);
305 
306                 edge.setTargetId(event.getId());
307                 event.getIncomingEdges().add(edge);
308 
309                 // The task cancels the boundary event
310                 CancellationRefType cancellationRef = new CancellationRefType();
311                 cancellationRef.setRefId(event.getId());
312                 task.getCancelNodeId().add(cancellationRef);
313 
314                 if (event.isInterrupting()) {
315 
316                     // Cancel the task
317                     cancellationRef = new CancellationRefType();
318                     cancellationRef.setRefId(task.getId());
319                     event.getCancelNodeId().add(cancellationRef);
320 
321                     // Cancel any other boundary events
322                     for (CpfEventType otherEvent : task.getBoundaryEvents()) {
323                         if (!event.equals(otherEvent)) {
324                             cancellationRef = new CancellationRefType();
325                             cancellationRef.setRefId(otherEvent.getId());
326                             event.getCancelNodeId().add(cancellationRef);
327                         }
328                     }
329                 }
330             }
331         }
332     }

334     /**
335      * Add a CPF Edge to the cancellation set of a group of CPF Work elements.
336      *
337      * @param cancelledEdge  a CPF Edge identifier
338      * @param cancellingWorks  a set of CPF Work elements
339      */
340     private void cancel(final EdgeType cancelledEdge, Set<WorkType> cancellingWorks) {
341         if (!cancellingWorks.isEmpty()) {
342             CancellationRefType cancellationRef = new CancellationRefType();
343             cancellationRef.setRefId(cancelledEdge.getId());
344 
345             for (WorkType work : cancellingWorks) {
346                 work.getCancelEdgeId().add(cancellationRef);
347             }
348         }
349     }
