27  import org.apromore.canoniser.bpmn.Initialization;
28  import org.apromore.canoniser.bpmn.cpf.CpfObjectRefType;
29  import org.apromore.canoniser.exception.CanoniserException;
30  import static  org.apromore.cpf.InputOutputType.OUTPUT;

34  /**
35   * BPMN Data Output Association with canonisation methods.
36   *
37   * @author <a href="mailto:simon.raboczi@uqconnect.edu.au">Simon Raboczi</a>
38   */
39  public class BpmnDataOutputAssociation extends TDataOutputAssociation {
40  
41      /** No-arg constructor. */
42      public BpmnDataOutputAssociation() { }
43  
44      /**
45       * Construct a BPMN Data Output Association corresponding to a CPF ObjectRef.
46       *
47       * @param objectRef  a CPF Object Reference of type {@link #OUTPUT}
48       * @param parent  the BPMN activity containing this instance
49       * @param initializer  BPMN document construction state
50       * @throws CanoniserException  if the association can't be constructed
51       */
52      public BpmnDataOutputAssociation(final CpfObjectRefType objectRef,
53                                       final TActivity        parent,
54                                       final Initializer      initializer) throws CanoniserException {
55          assert OUTPUT.equals(objectRef.getType()) : objectRef.getId() + " is not typed as an output";
56          initializer.populateBaseElement(this, objectRef);
57  
58          // There's a bug in JAXB that makes it impossible to directly add elements to collections of IDREFs, like sourceRef
59          // As a workaround, I put the id of the sourceRef into an attribute and fix it later using XSLT
60          getOtherAttributes().put(new QName("workaround"), parent.getId());
61  
62          initializer.defer(new Initialization() {
63              public void initialize() {
64                  assert initializer.findElement(objectRef.getObjectId()) != null;
65                  setTargetRef(initializer.findElement(objectRef.getObjectId()));
66              }
67          });
68      }
69  }
