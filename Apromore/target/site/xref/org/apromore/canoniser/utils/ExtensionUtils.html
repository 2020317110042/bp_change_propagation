70      /**
71       * @param obj  an XML {@link Node}
72       * @param namespace  the namespace of an extension name
73       * @param localPart  the local part of an extension name
74       * @return whether the <code>obj</code>'s QName matches the given extension name
75       */
76      public static boolean isValidFragment(final Object obj, final String namespace, final String localPart) {
77          return ((Node) obj).getNamespaceURI().equals(namespace) && ((Node) obj).getLocalName().equals(localPart);
78      }

80      /**
81       * 'Unmarshal' a native object to its expected class.
82       * Will throw an JAXBException if class is not known.
83       * Note this method will also convert any non-matching Object!
84       *
85       * @param object
86       *            returned of 'getAny' or similiar
87       * @param expectedClass
88       *            object of this class will be returned
89       * @param nativeContext
90       *            context for marshalling the origin format
91       * @return JAXB Object matching expected class
92       * @throws CanoniserException
93       */
94      public static <T> T unmarshalFragment(final Object      object,
95                                            final Class<T>    expectedClass,
96                                            final JAXBContext nativeContext) throws CanoniserException {
97          try {
98              if (nativeContext != null) {
99                  final Unmarshaller u = nativeContext.createUnmarshaller();
100                 final JAXBElement<T> jaxbElement = u.unmarshal((Node) object, expectedClass);
101                 return jaxbElement.getValue();
102             } else {
103                 throw new CanoniserException("Missing JAXBContext");
104             }
105         } catch (final JAXBException e) {
106             throw new CanoniserException("Failed to parse extension with expected class " + expectedClass.getName(), e);
107         }
108     }

110     /**
111      * 'Marshal' a JAXB object of some native schema to a DOM Node that can be added to 'xs:any'.
112      *
113      * @param elementName
114      *            to use as local part
115      * @param object
116      *            to be marshaled
117      * @param expectedClass
118      *            class of the object
119      * @param nativeNS
120      *            Namespace URL for the origin format
121      * @param nativeContext
122      *            context for marshalling the origin format
123      * @return XML Element containing the markup fragment
124      * @throws CanoniserException
125      */
126     public static <T> Element marshalFragment(final String      elementName,
127                                               final T           object,
128                                               final Class<T>    expectedClass,
129                                               final String      nativeNS,
130                                               final JAXBContext nativeContext) throws CanoniserException {
131         try {
132             if (nativeContext != null) {
133                 final Marshaller m = nativeContext.createMarshaller();
134                 m.setProperty(Marshaller.JAXB_FRAGMENT, Boolean.TRUE);
135                 final JAXBElement<T> element = new JAXBElement<T>(new QName(nativeNS, elementName), expectedClass, object);
136                 final DocumentBuilderFactory dbf = DocumentBuilderFactory.newInstance();
137                 dbf.setNamespaceAware(false);
138                 Document doc;
139                 try {
140                     doc = dbf.newDocumentBuilder().newDocument();
141                 } catch (final ParserConfigurationException e) {
142                     throw new CanoniserException("Could not build document while marshalling fragment. This should never happen!", e);
143                 }
144                 m.marshal(element, doc);
145                 return doc.getDocumentElement();
146             } else {
147                 throw new CanoniserException("Missing JAXBContext");
148             }
149         } catch (final JAXBException e) {
150             throw new CanoniserException("Failed to add extension with name " + elementName, e);
151         }
152     }

245     /**
246      * Sets a value on a CPF element which can be read back using {@link #hasExtension}.
247      *
248      * @param attributes  the attributes of the CPF element to annotate
249      * @param name  {@link TypeAttribute#getName} of the attribute to flag
250      * @param value  whether to flag the attribute as present or absent
251      */
252     public static void flagExtension(final List<TypeAttribute> attributes, final String name, boolean value) {
253         if (value) {
254             // Check whether there's already an existing flag
255             for (TypeAttribute attribute : attributes) {
256                 if (name.equals(attribute.getName())) {
257                     attribute.setValue(null);
258                     return;  // already flagged, so nothing needs to be changed
259                 }
260             }
261 
262             // Didn't find an existing flag, so create and add one
263             TypeAttribute attribute = new TypeAttribute();
264             attribute.setName(name);
265             attributes.add(attribute);
266 
267         } else {
268             // Remove any existing flags
269             Iterator<TypeAttribute> i = attributes.iterator();
270             while (i.hasNext()) {
271                 if (name.equals(i.next().getName())) {
272                     i.remove();
273                 }
274             }
275         }
276     }

290     /**
291      * Sets a value on a CPF element which can be read back using {@link #getString}.
292      *
293      * @param attributes  the attributes of the CPF element to annotate
294      * @param name  {@link TypeAttribute#getName} of the attribute to flag
295      * @param value  stored text
296      */
297     public static void setString(final List<TypeAttribute> attributes, final String name, final String value) {
298 
299         // Remove any existing attributes for this name
300         Iterator<TypeAttribute> i = attributes.iterator();
301         while (i.hasNext()) {
302             if (name.equals(i.next().getName())) {
303                 i.remove();
304             }
305         }
306 
307         if (name != null) {
308             // Add an attribute
309             TypeAttribute attribute = new TypeAttribute();
310             attribute.setName(name);
311             attribute.setValue(value);
312             attributes.add(attribute);
313         }
314 
315     }

317     /**
318      * Get all matching extension attribute from a CPF Node.
319      *
320      * @param node
321      *            any CPF node
322      * @param name
323      *            name of the extension
324      * @return List of TypeAttribute
325      */
326     public static List<TypeAttribute> getExtensionAttributes(final NodeType node, final String name) {
327         List<TypeAttribute> attrList = new ArrayList<TypeAttribute>();
328         for (TypeAttribute attr :node.getAttribute()) {
329             if (name.equals(attr.getName()) || (YAWLSCHEMA_URL + "/" + name).equals(attr.getName())) {
330                 attrList.add(attr);
331             }
332         }
333         return attrList;
334     }

347     private static TypeAttribute getExtensionFromAttributes(final List<TypeAttribute> attributeList, final String name) {
348         for (TypeAttribute attr : attributeList) {
349             if (name.equals(attr.getName()) || (YAWLSCHEMA_URL + "/" + name).equals(attr.getName())) {
350                 return attr;
351             }
352         }
353         return null;
354     }

374     /**
375      * Get an extension attribute from a list of CPF attributes and unmarshals it to using the native namespace.
376      *
377      * @param attributes CPF list of attributes
378      * @param elementName of Extension
379      * @param expectedClass from native schema
380      * @param defaultValue if not found
381      * @return Object of expectedClass
382      */
383     public static <T> T getFromExtension(final List<TypeAttribute> attributes,
384                                          final String elementName,
385                                          final Class<T> expectedClass,
386                                          final T defaultValue,
387                                          final JAXBContext nativeContext) {
388 
389         TypeAttribute attr = getExtensionFromAttributes(attributes, elementName);
390         if (attr != null && attr.getAny() != null) {
391             try {
392                 return ExtensionUtils.unmarshalFragment(attr.getAny(), expectedClass, nativeContext);
393             } catch (CanoniserException e) {
394                 LOGGER.warn("Could unmarshal fragment from extension!", e);
395                 return defaultValue;
396             }
397         }
398         return defaultValue;
399     }

401     /**
402      * Get an extension element from an AnnotationsType.
403      *
404      * @param annotation
405      * @param elementName
406      * @param expectedClass
407      * @param defaultValue
408      * @param nativeContext
409      * @return an object of type T in any case
410      */
411     public static <T> T getFromAnnotationsExtension(final AnnotationType annotation,
412                                                     final String         elementName,
413                                                     final Class<T>       expectedClass,
414                                                     final T              defaultValue,
415                                                     final JAXBContext    nativeContext) {
416 
417         for (final Object extObj : annotation.getAny()) {
418             try {
419                 if (ExtensionUtils.isValidFragment(extObj, ExtensionUtils.YAWLSCHEMA_URL, elementName)) {
420                     return ExtensionUtils.unmarshalFragment(extObj, expectedClass, nativeContext);
421                 }
422             } catch (final CanoniserException e) {
423                 LOGGER.warn("Could not convert extension {} with type {}", new String[] { elementName, expectedClass.getSimpleName() }, e);
424             }
425         }
426         return defaultValue;
427     }
