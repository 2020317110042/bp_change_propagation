83      public void execute(ITransaction transaction, HttpServletRequest req, HttpServletResponse resp) throws IOException, LockFailedException {
84          LOGGER.trace("-- " + this.getClass().getName());
85  
86          // Retrieve the resources
87          String path = getCleanPath(getRelativePath(req));
88          String tempLockOwner = "doPropfind" + System.currentTimeMillis() + req.toString();
89          _depth = getDepth(req);
90  
91          if (_resourceLocks.lock(transaction, path, tempLockOwner, false, _depth, TEMP_TIMEOUT, TEMPORARY)) {
92              StoredObject so;
93              try {
94                  so = _store.getStoredObject(transaction, path);
95                  if (so == null) {
96                      resp.setContentType("text/xml; charset=UTF-8");
97                      resp.sendError(HttpServletResponse.SC_NOT_FOUND, req.getRequestURI());
98                      return;
99                  }
100 
101                 List<String> properties = null;
102                 path = getCleanPath(getRelativePath(req));
103                 int propertyFindType = FIND_ALL_PROP;
104                 Node propNode = null;
105 
106                 if (req.getContentLength() > 0) {
107                     DocumentBuilder documentBuilder = getDocumentBuilder();
108                     try {
109                         Document document = documentBuilder.parse(new InputSource(req.getInputStream()));
110                         Element rootElement = document.getDocumentElement();
111                         propNode = XMLHelper.findSubElement(rootElement, "prop");
112                         if (propNode != null) {
113                             propertyFindType = FIND_BY_PROPERTY;
114                         } else if (XMLHelper.findSubElement(rootElement, "propname") != null) {
115                             propertyFindType = FIND_PROPERTY_NAMES;
116                         } else if (XMLHelper.findSubElement(rootElement, "allprop") != null) {
117                             propertyFindType = FIND_ALL_PROP;
118                         }
119                     } catch (Exception e) {
120                         resp.sendError(WebDavStatus.SC_INTERNAL_SERVER_ERROR);
121                         return;
122                     }
123                 } else {
124                     propertyFindType = FIND_ALL_PROP;
125                 }
126 
127                 HashMap<String, String> namespaces = new HashMap<>();
128                 namespaces.put("DAV:", "D");
129 
130                 if (propertyFindType == FIND_BY_PROPERTY) {
131                     propertyFindType = 0;
132                     properties = XMLHelper.getPropertiesFromXML(propNode);
133                 }
134 
135                 resp.setStatus(WebDavStatus.SC_MULTI_STATUS);
136                 resp.setContentType("text/xml; charset=UTF-8");
137 
138                 XMLWriter generatedXML = new XMLWriter(resp.getWriter(), namespaces);
139                 generatedXML.writeXMLHeader();
140                 generatedXML.writeElement("DAV::multistatus", XMLWriter.OPENING);
141                 if (_depth == 0) {
142                     parseProperties(transaction, req, generatedXML, path, propertyFindType, properties, _mimeTyper.getMimeType(path));
143                 } else {
144                     recursiveParseProperties(transaction, path, req, generatedXML, propertyFindType, properties, _depth, _mimeTyper.getMimeType(path));
145                 }
146                 generatedXML .writeElement("DAV::multistatus", XMLWriter.CLOSING);
147                 generatedXML.sendData();
148             } catch (AccessDeniedException e) {
149                 resp.sendError(WebDavStatus.SC_FORBIDDEN);
150             } catch (WebDavException e) {
151                 LOGGER.warn("Sending internal error!");
152                 resp.sendError(WebDavStatus.SC_INTERNAL_SERVER_ERROR);
153             } catch (ServletException e) {
154                 e.printStackTrace();
155             } finally {
156                 _resourceLocks.unlockTemporaryLockedObjects(transaction, path, tempLockOwner);
157             }
158         } else {
159             Hashtable<String, Integer> errorList = new Hashtable<>();
160             errorList.put(path, WebDavStatus.SC_LOCKED);
161             sendReport(req, resp, errorList);
162         }
163     }

166     /**
167      * goes recursive through all folders. used by propfind
168      * 
169      * @param currentPath the current path
170      * @param req HttpServletRequest
171      * @param generatedXML
172      * @param propertyFindType
173      * @param properties
174      * @param depth depth of the propfind
175      * @throws WebDavException if an error in the underlying store occurs
176      */
177     private void recursiveParseProperties(ITransaction transaction, String currentPath, HttpServletRequest req, XMLWriter generatedXML,
178             int propertyFindType, List<String> properties, int depth, String mimeType) throws WebDavException {
179         parseProperties(transaction, req, generatedXML, currentPath, propertyFindType, properties, mimeType);
180         if (depth > 0) {
181             String[] names = _store.getChildrenNames(transaction, currentPath);
182             names = names == null ? new String[] {} : names;
183             String newPath;
184 
185             for (String name : names) {
186                 newPath = currentPath;
187                 if (!(newPath.endsWith("/"))) {
188                     newPath += "/";
189                 }
190                 newPath += name;
191                 recursiveParseProperties(transaction, newPath, req, generatedXML, propertyFindType, properties, depth - 1, mimeType);
192             }
193         }
194     }

196     /**
197      * Propfind helper method.
198      * 
199      * @param req The servlet request
200      * @param generatedXML  XML response to the Propfind request
201      * @param path  Path of the current resource
202      * @param type Propfind type
203      * @param propertiesVector If the propfind type is find properties by name, then this Vector contains those properties
204      */
205     private void parseProperties(ITransaction transaction, HttpServletRequest req, XMLWriter generatedXML, String path,
206             int type, List<String> propertiesVector, String mimeType) throws WebDavException {
207         StoredObject so = _store.getStoredObject(transaction, path);
208 
209         boolean isFolder = so.isFolder();
210         String creationdate = CREATION_DATE_FORMAT.format(so.getCreationDate());
211         String lastModified = LAST_MODIFIED_DATE_FORMAT.format(so.getLastModified());
212         String resourceLength = String.valueOf(so.getResourceLength());
213         generatedXML.writeElement("DAV::response", XMLWriter.OPENING);
214         String status = "HTTP/1.1 " + WebDavStatus.SC_OK + " " + WebDavStatus.getStatusText(WebDavStatus.SC_OK);
215 
216         generatedXML.writeElement("DAV::href", XMLWriter.OPENING);
217         generatedXML.writeText(rewriteUrl(getUri(req, path, isFolder)));
218         generatedXML.writeElement("DAV::href", XMLWriter.CLOSING);
219 
220         String resourceName = path;
221         int lastSlash = path.lastIndexOf('/');
222         if (lastSlash != -1) {
223             resourceName = resourceName.substring(lastSlash + 1);
224         }
225 
226         switch (type) {
227         case FIND_ALL_PROP:
228             generatedXML.writeElement("DAV::propstat", XMLWriter.OPENING);
229             generatedXML.writeElement("DAV::prop", XMLWriter.OPENING);
230             generatedXML.writeProperty("DAV::creationdate", creationdate);
231             generatedXML.writeElement("DAV::displayname", XMLWriter.OPENING);
232             generatedXML.writeData(resourceName);
233             generatedXML.writeElement("DAV::displayname", XMLWriter.CLOSING);
234             if (!isFolder) {
235                 generatedXML.writeProperty("DAV::getlastmodified", lastModified);
236                 generatedXML.writeProperty("DAV::getcontentlength", resourceLength);
237                 if (mimeType != null) {
238                     generatedXML.writeProperty("DAV::getcontenttype",  mimeType);
239                 }
240                 generatedXML.writeProperty("DAV::getetag", getETag(so));
241                 generatedXML.writeElement("DAV::resourcetype", XMLWriter.NO_CONTENT);
242             } else {
243                 generatedXML.writeElement("DAV::resourcetype", XMLWriter.OPENING);
244                 generatedXML.writeElement("DAV::collection", XMLWriter.NO_CONTENT);
245                 generatedXML.writeElement("DAV::resourcetype",  XMLWriter.CLOSING);
246             }
247 
248             writeSupportedLockElements(transaction, generatedXML, path);
249             writeLockDiscoveryElements(transaction, generatedXML, path);
250 
251             generatedXML.writeProperty("DAV::source", "");
252             generatedXML.writeElement("DAV::prop", XMLWriter.CLOSING);
253             generatedXML.writeElement("DAV::status", XMLWriter.OPENING);
254             generatedXML.writeText(status);
255             generatedXML.writeElement("DAV::status", XMLWriter.CLOSING);
256             generatedXML.writeElement("DAV::propstat", XMLWriter.CLOSING);
257             break;
258 
259         case FIND_PROPERTY_NAMES:
260             generatedXML.writeElement("DAV::propstat", XMLWriter.OPENING);
261             generatedXML.writeElement("DAV::prop", XMLWriter.OPENING);
262             generatedXML.writeElement("DAV::creationdate", XMLWriter.NO_CONTENT);
263             generatedXML.writeElement("DAV::displayname", XMLWriter.NO_CONTENT);
264             if (!isFolder) {
265                 generatedXML.writeElement("DAV::getcontentlanguage", XMLWriter.NO_CONTENT);
266                 generatedXML.writeElement("DAV::getcontentlength", XMLWriter.NO_CONTENT);
267                 generatedXML.writeElement("DAV::getcontenttype", XMLWriter.NO_CONTENT);
268                 generatedXML.writeElement("DAV::getetag", XMLWriter.NO_CONTENT);
269                 generatedXML.writeElement("DAV::getlastmodified", XMLWriter.NO_CONTENT);
270             }
271             generatedXML .writeElement("DAV::resourcetype", XMLWriter.NO_CONTENT);
272             generatedXML.writeElement("DAV::supportedlock",  XMLWriter.NO_CONTENT);
273             generatedXML.writeElement("DAV::source", XMLWriter.NO_CONTENT);
274             generatedXML.writeElement("DAV::prop", XMLWriter.CLOSING);
275             generatedXML.writeElement("DAV::status", XMLWriter.OPENING);
276             generatedXML.writeText(status);
277             generatedXML.writeElement("DAV::status", XMLWriter.CLOSING);
278             generatedXML.writeElement("DAV::propstat", XMLWriter.CLOSING);
279             break;
280 
281         case FIND_BY_PROPERTY:
282             List<String> propertiesNotFound = new ArrayList<>();
283 
284             generatedXML.writeElement("DAV::propstat", XMLWriter.OPENING);
285             generatedXML.writeElement("DAV::prop", XMLWriter.OPENING);
286 
287             for (String property : propertiesVector) {
288                 switch (property) {
289                     case "DAV::creationdate":
290                         generatedXML.writeProperty("DAV::creationdate", creationdate);
291                         break;
292                     case "DAV::displayname":
293                         generatedXML.writeElement("DAV::displayname", XMLWriter.OPENING);
294                         generatedXML.writeData(resourceName);
295                         generatedXML.writeElement("DAV::displayname", XMLWriter.CLOSING);
296                         break;
297                     case "DAV::getcontentlanguage":
298                         if (isFolder) {
299                             propertiesNotFound.add(property);
300                         } else {
301                             generatedXML.writeElement("DAV::getcontentlanguage",  XMLWriter.NO_CONTENT);
302                         }
303                         break;
304                     case "DAV::getcontentlength":
305                         if (isFolder) {
306                             propertiesNotFound.add(property);
307                         } else {
308                             generatedXML.writeProperty("DAV::getcontentlength", resourceLength);
309                         }
310                         break;
311                     case "DAV::getcontenttype":
312                         if (isFolder) {
313                             propertiesNotFound.add(property);
314                         } else {
315                             generatedXML.writeProperty("DAV::getcontenttype", mimeType);
316                         }
317                         break;
318                     case "DAV::getetag":
319                         if (isFolder || so.isNullResource()) {
320                             propertiesNotFound.add(property);
321                         } else {
322                             generatedXML.writeProperty("DAV::getetag", getETag(so));
323                         }
324                         break;
325                     case "DAV::getlastmodified":
326                         if (isFolder) {
327                             propertiesNotFound.add(property);
328                         } else {
329                             generatedXML.writeProperty("DAV::getlastmodified", lastModified);
330                         }
331                         break;
332                     case "DAV::resourcetype":
333                         if (isFolder) {
334                             generatedXML.writeElement("DAV::resourcetype", XMLWriter.OPENING);
335                             generatedXML.writeElement("DAV::collection", XMLWriter.NO_CONTENT);
336                             generatedXML.writeElement("DAV::resourcetype", XMLWriter.CLOSING);
337                         } else {
338                             generatedXML.writeElement("DAV::resourcetype", XMLWriter.NO_CONTENT);
339                         }
340                         break;
341                     case "DAV::source":
342                         generatedXML.writeProperty("DAV::source", "");
343                         break;
344                     case "DAV::supportedlock":
345                         writeSupportedLockElements(transaction, generatedXML, path);
346                         break;
347                     case "DAV::lockdiscovery":
348                         writeLockDiscoveryElements(transaction, generatedXML, path);
349                         break;
350                     default:
351                         propertiesNotFound.add(property);
352                         break;
353                 }
354             }
355 
356             generatedXML.writeElement("DAV::prop", XMLWriter.CLOSING);
357             generatedXML.writeElement("DAV::status", XMLWriter.OPENING);
358             generatedXML.writeText(status);
359             generatedXML.writeElement("DAV::status", XMLWriter.CLOSING);
360             generatedXML.writeElement("DAV::propstat", XMLWriter.CLOSING);
361 
362             Iterator<String> propertiesNotFoundList = propertiesNotFound.iterator();
363 
364             if (propertiesNotFoundList.hasNext()) {
365                 status = "HTTP/1.1 " + WebDavStatus.SC_NOT_FOUND + " " + WebDavStatus.getStatusText(WebDavStatus.SC_NOT_FOUND);
366                 generatedXML.writeElement("DAV::propstat", XMLWriter.OPENING);
367                 generatedXML.writeElement("DAV::prop", XMLWriter.OPENING);
368 
369                 while (propertiesNotFoundList.hasNext()) {
370                     generatedXML.writeElement(propertiesNotFoundList.next(), XMLWriter.NO_CONTENT);
371                 }
372 
373                 generatedXML.writeElement("DAV::prop", XMLWriter.CLOSING);
374                 generatedXML.writeElement("DAV::status", XMLWriter.OPENING);
375                 generatedXML.writeText(status);
376                 generatedXML.writeElement("DAV::status", XMLWriter.CLOSING);
377                 generatedXML.writeElement("DAV::propstat", XMLWriter.CLOSING);
378             }
379 
380             break;
381         }
382 
383         generatedXML.writeElement("DAV::response", XMLWriter.CLOSING);
384         so = null;
385     }

387     private String getUri(HttpServletRequest req, String path, boolean folder) {
388         String href = req.getContextPath();
389         String servletPath = req.getServletPath();
390         if (servletPath != null) {
391             if ((href.endsWith("/")) && (servletPath.startsWith("/"))) {
392                 href += servletPath.substring(1);
393             } else {
394                 href += servletPath;
395             }
396         }
397         if ((href.endsWith("/")) && (path.startsWith("/"))) {
398             href += path.substring(1);
399         } else {
400             href += path;
401         }
402         if ((folder) && (!href.endsWith("/"))) {
403             href += "/";
404         }
405         return href;
406     }

408     private void writeSupportedLockElements(ITransaction transaction, XMLWriter generatedXML, String path) {
409         LockedObject lo = _resourceLocks.getLockedObjectByPath(transaction, path);
410         generatedXML.writeElement("DAV::supportedlock", XMLWriter.OPENING);
411         if (lo == null) {

413             generatedXML.writeElement("DAV::lockscope", XMLWriter.OPENING);
414             generatedXML.writeElement("DAV::exclusive", XMLWriter.NO_CONTENT);
415             generatedXML.writeElement("DAV::lockscope", XMLWriter.CLOSING);
416             generatedXML.writeElement("DAV::locktype", XMLWriter.OPENING);
417             generatedXML.writeElement("DAV::write", XMLWriter.NO_CONTENT);
418             generatedXML.writeElement("DAV::locktype", XMLWriter.CLOSING);
419             generatedXML.writeElement("DAV::lockentry", XMLWriter.CLOSING);
420             generatedXML.writeElement("DAV::lockentry", XMLWriter.OPENING);
421             generatedXML.writeElement("DAV::lockscope", XMLWriter.OPENING);
422             generatedXML.writeElement("DAV::shared", XMLWriter.NO_CONTENT);
423             generatedXML.writeElement("DAV::lockscope", XMLWriter.CLOSING);
424             generatedXML.writeElement("DAV::locktype", XMLWriter.OPENING);
425             generatedXML.writeElement("DAV::write", XMLWriter.NO_CONTENT);
426             generatedXML.writeElement("DAV::locktype", XMLWriter.CLOSING);
427             generatedXML.writeElement("DAV::lockentry", XMLWriter.CLOSING);
428         } else {
429             if (lo.isShared()) {
430                 generatedXML.writeElement("DAV::lockentry", XMLWriter.OPENING);
431                 generatedXML.writeElement("DAV::lockscope", XMLWriter.OPENING);
432                 generatedXML.writeElement("DAV::shared", XMLWriter.NO_CONTENT);

436                 generatedXML.writeElement("DAV::locktype", XMLWriter.CLOSING);
437                 generatedXML.writeElement("DAV::lockentry", XMLWriter.CLOSING);

445     private void writeLockDiscoveryElements(ITransaction transaction, XMLWriter generatedXML, String path) {
446         LockedObject lo = _resourceLocks.getLockedObjectByPath(transaction, path);
447 
448         if (lo != null && !lo.hasExpired()) {
449             generatedXML.writeElement("DAV::lockdiscovery", XMLWriter.OPENING);
450             generatedXML.writeElement("DAV::activelock", XMLWriter.OPENING);
451             generatedXML.writeElement("DAV::locktype", XMLWriter.OPENING);
452             generatedXML.writeProperty("DAV::" + lo.getType());
453             generatedXML.writeElement("DAV::locktype", XMLWriter.CLOSING);
454             generatedXML.writeElement("DAV::lockscope", XMLWriter.OPENING);
455             if (lo.isExclusive()) {
456                 generatedXML.writeProperty("DAV::exclusive");
457             } else {
458                 generatedXML.writeProperty("DAV::shared");
459             }
460             generatedXML.writeElement("DAV::lockscope", XMLWriter.CLOSING);
461             generatedXML.writeElement("DAV::depth", XMLWriter.OPENING);
462             if (_depth == INFINITY) {
463                 generatedXML.writeText("Infinity");
464             } else {
465                 generatedXML.writeText(String.valueOf(_depth));
466             }
467             generatedXML.writeElement("DAV::depth", XMLWriter.CLOSING);
468 
469             String[] owners = lo.getOwner();
470             if (owners != null) {
471                 for (String owner : owners) {
472                     generatedXML.writeElement("DAV::owner", XMLWriter.OPENING);
473                     generatedXML.writeElement("DAV::href", XMLWriter.OPENING);
474                     generatedXML.writeText(owner);
475                     generatedXML.writeElement("DAV::href", XMLWriter.CLOSING);
476                     generatedXML.writeElement("DAV::owner", XMLWriter.CLOSING);
477                 }
478             } else {
479                 generatedXML.writeElement("DAV::owner", XMLWriter.NO_CONTENT);
480             }
481 
482             int timeout = (int) (lo.getTimeoutMillis() / 1000);
483             String timeoutStr = Integer.toString(timeout);
484             generatedXML.writeElement("DAV::timeout", XMLWriter.OPENING);
485             generatedXML.writeText("Second-" + timeoutStr);
486             generatedXML.writeElement("DAV::timeout", XMLWriter.CLOSING);
487 
488             String lockToken = lo.getID();
489 
490             generatedXML.writeElement("DAV::locktoken", XMLWriter.OPENING);
491             generatedXML.writeElement("DAV::href", XMLWriter.OPENING);
492             generatedXML.writeText("opaquelocktoken:" + lockToken);
493             generatedXML.writeElement("DAV::href", XMLWriter.CLOSING);
494             generatedXML.writeElement("DAV::locktoken", XMLWriter.CLOSING);
495             generatedXML.writeElement("DAV::activelock", XMLWriter.CLOSING);
496             generatedXML.writeElement("DAV::lockdiscovery", XMLWriter.CLOSING);
497 
498         } else {
499             generatedXML.writeElement("DAV::lockdiscovery", XMLWriter.NO_CONTENT);
500         }
501 
502         lo = null;
503     }
