74      public EditOneProcessController(MainController mainController, EditListProcessesController editListProcessesController,
75              ProcessSummaryType processType, VersionSummaryType versionType)
76              throws SuspendNotAllowedException, InterruptedException, ExceptionFormats {
77          mainC = mainController;
78          editListProcessesC = editListProcessesController;
79          process = processType;
80          version = versionType;
81  
82          chooseNativeW = (Window) Executions.createComponents("macros/choosenative.zul", null, null);
83          chooseNativeW.setTitle("Edit process " + process.getName() + ", " + version.getName() + ".");
84          Rows rows = (Rows) chooseNativeW.getFirstChild().getFirstChild().getFirstChild().getNextSibling();
85          Row warning = (Row) rows.getFirstChild();
86          Row nativeTypesR = (Row) warning.getNextSibling();
87          Row annotationR = (Row) nativeTypesR.getNextSibling();
88          Row readOnlyR = (Row) annotationR.getNextSibling();
89          Row paramsR = (Row) readOnlyR.getNextSibling();
90          Row buttonsR = (Row) paramsR.getNextSibling().getNextSibling();
91          nativeTypesLB = (Listbox) nativeTypesR.getFirstChild().getNextSibling();
92          annotationsLB = (Listbox) annotationR.getFirstChild().getNextSibling();
93          annotationOnlyCB = (Checkbox) readOnlyR.getFirstChild().getNextSibling();
94          Button okB = (Button) buttonsR.getFirstChild().getFirstChild();
95          Button cancelB = (Button) okB.getNextSibling();
96          Button cancelAllB = (Button) cancelB.getNextSibling();
97  
98          // enable cancelAll button if at least 1 process versions left.
99          cancelAllB.setVisible(editListProcessesC.getToEditList().size() > 0);
100 
101         // build native format listbox
102         HashMap<String, String> formats = mainC.getNativeTypes();
103         Set<String> extensions = formats.keySet();
104         List<String> sorted = CollectionUtil.asSortedList(extensions);
105         Iterator<String> it = sorted.iterator();
106         Listitem cbi;
107         while (it.hasNext()) {
108             String label = formats.get(it.next());
109             if (!label.equals("AML fragment")&&!label.equals("XPDL 2.2")) {
110                 cbi = new Listitem();
111                 nativeTypesLB.appendChild(cbi);
112                 cbi.setLabel(label);
113                 if (process.getOriginalNativeType() != null && process.getOriginalNativeType().equals(cbi.getLabel())) {
114                     cbi.setSelected(true);
115                 }
116             }
117         }
118         if (nativeTypesLB.getSelectedCount() == 0) {
119             nativeTypesLB.setSelectedIndex(0);
120         }
121 
122         // Build list of annotations associated with the process version
123         annotationsLB.setDisabled(true);
124         annotationOnlyCB.setDisabled(true);
125         if (version.getAnnotations().size() > 0) {
126             for (int i = 0; i < version.getAnnotations().size(); i++) {
127                 String native_type = version.getAnnotations().get(i).getNativeType();
128                 for (int k = 0; k < version.getAnnotations().get(i).getAnnotationName().size(); k++) {
129                     cbi = new Listitem();
130                     annotationsLB.appendChild(cbi);
131                     cbi.setLabel(version.getAnnotations().get(i).getAnnotationName().get(k) + " (" + native_type + ")");
132                     cbi.setValue(version.getAnnotations().get(i).getAnnotationName().get(k));
133                     if (Constants.INITIAL_ANNOTATION.compareTo((String) cbi.getValue()) == 0 || version.getAnnotations().size() == 1) {
134                         cbi.setSelected(true);
135                     }
136                 }
137             }
138             noAnnotationI = new Listitem();
139             noAnnotationI.setLabel(Constants.NO_ANNOTATIONS);
140             annotationsLB.appendChild(noAnnotationI);
141         } else {
142             noAnnotationI = new Listitem();
143             noAnnotationI.setLabel(Constants.NO_ANNOTATIONS);
144             noAnnotationI.setSelected(true);
145             annotationsLB.appendChild(noAnnotationI);
146         }
147 
148         // setup the class that will help build the parameters for a canoniser
149         pluginPropertiesHelper = new PluginPropertiesHelper(getService(), (Grid) this.chooseNativeW.getFellow("canoniserPropertiesGrid"));
150 
151         // setup the events
152         nativeTypesLB.addEventListener(Events.ON_SELECT,
153                 new EventListener<Event>() {
154                     public void onEvent(Event event) throws Exception {
155                         syncAnnotationLB();
156                         updateCanoniserParameters();
157                     }
158                 });
159         annotationsLB.addEventListener(Events.ON_SELECT,
160                 new EventListener<Event>() {
161                     public void onEvent(Event event) throws Exception {
162                         syncReadOnlyR();
163                     }
164                 });
165         annotationOnlyCB.addEventListener(Events.ON_CHECK,
166                 new EventListener<Event>() {
167                     public void onEvent(Event event) throws Exception {
168                         syncListboxe();
169                     }
170                 });
171         okB.addEventListener(Events.ON_CLICK,
172                 new EventListener<Event>() {
173                     public void onEvent(Event event) throws Exception {
174                         editProcess();
175                     }
176                 });
177         chooseNativeW.addEventListener(Events.ON_OK,
178                 new EventListener<Event>() {
179                     public void onEvent(Event event) throws Exception {
180                         editProcess();
181                     }
182                 });
183         cancelB.addEventListener(Events.ON_CLICK,
184                 new EventListener<Event>() {
185                     public void onEvent(Event event) throws Exception {
186                         cancel();
187                     }
188                 });
189         cancelAllB.addEventListener(Events.ON_CLICK,
190                 new EventListener<Event>() {
191                     public void onEvent(Event event) throws Exception {
192                         cancelAll();
193                     }
194                 });
195         chooseNativeW.doModal();
196     }

213     /**
214      * If the native type is selected is the same as process original type then disable the annotations lb. But if the annotation format type isn't the same as the process native type (i.e. someone saved a bpmn as an epml) the don't disable.
215      */
216     protected void syncAnnotationLB() {
217         if (nativeTypesLB.getSelectedItem() != null) {
218             Listitem selected = nativeTypesLB.getSelectedItem();
219             if (process.getOriginalNativeType() != null && process.getOriginalNativeType().equals(selected.getLabel())) {
220                 Listitem annotation = annotationsLB.getSelectedItem();
221                 if (annotation != null && annotation.getLabel().contains(process.getOriginalNativeType())) {
222                     annotationsLB.setDisabled(true);
223                     annotationOnlyCB.setDisabled(true);
224                 } else {
225                     annotationsLB.setDisabled(false);
226                     annotationOnlyCB.setDisabled(false);
227                 }
228             } else {
229                 annotationsLB.setDisabled(false);
230                 annotationOnlyCB.setDisabled(noAnnotationI.isSelected());
231             }
232         }
233     }

248     protected void editProcess() throws Exception {
249         if (nativeTypesLB.getSelectedItem() == null || nativeTypesLB.getSelectedItem() != null
250                 && nativeTypesLB.getSelectedItem().getLabel().compareTo("") == 0) {
251             Messagebox.show(CHOOSE_NATIVE, ATTENTION, Messagebox.OK, Messagebox.ERROR);
252         } else {
253             String readOnly;
254             Listitem cbi = nativeTypesLB.getSelectedItem();
255             String nativeType = cbi.getLabel();
256             String annotation = null;
257             if (annotationsLB.getSelectedItem() != null && Constants.NO_ANNOTATIONS.compareTo(annotationsLB.getSelectedItem().getLabel()) != 0) {
258                 annotation = annotationsLB.getSelectedItem().getValue();
259             }
260             if (annotationOnlyCB.isChecked()) {
261                 readOnly = TRUE;
262             } else {
263                 readOnly = FALSE;
264             }
265 
266             mainC.editProcess(process, version, nativeType, annotation, readOnly,
267                     pluginPropertiesHelper.readPluginProperties(Canoniser.DECANONISE_PARAMETER));
268             editListProcessesC.deleteFromToBeEdited(this);
269             closePopup();
270         }
271     }

287     private void readCanoniserInfos(final String nativeType) throws InterruptedException {
288         try {
289             canoniserInfos = getService().readCanoniserInfo(nativeType);
290 
291             if (canoniserInfos.size() >= 1) {
292                 List<String> canoniserNames = new ArrayList<>();
293                 for (PluginInfo cInfo: canoniserInfos) {
294                     canoniserNames.add(cInfo.getName());
295                 }
296 
297                 Row canoniserSelectionRow = (Row) this.chooseNativeW.getFellow("canoniserSelectionRow");
298                 if (canoniserCB != null) {
299                     canoniserCB.detach();
300                 }
301                 canoniserCB = new SelectDynamicListController(canoniserNames);
302                 canoniserCB.setAutodrop(true);
303                 canoniserCB.setWidth("85%");
304                 canoniserCB.setHeight("100%");
305                 canoniserCB.setAttribute("hflex", "1");
306                 canoniserCB.setSelectedIndex(0);
307                 canoniserSelectionRow.appendChild(canoniserCB);
308 
309                 canoniserCB.addEventListener("onSelect", new EventListener<Event>() {
310                     @Override
311                     public void onEvent(final Event event) throws Exception {
312                         if (event instanceof SelectEvent) {
313                             String selectedCanoniser = ((SelectEvent) event).getSelectedItems().iterator().next().toString();
314                             for (PluginInfo info: canoniserInfos) {
315                                 if (info.getName().equals(selectedCanoniser)) {
316                                     pluginPropertiesHelper.showPluginProperties(info, Canoniser.DECANONISE_PARAMETER);
317                                 }
318                             }
319                         }
320                     }
321                 });
322 
323                 PluginInfo canoniserInfo = canoniserInfos.iterator().next();
324                 pluginPropertiesHelper.showPluginProperties(canoniserInfo, Canoniser.DECANONISE_PARAMETER);
325 
326             } else {
327                 Messagebox.show(MessageFormat.format("Import failed (No Canoniser found for native type {0})", nativeType), "Attention", Messagebox.OK, Messagebox.ERROR);
328                 cancel();
329             }
330         } catch (Exception e) {
331             Messagebox.show("Reading Canoniser info failed (" + e.getMessage() + ")", "Attention", Messagebox.OK, Messagebox.ERROR);
332         }
333     }
