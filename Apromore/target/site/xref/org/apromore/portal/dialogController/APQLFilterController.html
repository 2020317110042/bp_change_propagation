21  package org.apromore.portal.dialogController;
22  
23  import org.apromore.model.ProcessSummariesType;
24  import org.zkoss.zk.ui.Executions;
25  import org.zkoss.zk.ui.event.Event;
26  import org.zkoss.zk.ui.event.EventListener;
27  import org.zkoss.zul.Button;
28  import org.zkoss.zul.Grid;
29  import org.zkoss.zul.Messagebox;
30  import org.zkoss.zul.Textbox;
31  
32  public class APQLFilterController extends BaseFilterController {
33  
34      private static final long serialVersionUID = -7879730951994569217L;
35  
36      public APQLFilterController(MainController mainController) {
37          super(mainController);
38  
39          Grid filterGrid = ((Grid) Executions.createComponents("macros/filter/apqlFilter.zul", getMainController(), null));
40  
41          final Textbox apqlQuery = (Textbox) filterGrid.getFellow("apqlQuery");
42          Button submitButton = (Button) filterGrid.getFellow("submitButton");
43  
44          submitButton.addEventListener("onClick", new EventListener<Event>() {
45              @Override
46              public void onEvent(final Event event) throws Exception {
47                  runAPQL(apqlQuery.getValue());
48              }
49          });
50  
51          appendChild(filterGrid);
52      }
53  
54  
55      private void runAPQL(String query) {
56          try {
57              ProcessSummariesType processSummaries = getService().runAPQLExpression(query);
58              int nbAnswers = processSummaries.getProcessSummary().size();
59              String message = "Search returned " + nbAnswers;
60              if (nbAnswers > 1) {
61                  message += " processes.";
62              } else {
63                  message += " process.";
64              }
65              getMainController().displayMessage(message);
66              getMainController().displayProcessSummaries(processSummaries, true);
67          } catch (Exception e)  {
68              Messagebox.show(e.getMessage(), "Error", Messagebox.OK, Messagebox.EXCLAMATION);
69          }
70      }

72  }
