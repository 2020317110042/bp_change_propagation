323     private boolean validateFields() {
324         boolean valid = true;
325         String message = "";
326         String title = "Missing Fields";
327 
328         Version newVersion = new Version(versionNumber.getText());
329         Version curVersion = new Version(editSession.getCurrentVersionNumber());
330         try {
331             if (this.save) {
332                 if (newVersion.compareTo(curVersion) < 0) {
333                     valid = false;
334                     message = message + "New Version number has to be greater than " + this.editSession.getCurrentVersionNumber();
335                     title = "Wrong Version Number";
336                 }
337                 if (this.branchName.getText().equals("") || this.branchName.getText() == null) {
338                     valid = false;
339                     message = message + "Branch Name cannot be empty";
340                     title = "Branch Name Empty";
341                 }
342             } else {
343                 if (this.modelName.getText().equals("") || this.modelName.getText() == null) {
344                     valid = false;
345                     message = message + "Model Name cannot be empty";
346                     title = "Model Name Empty";
347                 }
348                 if (this.modelName.getText().equals(this.editSession.getProcessName())) {
349                     valid = false;
350                     message = message + "Model Name has to be different from " + this.editSession.getProcessName();
351                     title = "Same Model Name";
352                 }
353             }
354             if (this.versionNumber.getText().equals("") || this.versionNumber.getText() == null) {
355                 valid = false;
356                 message = message + "Version Number cannot be empty";
357                 title = "Version Number Empty";
358             }
359             if (!message.equals("")) {
360                 Messagebox.show(message, title, Messagebox.OK, Messagebox.INFORMATION);
361             }
362         } catch (Exception e) {
363             valid = false;
364         }
365         return valid;
366     }

396     private String createNewBranchName(String currBranchName) {
397         String branchName;
398         if (currBranchName.equalsIgnoreCase("Main") || !currBranchName.matches("B[0-9]+")) {
399             branchName = "B1";
400         } else {
401             Integer branchVersionNumber = 0;
402             Matcher matcher = Pattern.compile("\\d+").matcher(currBranchName);
403             if (matcher.find()) {
404                 branchVersionNumber = Integer.valueOf(matcher.group());
405             }
406             branchName = "B" + branchVersionNumber + 1;
407         }
408         return branchName;
409     }
