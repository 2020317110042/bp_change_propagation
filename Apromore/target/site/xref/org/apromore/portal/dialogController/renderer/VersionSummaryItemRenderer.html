42  public class VersionSummaryItemRenderer implements ListitemRenderer {
43  
44      private MainController mainController;
45  
46  
47      public VersionSummaryItemRenderer(MainController main) {
48          this.mainController = main;
49      }
50  
51      /*
52       * (non-Javadoc)
53       * @see org.zkoss.zul.ListitemRenderer#render(org.zkoss.zul.Listitem, java.lang.Object, int)
54       */
55      @Override
56      public void render(Listitem listItem, Object obj, int index) {
57          renderVersionSummary(listItem, (VersionDetailType) obj);
58      }
59  
60      private void renderVersionSummary(final Listitem listItem, final VersionDetailType data) {
61          listItem.appendChild(renderVersionName(data.getVersion()));
62          listItem.appendChild(renderVersionVersion(data.getVersion()));
63          listItem.appendChild(renderVersionLastUpdate(data.getVersion()));
64          listItem.appendChild(renderVersionAnnotations(data.getVersion()));
65  
66          listItem.addEventListener(Events.ON_DOUBLE_CLICK, new EventListener<Event>() {
67              @Override
68              public void onEvent(Event event) throws Exception {
69                  AnnotationsType annotation = getLastestAnnotation(data.getVersion().getAnnotations());
70                  if (annotation != null) {
71                      mainController.editProcess(data.getProcess(), data.getVersion(), getNativeType(annotation.getNativeType()),
72                          annotation.getAnnotationName().get(0), "false", new HashSet<RequestParameterType<?>>());
73                  } else {
74                      mainController.editProcess(data.getProcess(), data.getVersion(), getNativeType(data.getProcess().getOriginalNativeType()),
75                              null, "false", new HashSet<RequestParameterType<?>>());
76                  }
77              }
78  
79              /* Sometimes we have merged models with no native type, we should give them a default so they can be edited. */
80              private String getNativeType(String origNativeType) {
81                  String nativeType = origNativeType;
82                  if (origNativeType == null || origNativeType.isEmpty()) {
83                      nativeType = "BPMN 2.0";
84                  }
85                  return nativeType;
86              }
87          });
88      }
89  
90      private Component renderVersionVersion(VersionSummaryType version) {
91          return wrapIntoListCell(new Label(version.getVersionNumber()));
92      }
93  
94      private Component renderVersionAnnotations(VersionSummaryType version) {
95          Listbox annotationLB = new Listbox();
96          Listitem annotationsI = new Listitem();
97          annotationLB.setMold("select");
98          annotationLB.setRows(1);
99          annotationLB.setWidth("100%");
100         annotationLB.setStyle(Constants.UNSELECTED_VERSION);
101         if (version.getAnnotations().size() > 0) {
102             for (int i = 0; i < version.getAnnotations().size(); i++) {
103                 String language = version.getAnnotations().get(i).getNativeType();
104                 for (int k = 0; k < version.getAnnotations().get(i).getAnnotationName().size(); k++) {
105                     annotationLB.appendChild(annotationsI);
106                     annotationsI.setLabel(version.getAnnotations().get(i).getAnnotationName().get(k) + " (" + language + ")");
107                 }
108             }
109             annotationLB.selectItem(annotationLB.getItemAtIndex(version.getAnnotations().size() - 1));
110         } else {
111             annotationLB.appendChild(annotationsI);
112             annotationsI.setLabel("No Annotations Found!");
113             annotationLB.selectItem(annotationLB.getItemAtIndex(0));
114         }
115         return wrapIntoListCell(annotationLB);
116     }
117 
118     private Component renderVersionLastUpdate(VersionSummaryType version) {
119         if (version.getLastUpdate() == null || version.getLastUpdate().isEmpty()) {
120             return wrapIntoListCell(new Label(version.getCreationDate()));
121         } else {
122             return wrapIntoListCell(new Label(version.getLastUpdate()));
123         }
124     }
125 
126     private Component renderVersionName(VersionSummaryType version) {
127         return wrapIntoListCell(new Label(version.getName()));
128     }
129 
130     private Listcell wrapIntoListCell(Component cp) {
131         Listcell lc = new Listcell();
132         lc.appendChild(cp);
133         return lc;
134     }
135 
136     private AnnotationsType getLastestAnnotation(List<AnnotationsType> annotations) {
137         if (annotations.size() > 0 && annotations.get(annotations.size() - 1) != null) {
138             return annotations.get(annotations.size() - 1);
139         }
140         return null;
141     }
