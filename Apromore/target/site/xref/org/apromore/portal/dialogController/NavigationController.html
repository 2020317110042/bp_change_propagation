21  package org.apromore.portal.dialogController;
22  
23  import java.util.Collection;
24  
25  import org.apromore.portal.common.FolderTree;
26  import org.apromore.portal.common.FolderTreeModel;
27  import org.apromore.portal.common.FolderTreeRenderer;
28  import org.zkoss.zk.ui.Component;
29  import org.zkoss.zk.ui.event.Event;
30  import org.zkoss.zk.ui.event.EventListener;
31  import org.zkoss.zk.ui.event.Events;
32  import org.zkoss.zul.Button;
33  import org.zkoss.zul.Tree;
34  import org.zkoss.zul.Treeitem;
35  import org.zkoss.zul.Window;
36  
37  public class                                         NavigationController extends BaseController {
38  
39      private MainController mainC;
40      private Tree tree;
41  
42      public NavigationController(MainController newMainC) throws Exception {
43          mainC = newMainC;
44  
45          Window treeW = (Window) mainC.getFellow("navigationcomp").getFellow("treeW");
46          treeW.setContentStyle("background-image: none; background-color: white");
47  
48          tree = (Tree) treeW.getFellow("tree");
49          tree.setStyle("background-image: none; background-color: white");
50  
51          Button expandBtn = (Button) treeW.getFellow("expand");
52          Button contractBtn = (Button) treeW.getFellow("contract");
53          expandBtn.addEventListener(Events.ON_CLICK, new EventListener<Event>() {
54              @Override
55              public void onEvent(Event event) throws Exception {
56                  doCollapseExpandAll(tree, true);
57              }
58          });
59          contractBtn.addEventListener(Events.ON_CLICK, new EventListener<Event>() {
60              @Override
61              public void onEvent(Event event) throws Exception {
62                  doCollapseExpandAll(tree, false);
63              }
64          });
65      }
66  
67  
68      /**
69       * Loads the workspace.
70       */
71      public void loadWorkspace() {
72          FolderTreeModel model = new FolderTreeModel(new FolderTree(false).getRoot());
73          tree.setItemRenderer(new FolderTreeRenderer(mainC));
74          tree.setModel(model);
75      }
76  
77      /* Expand or Collapse the tree. */
78      private void doCollapseExpandAll(Component component, boolean open) {
79          if (component instanceof Treeitem) {
80              Treeitem treeitem = (Treeitem) component;
81              treeitem.setOpen(open);
82          }
83          Collection<?> children = component.getChildren();
84          if (children != null) {
85              for (Object child : children) {
86                  doCollapseExpandAll((Component) child, open);
87  
88              }
89          }
90      }

92  }
