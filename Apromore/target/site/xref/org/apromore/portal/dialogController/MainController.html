205     public void loadWorkspace() {
206         updateActions();
207 
208         setHeaderText((Toolbarbutton) this.getFellow("releaseNotes"));
209         String userId = UserSessionManager.getCurrentUser().getId();
210         int currentParentFolderId = UserSessionManager.getCurrentFolder() == null || UserSessionManager.getCurrentFolder().getId() == 0 ? 0 : UserSessionManager.getCurrentFolder().getId();
211 
212         this.loadTree();
213 
214         List<FolderType> folders = this.getService().getSubFolders(userId, currentParentFolderId);
215         if (UserSessionManager.getCurrentFolder() != null) {
216             FolderType folder = UserSessionManager.getCurrentFolder();
217             folder.getFolders().clear();
218             for (FolderType newFolder : folders) {
219                 folder.getFolders().add(newFolder);
220             }
221             UserSessionManager.setCurrentFolder(folder);
222         }
223     }

255     // disable/enable features depending on user status
256     public void updateActions() {
257         Boolean connected = UserSessionManager.getCurrentUser() != null;
258 
259         // disable/enable menu items in menu bar
260         for (Component C : this.menu.getMenuB().getFellows()) {
261             if (C.getClass().getName().compareTo("org.zkoss.zul.Menuitem") == 0) {
262                 if (C.getId().equals("designPatternCr")) {
263                     ((Menuitem) C).setDisabled(true);
264                 } else if (C.getId().equals("designReference")) {
265                     ((Menuitem) C).setDisabled(true);
266                 } else if (C.getId().equals("designPatternCo")) {
267                     ((Menuitem) C).setDisabled(true);
268 //              } else if (C.getId().equals("designConfiguration")) {
269 //                  ((Menuitem) C).setDisabled(true);
270                 } else if (C.getId().equals("designExtension")) {
271                     ((Menuitem) C).setDisabled(true);
272                 } else {
273                     ((Menuitem) C).setDisabled(!connected);
274                 }
275             }
276         }
277     }

279     public void reloadProcessSummaries() {
280         ProcessSummariesType processSummaries = new ProcessSummariesType();
281         processSummaries.getProcessSummary().clear();
282         UserType user = UserSessionManager.getCurrentUser();
283         FolderType currentFolder = UserSessionManager.getCurrentFolder();
284 
285         processSummaries = getService().getProcesses(user.getId(), currentFolder == null ? 0 : currentFolder.getId());
286 
287         String message = processSummaries.getProcessSummary().size() + " out of " + processSummaries.getTotalProcessCount();
288         if (processSummaries.getTotalProcessCount() > 1) {
289             message += " processes.";
290         } else {
291             message += " process.";
292         }
293 
294         this.displayMessage(message);
295         this.simplesearch.clearSearches();
296         this.displayProcessSummaries(processSummaries, false);
297 
298         loadWorkspace();
299     }

312     /**
313      * Send request to Manager: deleted process versions given as parameter
314      * @param processVersions a selection of process versions to delete.
315      * @throws InterruptedException
316      */
317     public void deleteProcessVersions(final Map<ProcessSummaryType, List<VersionSummaryType>> processVersions) throws InterruptedException {
318         try {
319             getService().deleteProcessVersions(processVersions);
320             switchToProcessSummaryView();
321             this.baseListboxController.refreshContent();
322             String message;
323             int nb = 0;
324 
325             // to count how many process version(s) deleted
326             Collection<List<VersionSummaryType>> sumTypes = processVersions.values();
327             for (List<VersionSummaryType> sumType : sumTypes) {
328                 nb += sumType.size();
329             }
330             if (nb > 1) {
331                 message = nb + " process versions deleted.";
332             } else {
333                 message = "One process version deleted.";
334             }
335             displayMessage(message);
336         } catch (Exception e) {
337             e.printStackTrace();
338             Messagebox.show("Deletion failed (" + e.getMessage() + ")", "Attention", Messagebox.OK, Messagebox.ERROR);
339         }
340     }

342     /**
343      * Call editor to edit process version whose id is processId, name is
344      * processName and version name is version. nativeType identifies language
345      * to be used to edit the process version. If annotation is instantiated, it
346      * identifies the annotation file to be used. If readOnly=1, annotations
347      * only are editable.
348      *
349      * @param process the process summary
350      * @param version the version of the process
351      * @param nativeType the native type of the process
352      * @param annotation the annotation of that process
353      * @param readOnly is this model readonly or not
354      * @param requestParameterTypes request parameters types.
355      * @throws InterruptedException
356      */
357     public void editProcess(final ProcessSummaryType process, final VersionSummaryType version, final String nativeType, final String annotation,
358             final String readOnly, Set<RequestParameterType<?>> requestParameterTypes) throws InterruptedException {
359         String instruction = "";
360 
361         EditSessionType editSession = new EditSessionType();
362         editSession.setDomain(process.getDomain());
363         editSession.setNativeType(nativeType.equals("XPDL 2.2")?"BPMN 2.0":nativeType);
364         editSession.setProcessId(process.getId());
365         editSession.setProcessName(process.getName());
366         editSession.setUsername(UserSessionManager.getCurrentUser().getUsername());
367         editSession.setPublicModel(process.isMakePublic());
368         editSession.setOriginalBranchName(version.getName());
369         editSession.setOriginalVersionNumber(version.getVersionNumber());
370         editSession.setCurrentVersionNumber(version.getVersionNumber());
371         editSession.setMaxVersionNumber(findMaxVersion(process));
372 
373         editSession.setCreationDate(version.getCreationDate());
374         editSession.setLastUpdate(version.getLastUpdate());
375         if (annotation == null) {
376             editSession.setWithAnnotation(false);
377         } else {
378             editSession.setWithAnnotation(true);
379             editSession.setAnnotation(annotation);
380         }
381 
382         try {
383             String id = UUID.randomUUID().toString();
384             SignavioSession session = new SignavioSession(editSession, this, process, version, requestParameterTypes);
385             UserSessionManager.setEditSession(id, session);
386 
387             String url = "macros/openModelInSignavio.zul?id=" + id;
388             instruction += "window.open('" + url + "');";
389 
390             Clients.evalJavaScript(instruction);
391         } catch (Exception e) {
392             Messagebox.show("Cannot edit " + process.getName() + " (" + e.getMessage() + ")", "Attention", Messagebox.OK, Messagebox.ERROR);
393         }
394     }

423     /**

462     /**
463      * Show the messages we get back from plugins.
464      * @param messages the messages to display to the user.
465      * @throws InterruptedException if the communication was interrupted for any reason.
466      */
467     public void showPluginMessages(final PluginMessages messages) throws InterruptedException {
468         if (messages != null) {
469             StringBuilder sb = new StringBuilder();
470             Iterator<PluginMessage> iter = messages.getMessage().iterator();
471             while (iter.hasNext()) {
472                 sb.append(iter.next().getValue());
473                 if (iter.hasNext()) {
474                     sb.append("\n\n");
475                 }
476             }
477             if (sb.length() > 0) {
478                 Messagebox.show(sb.toString(), "Plugin Warnings", Messagebox.OK, Messagebox.EXCLAMATION);
479             }
480         }
481     }

582     @Command
583     protected void displayInstalledPlugins() throws InterruptedException {
584         final Window pluginWindow = (Window) Executions.createComponents("macros/pluginInfo.zul", this, null);
585         Listbox infoListBox = (Listbox) pluginWindow.getFellow("pluginInfoListBox");
586         try {
587             List<PluginInfo> installedPlugins = new ArrayList<>(getService().readInstalledPlugins(null));
588             infoListBox.setModel(new ListModelList<>(installedPlugins, false));
589             infoListBox.setItemRenderer(new ListitemRenderer() {
590                 @Override
591                 public void render(final Listitem item, final Object data, final int index) throws Exception {
592                     if (data != null && data instanceof PluginInfo) {
593                         PluginInfo info = (PluginInfo) data;
594                         item.appendChild(new Listcell(info.getName()));
595                         item.appendChild(new Listcell(info.getVersion()));
596                         item.appendChild(new Listcell(info.getType()));
597                         Listcell dCell = new Listcell();
598                         Label dLabel = new Label(info.getDescription());
599                         dLabel.setWidth("100px");
600                         dLabel.setMultiline(true);
601                         dCell.appendChild(dLabel);
602                         item.appendChild(dCell);
603                         item.appendChild(new Listcell(info.getAuthor()));
604                         item.appendChild(new Listcell(info.getEmail()));
605                     }
606                 }
607             });
608             Button buttonOk = (Button) pluginWindow.getFellow("ok");
609             buttonOk.addEventListener("onClick", new EventListener<Event>() {
610                 @Override
611                 public void onEvent(final Event event) throws Exception {
612                     pluginWindow.detach();
613                 }
614             });
615             pluginWindow.doModal();
616         } catch (Exception e) {
617             Messagebox.show("Error retrieving installed Plugins: "+e.getMessage(), "Error", Messagebox.OK,
618                     Messagebox.ERROR);
619         }
620     }

695     /* From a list of version summary types find the max version number. */
696     private String findMaxVersion(ProcessSummaryType process) {
697         Version versionNum;
698         Version max = new Version(0, 0);
699         for (VersionSummaryType version : process.getVersionSummaries()) {
700             versionNum = new Version(version.getVersionNumber());
701             if (versionNum.compareTo(max) > 0) {
702                 max = versionNum;
703             }
704         }
705         return max.toString();
706     }
