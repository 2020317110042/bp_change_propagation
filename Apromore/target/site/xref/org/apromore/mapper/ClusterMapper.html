66      public static ClusterType convertClusterToClusterType(org.apromore.service.model.Cluster c) {
67          ClusterType ct = new ClusterType();
68          ct.setClusterLabel(Integer.toString(c.getCluster().getSize()));
69          ct.setClusterId(c.getCluster().getId().toString());
70          ct.setClusterSize(c.getCluster().getSize());
71          ct.setMedoidId(c.getCluster().getMedoidId());
72          ct.setAvgFragmentSize(c.getCluster().getAvgFragmentSize());
73          ct.setStandardizationEffort(c.getCluster().getStandardizingEffort());
74          ct.setRefactoringGain(c.getCluster().getRefactoringGain());
75          ct.setBCR(c.getCluster().getBCR());
76  
77          List<MemberFragment> fs = c.getFragments();
78          for (MemberFragment f : fs) {
79              FragmentData fd = new FragmentData();
80              fd.setFragmentId(f.getFragmentId());
81              fd.setFragmentSize(f.getFragmentSize());
82              fd.setDistance(f.getDistance());
83  
84              List<ProcessAssociation> pas = f.getProcessAssociations();
85              for (ProcessAssociation pa : pas) {
86                  ProcessAssociationsType patype = new ProcessAssociationsType();
87                  patype.setProcessId(pa.getProcessId());
88                  patype.setProcessName(pa.getProcessName());
89                  patype.setBranchName(pa.getProcessBranchName());
90                  patype.setProcessVersionId(pa.getProcessVersionId());
91                  patype.setProcessVersionNumber(pa.getProcessVersionNumber());
92                  fd.getProcessAssociations().add(patype);
93              }
94  
95              ct.getFragments().add(fd);
96          }
97  
98          return ct;
99      }

112     /**
113      * @param clusterSettingsType
114      * @return
115      */
116     public static ClusterSettings convertClusterSettingsTypeToClusterSettings(ClusterSettingsType clusterSettingsType) {
117         ClusterSettings clusterSettings = new ClusterSettings();
118         clusterSettings.setAlgorithm(clusterSettingsType.getAlgorithm());
119         List<ClusteringParameterType> params = clusterSettingsType.getClusteringParams();
120         for (ClusteringParameterType param : params) {
121             if ("maxdistance".equalsIgnoreCase(param.getParamName())) {
122                 double maxDistance = Double.parseDouble(param.getParmaValue());
123                 clusterSettings.setMaxNeighborGraphEditDistance(maxDistance);
124             }
125         }
126         ConstrainedProcessIdsType cpidsType = clusterSettingsType.getConstrainedProcessIds();
127         if (cpidsType != null) {
128             List<Integer> constrainedProcessIds = cpidsType.getProcessId();
129             if (constrainedProcessIds != null && !constrainedProcessIds.isEmpty()) {
130                 clusterSettings.setConstrainedProcessIds(constrainedProcessIds);
131             }
132         }
133 
134         return clusterSettings;
135     }

153     public static PairDistancesType convertPairDistancesToPairDistancesType(Map<FragmentPair, Double> pairDistances) {
154         PairDistancesType pdt = new PairDistancesType();
155         for (FragmentPair pair : pairDistances.keySet()) {
156             PairDistanceType p = new PairDistanceType();
157             p.setFragmentId1(pair.getFid1());
158             p.setFragmentId2(pair.getFid2());
159             p.setDistance(pairDistances.get(pair));
160             pdt.getPairDistance().add(p);
161         }
162         return pdt;
163     }
