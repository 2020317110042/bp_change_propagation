21  package org.apromore.dao;
22  
23  import org.apromore.dao.model.Node;
24  import org.springframework.data.jpa.repository.JpaRepository;
25  import org.springframework.data.jpa.repository.Query;
26  import org.springframework.stereotype.Repository;
27  
28  import java.util.List;
29  
30  /**
31   * Interface domain model Data access object Node.
32   *
33   * @author <a href="mailto:cam.james@gmail.com">Cameron James</a>
34   * @version 1.0
35   * @see org.apromore.dao.model.Node
36   */
37  @Repository
38  public interface NodeRepository extends JpaRepository<Node, Integer> {
39  
40      /**
41       * Find the Node by It's Uri.
42       * @param uri the uri to search for.
43       * @return the found Node.
44       */
45      @Query("SELECT n FROM Node n JOIN n.nodeMappings nm JOIN nm.fragmentVersion f WHERE n.uri = ?1 and f.id = ?2")
46      Node findNodeByUriAndFragmentVersion(String uri, Integer fragmentId);
47  
48      /**
49       * Find Nodes by using the Fragment URI.
50       * @param fragmentURI the fragment uri.
51       * @return the list of found nodes.
52       */
53      @Query("SELECT n FROM Node n JOIN n.nodeMappings nm JOIN nm.fragmentVersion f WHERE f.uri = ?1")
54      List<Node> getNodesByFragmentURI(String fragmentURI);
55  
56  }
