113     /**
114      * Creates a version identifier from the specified string.
115      * <p/>
116      * <p/>
117      * Version string grammar:
118      * <p/>
119      * <pre>
120      * version ::= major('.'minor('.'micro('.'qualifier)?)?)?
121      * major ::= digit+
122      * minor ::= digit+
123      * micro ::= digit+
124      * qualifier ::= (alpha|digit|'_'|'-')+
125      * digit ::= [0..9]
126      * alpha ::= [a..zA..Z]
127      * </pre>
128      *
129      * @param version String representation of the version identifier. There
130      *                must be no whitespace in the argument.
131      * @throws IllegalArgumentException If {@code version} is improperly
132      *                                  formatted.
133      */
134     public Version(String version) {
135         Integer maj;
136         Integer min = null;
137         Integer mic = null;
138         String qual = "";
139 
140         try {
141             StringTokenizer st = new StringTokenizer(version, SEPARATOR, true);
142             maj = parseInt(st.nextToken(), version);
143 
144             if (st.hasMoreTokens()) { // minor
145                 st.nextToken(); // consume delimiter
146                 min = parseInt(st.nextToken(), version);
147 
148                 if (st.hasMoreTokens()) { // micro
149                     st.nextToken(); // consume delimiter
150                     mic = parseInt(st.nextToken(), version);
151 
152                     if (st.hasMoreTokens()) { // qualifier separator
153                         st.nextToken(); // consume delimiter
154                         qual = st.nextToken(""); // remaining string
155 
156                         if (st.hasMoreTokens()) { // fail safe
157                             throw new IllegalArgumentException("invalid version \"" + version + "\": invalid format");
158                         }
159                     }
160                 }
161             }
162         } catch (NoSuchElementException e) {
163             IllegalArgumentException iae = new IllegalArgumentException("invalid version \"" + version + "\": invalid format");
164             iae.initCause(e);
165             throw iae;
166         }
167 
168         major = maj;
169         minor = min;
170         micro = mic;
171         qualifier = qual;
172         validate();
173     }

175     /**
176      * Parse numeric component into an int.
177      *
178      * @param value   Numeric component
179      * @param version Complete version string for exception message, if any
180      * @return int value of numeric component

182     private static int parseInt(String value, String version) {
183         try {
184             return Integer.parseInt(value);
185         } catch (NumberFormatException e) {
186             IllegalArgumentException iae = new IllegalArgumentException("invalid version \"" + version + "\": non-numeric \"" + value + "\"");
187             iae.initCause(e);
188             throw iae;
189         }
190     }

192     /**
193      * Called by the Version constructors to validate the version components.
194      *
195      * @throws IllegalArgumentException If the numerical components are negative
196      *                                  or the qualifier string is invalid.
197      */
198     private void validate() {
199         if (major < 0) {
200             throw new IllegalArgumentException("invalid version \"" + toString0() + "\": negative number \"" + major + "\"");
201         }
202         if (micro != null && minor < 0) {
203             throw new IllegalArgumentException("invalid version \"" + toString0() + "\": negative number \"" + minor + "\"");
204         }
205         if (micro != null && micro < 0) {
206             throw new IllegalArgumentException("invalid version \"" + toString0() + "\": negative number \"" + micro + "\"");
207         }
208         for (char ch : qualifier.toCharArray()) {
209             if (('A' <= ch) && (ch <= 'Z')) {
210                 continue;
211             }
212             if (('a' <= ch) && (ch <= 'z')) {
213                 continue;
214             }
215             if (('0' <= ch) && (ch <= '9')) {
216                 continue;
217             }
218             if ((ch == '_') || (ch == '-')) {
219                 continue;
220             }
221             throw new IllegalArgumentException("invalid version \"" + toString0() + "\": invalid qualifier \"" + qualifier + "\"");
222         }
223     }

225     /**
226      * Parses a version identifier from the specified string.
227      * <p/>
228      * <p/>
229      * See {@code Version(String)} for the format of the version string.
230      *
231      * @param version String representation of the version identifier. Leading
232      *                and trailing whitespace will be ignored.
233      * @return A {@code Version} object representing the version identifier. If
234      * {@code version} is {@code null} or the empty string then
235      * {@code emptyVersion} will be returned.
236      * @throws IllegalArgumentException If {@code version} is improperly
237      *                                  formatted.
238      */
239     public static Version parseVersion(String version) {
240         if (version == null) {
241             return emptyVersion;
242         }
243 
244         version = version.trim();
245         if (version.length() == 0) {
246             return emptyVersion;
247         }
248 
249         return new Version(version);
250     }

302     /**
303      * Internal toString behavior
304      *
305      * @return The string representation of this version identifier.
306      */
307     String toString0() {
308         if (versionString != null) {
309             return versionString;
310         }
311         int q = qualifier.length();
312         StringBuilder result = new StringBuilder(20 + q);
313         result.append(major);
314         if (minor != null) {
315             result.append(SEPARATOR);
316             result.append(minor);
317         }
318         if (micro != null) {
319             result.append(SEPARATOR);
320             result.append(micro);
321         }
322         if (q > 0) {
323             result.append(SEPARATOR);
324             result.append(qualifier);
325         }
326         return versionString = result.toString();
327     }

335         if (hash != 0) {
336             return hash;
337         }
338         int h = 31 * 17;
339         h = 31 * h + major;
340         h = 31 * h + minor;
341         h = 31 * h + micro;

375     /**
376      * Compares this {@code Version} object to another {@code Version}.
377      * <p/>
378      * <p/>
379      * A version is considered to be <b>less than</b> another version if its
380      * major component is less than the other version's major component, or the
381      * major components are equal and its minor component is less than the other
382      * version's minor component, or the major and minor components are equal
383      * and its micro component is less than the other version's micro component,
384      * or the major, minor and micro components are equal and it's qualifier
385      * component is less than the other version's qualifier component (using
386      * {@code String.compareTo}).
387      * <p/>
388      * <p/>
389      * A version is considered to be <b>equal to</b> another version if the
390      * major, minor and micro components are equal and the qualifier component
391      * is equal (using {@code String.compareTo}).
392      *
393      * @param other The {@code Version} object to be compared.
394      * @return A negative integer, zero, or a positive integer if this version
395      * is less than, equal to, or greater than the specified
396      * {@code Version} object.
397      * @throws ClassCastException If the specified object is not a
398      *                            {@code Version} object.
399      */
400     @Override
401     public int compareTo(Version other) {
402         if (other == this) { // quicktest
403             return 0;
404         }
405 
406         int result = major - other.major;
407         if (result != 0) {
408             return result;
409         }
410 
411 
412         if (minor != null) {
413             result = minor - other.minor;
414             if (result != 0) {
415                 return result;
416             }
417         }
418 
419         if (micro != null) {
420             result = micro - other.micro;
421             if (result != 0) {
422                 return result;
423             }
424         }
425 
426         return qualifier.compareTo(other.qualifier);
427     }
