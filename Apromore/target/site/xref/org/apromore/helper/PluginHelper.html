47  /**
48   * Helps converting the XML representation of Messages and Properties of {@link PluginResult} and {@link PluginRequest} forth and back.
49   *
50   * @author <a href="mailto:felix.mannhardt@smail.wir.h-brs.de">Felix Mannhardt (Bonn-Rhein-Sieg University oAS)</a>

53  public final class PluginHelper {

55      private static final Logger LOGGER = LoggerFactory.getLogger(PluginHelper.class);

57      private static final ObjectFactory OBJECT_FACTORY = new ObjectFactory();

59      private PluginHelper() {
60          super();

63      /**
64       * Converts the XML representation of Plugin properties to a Set of RequestPropertyType for usage in Plugins
65       *
66       * @param xmlProperties from web service
67       * @return Set of RequestPropertyType
68       */
69      public static Set<RequestParameterType<?>> convertToRequestParameters(final PluginParameters xmlProperties) {
70          Set<RequestParameterType<?>> properties = new HashSet<RequestParameterType<?>>();
71          if (xmlProperties != null) {
72              for (PluginParameter xmlProp : xmlProperties.getParameter()) {
73                  RequestParameterType<?> requestParam = convertToRequestParameter(xmlProp);
74                  if (requestParam != null) {
75                      properties.add(requestParam);
76                  }
77              }
78          }
79          return properties;
80      }
81  
82      /**
83       * Converts the XML representation of a Plug-in property to a RequestPropertyType for usage in Plug-ins
84       *
85       * @param xmlProp single property from web service
86       * @return null if conversion failed
87       */
88      public static RequestParameterType<?> convertToRequestParameter(final PluginParameter xmlProp) {
89          String clazz = xmlProp.getClazz();
90          if (clazz != null && xmlProp.getId() != null && xmlProp.getValue() != null) {
91              try {
92                  // Look if class is valid
93                  ParameterType.class.getClassLoader().loadClass(clazz);
94                  return createPluginProperty(xmlProp);
95              } catch (ClassNotFoundException e) {
96                  LOGGER.error("Could not convert request property", e);
97              }
98          }
99          return null;
100     }
101 
102     private static RequestParameterType<?> createPluginProperty(final PluginParameter xmlProp) {
103         Object value = xmlProp.getValue();
104         Object convertedValue = null;
105         if (value instanceof DataHandler) {
106             try {
107                 convertedValue = ((DataHandler) value).getInputStream();
108             } catch (IOException e) {
109                 LOGGER.error("Could not convert InputStream property", e);
110             }
111         } else {
112             convertedValue = value;
113         }
114         if (convertedValue != null) {
115             return new RequestParameterType<Object>(xmlProp.getId(), convertedValue);
116         } else {
117             return null;
118         }
119     }

121     /**
122      * Converts the PropertyType of a Plugin to XML representation for transport through web service
123      *
124      * @param pluginProperties from Plugin
125      * @return XML representation for use in web service
126      */
127     public static PluginParameters convertFromPluginParameters(final Set<? extends ParameterType<?>> pluginProperties) {
128         PluginParameters xmlProperties = OBJECT_FACTORY.createPluginParameters();
129         for (ParameterType<?> p : pluginProperties) {
130             PluginParameter xmlProp = OBJECT_FACTORY.createPluginParameter();
131             xmlProp.setId(p.getId());
132             xmlProp.setName(p.getName());
133             xmlProp.setDescription(p.getDescription());
134             xmlProp.setIsMandatory(p.isMandatory());
135             xmlProp.setClazz(p.getValueType().getCanonicalName());
136             if (p.getValue() instanceof InputStream) {
137                 // Property will be rendered as FileInput
138                 xmlProp.setValue(null);
139             } else {
140                 xmlProp.setValue(p.getValue());
141             }
142             xmlProp.setCategory(p.getCategory());
143             xmlProperties.getParameter().add(xmlProp);
144         }
145         return xmlProperties;
146     }

148     public static PluginMessages convertFromPluginMessages(final List<? extends PluginMessage> messages) {
149         PluginMessages xmlMessages = OBJECT_FACTORY.createPluginMessages();
150         for (PluginMessage msg : messages) {
151             org.apromore.model.PluginMessage xmlMessage = OBJECT_FACTORY.createPluginMessage();
152             xmlMessage.setValue(msg.getMessage());
153             xmlMessages.getMessage().add(xmlMessage);
154         }
155         return xmlMessages;
156     }
157 
158     public static List<PluginMessage> convertToPluginMessages(final PluginMessages message) {
159         List<PluginMessage> messageList = new ArrayList<PluginMessage>();
160         if (message != null) {
161             for (org.apromore.model.PluginMessage msg : message.getMessage()) {
162                 messageList.add(new PluginMessageImpl(msg.getValue()));
163             }
164         }
165         return messageList;
166     }
167 
168     public static PluginInfo convertPluginInfo(final Plugin plugin) {
169         PluginInfo pluginInfo = new PluginInfo();
170         pluginInfo.setName(plugin.getName());
171         pluginInfo.setVersion(plugin.getVersion());
172         pluginInfo.setDescription(plugin.getDescription());
173         pluginInfo.setType(plugin.getType());
174         pluginInfo.setAuthor(plugin.getAuthor());
175         pluginInfo.setEmail(plugin.getEMail());
176         return pluginInfo;
177     }
