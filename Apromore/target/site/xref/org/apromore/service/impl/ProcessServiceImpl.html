182     /**
183      * @see org.apromore.service.ProcessService#importProcess(String, Integer, String, Version, String, org.apromore.service.model.CanonisedProcess, String, String, String, String, boolean)
184      * {@inheritDoc}
185      */
186     @Override
187     @Transactional(readOnly = false)
188     public ProcessModelVersion importProcess(final String username, final Integer folderId, final String processName,
189             final Version version, final String natType, final CanonisedProcess cpf, final String domain,
190             final String documentation, final String created, final String lastUpdate, final boolean publicModel) throws ImportException {
191         LOGGER.debug("Executing operation canoniseProcess");
192 
193         if (cpf == null) {
194             LOGGER.error("Process " + processName + " Failed to import correctly.");
195             throw new ImportException("Process " + processName + " Failed to import correctly.");
196         } else if ((folderId.equals(0) && processRepo.findUniqueByName(processName) != null) ||
197                 (processRepo.findByNameAndFolderId(processName, folderId) != null)) {
198             LOGGER.error("Process " + processName + " was found to already exist in the Repository.");
199             throw new ImportException("Process " + processName + " was found to already exist in the Repository.");
200         }
201 
202         ProcessModelVersion pmv;
203         try {
204             User user = userSrv.findUserByLogin(username);
205             NativeType nativeType = formatSrv.findNativeType(natType);
206             Process process = insertProcess(processName, user, nativeType, domain, folderId, created, publicModel);
207 
208             pmv = addProcess(process, processName, version, Constants.TRUNK_NAME, created, lastUpdate, cpf, nativeType);
209             formatSrv.storeNative(processName, pmv, created, lastUpdate, user, nativeType, Constants.INITIAL_ANNOTATION, cpf);
210             workspaceSrv.addProcessToFolder(process.getId(), folderId);
211 
212         } catch (UserNotFoundException | JAXBException | IOException e) {
213             LOGGER.error("Failed to import process {} with native type {}", processName, natType);
214             LOGGER.error("Original exception was: ", e);
215             throw new ImportException(e);
216         }
217 
218         return pmv;
219     }

221     /**
222      * @see org.apromore.service.ProcessService#updateProcess(Integer, String, String, String, Version, Version, org.apromore.dao.model.User, String, org.apromore.dao.model.NativeType, org.apromore.service.model.CanonisedProcess)
223      * {@inheritDoc}
224      */
225     @Override
226     @Transactional(readOnly = false)
227     @Event(message = HistoryEnum.UPDATE_PROCESS_MODEL)
228     public ProcessModelVersion updateProcess(final Integer processId, final String processName, final String originalBranchName,
229             final String newBranchName, final Version versionNumber, final Version originalVersionNumber, final User user, final String lockStatus,
230             final NativeType nativeType, final CanonisedProcess cpf) throws ImportException, RepositoryException {
231         ProcessModelVersion pmv;
232         String now = new SimpleDateFormat(Constants.DATE_FORMAT).format(new Date());
233 
234         try {
235             if (!StringUtils.equals(originalBranchName, newBranchName)) {
236                 Process process = processRepo.findOne(processId);
237                 pmv = addProcess(process, processName, versionNumber, newBranchName, now, now, cpf, nativeType);
238             } else {
239                 // Perform the update
240                 if (user == null) {
241                     throw new ImportException("Permission to change this model denied.  No user specified.");
242                 } else if (canUserWriteProcess(user, processId)) {
243                     pmv = updateExistingProcess(processId, processName, originalBranchName, versionNumber, originalVersionNumber, lockStatus, cpf, nativeType);
244                 } else {
245                     throw new ImportException("Permission to change this model denied.  Try saving as a new branch instead.");
246                 }
247             }
248 
249             formatSrv.storeNative(processName, pmv, now, now, user, nativeType, versionNumber.toString(), cpf);
250         } catch (RepositoryException | JAXBException | IOException e) {
251             LOGGER.error("Failed to update process {}", processName);
252             LOGGER.error("Original exception was: ", e);
253             throw new RepositoryException("Failed to Update process model.", e);
254         }
255 
256         return pmv;
257     }

259     /**
260      * @param user  a user
261      * @param processId  identifier for a process
262      * @return whether the <var>user</var> should be allowed to update the process identified by <var>processId</var>
263      */
264     private boolean canUserWriteProcess(User user, Integer processId) {
265         for (GroupProcess gp: groupProcessRepo.findByProcessAndUser(processId, user.getRowGuid())) {
266             if (gp.getHasWrite()) {
267                  return true;
268             }
269         }
270         return false;
271     }

273     /**
274      * @see org.apromore.service.ProcessService#exportProcess(String, Integer, String, Version, String, String, boolean, java.util.Set)
275      * {@inheritDoc}
276      */
277     @Override
278     public ExportFormatResultType exportProcess(final String name, final Integer processId, final String branch, final Version version,
279             final String format, final String annName, final boolean withAnn, Set<RequestParameterType<?>> canoniserProperties)
280             throws ExportFormatException {
281         try {
282             // Debug tracing of the authenticated principal
283             org.springframework.security.core.Authentication auth =
284                 org.springframework.security.core.context.SecurityContextHolder.getContext().getAuthentication();
285             if (auth != null) {
286                 LOGGER.info("Authentication principal=" + auth.getPrincipal() + " details=" + auth.getDetails() + " thread=" + Thread.currentThread());
287             } else {
288                 LOGGER.info("Authentication is null");
289             }
290 
291             ExportFormatResultType exportResult = new ExportFormatResultType();
292 
293             // Work out if we are looking at the original format or native format for this model.
294             if (isRequestForNativeFormat(processId, branch, version, format)) {
295                 Native native_ = nativeRepo.getNative(processId, branch, version.toString(), format);
296                 String content_ = native_.getContent();
297                 ByteArrayDataSource byteArrayDataSource = new ByteArrayDataSource(content_, "text/xml");
298                 DataHandler dataHandler = new DataHandler(byteArrayDataSource);
299                 exportResult.setNative(dataHandler);
300             } else if (isRequestForAnnotationsOnly(format)) {
301                 exportResult.setNative(new DataHandler(new ByteArrayDataSource(annotationRepo.getAnnotation(processId, branch,
302                         version.toString(), AnnotationHelper.getAnnotationName(annName)).getContent(), "text/xml")));
303             } else {
304                 CanonicalProcessType cpt = getProcessModelVersion(processId, name, branch, version, false);
305                 Process process;
306                 if (format.equals(Constants.CANONICAL)) {
307                     exportResult.setNative(new DataHandler(new ByteArrayDataSource(canoniserSrv.CPFtoString(cpt), Constants.XML_MIMETYPE)));
308                 } else {
309                     DecanonisedProcess dp;
310                     AnnotationsType anf = null;
311                     process = processRepo.findOne(processId);
312                     if (withAnn) {
313                         Annotation ann = annotationRepo.getAnnotation(processId, branch, version.toString(), annName);
314                         if (ann != null) {
315                             String annotation = ann.getContent();
316                             if (annotation != null && !annotation.equals("")) {
317                                 ByteArrayDataSource dataSource = new ByteArrayDataSource(annotation, Constants.XML_MIMETYPE);
318                                 anf = ANFSchema.unmarshalAnnotationFormat(dataSource.getInputStream(), false).getValue();
319                             }
320                         }
321 
322                         if (ann != null && !process.getNativeType().getNatType().equalsIgnoreCase(ann.getNatve().getNativeType().getNatType())) {
323                             anf = annotationSrv.preProcess(ann.getNatve().getNativeType().getNatType(), format, cpt, anf);
324                         } else {
325                             anf = annotationSrv.preProcess(process.getNativeType().getNatType(), format, cpt, anf);
326                         }
327                     } else if (annName == null) {
328                         anf = annotationSrv.preProcess(null, format, cpt, null);
329                     }
330 
331                     dp = canoniserSrv.deCanonise(format, cpt, anf, canoniserProperties);
332 
333                     exportResult.setMessage(PluginHelper.convertFromPluginMessages(dp.getMessages()));
334                     exportResult.setNative(new DataHandler(new ByteArrayDataSource(dp.getNativeFormat(), Constants.XML_MIMETYPE)));
335                 }
336             }
337 
338             return exportResult;
339         } catch (Exception e) {
340             LOGGER.error("Failed to export process model {} to format {}", name, format);
341             LOGGER.error("Original exception was: ", e);
342             throw new ExportFormatException(e);
343         }
344     }

348      * @see org.apromore.service.ProcessService#updateProcessMetaData(Integer, String, String, String, Version, Version, String, boolean)
349      * {@inheritDoc}
350      */
351     @Override
352     @Transactional(readOnly = false)
353     public void updateProcessMetaData(final Integer processId, final String processName, final String domain, final String username,
354             final Version preVersion, final Version newVersion, final String ranking, final boolean isPublic) throws UpdateProcessException {
355         LOGGER.debug("Executing operation update process meta data.");
356 
357         try {
358             boolean updatePublicSecurity = false;
359 
360             ProcessModelVersion processModelVersion = processModelVersionRepo.getCurrentProcessModelVersion(processId, preVersion.toString());
361             ProcessBranch branch = processModelVersion.getProcessBranch();
362             Process process = processRepo.findOne(processId);
363 
364             if (process.getPublicModel() != isPublic) {
365                 updatePublicSecurity = true;
366             }
367 
368             process.setDomain(domain);
369             process.setName(processName);
370             process.setRanking(ranking);
371             process.setUser(userSrv.findUserByLogin(username));
372             process.setPublicModel(isPublic);
373             processModelVersion.setVersionNumber(newVersion.toString());
374 
375             updateNative(processModelVersion.getNativeDocument(), processName, username, newVersion);
376             if (updatePublicSecurity) {
377                 if (isPublic) {
378                     workspaceSrv.createPublicStatusForUsers(process);
379                 } else {
380                     workspaceSrv.removePublicStatusForUsers(process);
381                 }
382             }
383 
384             processRepo.save(process);
385             processModelVersionRepo.save(processModelVersion);
386             processBranchRepo.save(branch);
387         } catch (Exception e) {
388             throw new UpdateProcessException(e.getMessage(), e.getCause());
389         }
390     }

430     /**
431      * @see ProcessService#getCanonicalFormat(ProcessModelVersion, String, String, boolean)
432      * {@inheritDoc}
433      */
434     @Override
435     @Transactional(readOnly = false)
436     public CanonicalProcessType getCanonicalFormat(final ProcessModelVersion pmv, final String processName, final String branchName, final boolean lock) {
437         Canonical canonical;
438         CanonicalProcessType tmp = new CanonicalProcessType();
439         try {
440             canonical = composerSrv.compose(pmv.getRootFragmentVersion());
441             canonical.setProperty(Constants.PROCESS_NAME, processName);
442             canonical.setProperty(Constants.BRANCH_NAME, branchName);
443             canonical.setProperty(Constants.BRANCH_ID, pmv.getProcessBranch().getId().toString());
444             canonical.setProperty(Constants.VERSION_NUMBER, pmv.getVersionNumber());
445             canonical.setProperty(Constants.PROCESS_MODEL_VERSION_ID, pmv.getId().toString());
446             canonical.setProperty(Constants.ROOT_FRAGMENT_ID, pmv.getRootFragmentVersion().getId().toString());
447             if (pmv.getProcessBranch().getProcess().getNativeType() != null) {
448                 canonical.setProperty(Constants.INITIAL_FORMAT, pmv.getProcessBranch().getProcess().getNativeType().getNatType());
449             } else {
450                 canonical.setProperty(Constants.INITIAL_FORMAT, null);
451             }
452             if (lock) {
453                 canonical.setProperty(Constants.LOCK_STATUS, Constants.LOCKED);
454             }
455             Map<String, Variants.Variant> variantMap = new HashMap<>();
456             for (ProcessModelAttribute attribute: pmv.getProcessModelAttributes()) {
457                 Element extensionElements = XMLUtils.stringToAnyElement(attribute.getAny());
458                 if (extensionElements != null) {
459                     java.lang.Object o = JAXBContext.newInstance(org.apromore.cpf.ObjectFactory.class,
460                                                                  com.processconfiguration.ObjectFactory.class)
461                                                     .createUnmarshaller()
462                                                     .unmarshal(extensionElements);
463                     canonical.setProperty(attribute.getName(), attribute.getValue(), o);
464 
465                     // Populate a map from variant IDs to the actual JAXB Variant objects
466                     if (o instanceof Variants) {
467                         Variants variants = (Variants) o;
468                         for (Variants.Variant variant: variants.getVariant()) {
469                             variantMap.put(variant.getId(), variant);
470                         }
471                     } else {
472                         LOGGER.warn("Ignoring extension element " + o);
473                     }
474                 } else {
475                     canonical.setProperty(attribute.getName(), attribute.getValue(), extensionElements);
476                 }
477             }
478 
479             // Kludge to work around limitations in JAXB, populating configurationAnnotation/configuration@variantRef ID references
480             for (Map.Entry entry: canonical.variantMap.entrySet()) {
481                 ((ConfigurationAnnotation.Configuration) entry.getKey()).setVariantRef(variantMap.get(entry.getValue()));
482             }
483 
484             tmp = converter.convert(canonical);
485         } catch (ExceptionDao | JAXBException e) {
486             String msg = "Failed to retrieve the current version of the process model " + processName + " - " + branchName;
487             LOGGER.error(msg, e);
488         }
489         return tmp;
490     }

540     /**
541      * Creates new versions for all ascendant fragments of originalFragment by
542      * replacing originalFragment with updatedFragment. New versions will be
543      * created for all process models which use any of the updated fragments as
544      * its root fragment. This method also releases locks of all ascendant
545      * fragments.
546      * @param originalFragment the original fragment id
547      * @param updatedFragment the updated fragment id
548      * @param composingFragments the composing fragments
549      */
550     @Override
551     @Transactional(readOnly = false)
552     public void propagateChangesWithLockRelease(final FragmentVersion originalFragment, final FragmentVersion updatedFragment,
553             final Set<FragmentVersion> composingFragments, final Version newVersionNumber) throws RepositoryException {
554         // create new versions for all process models, which use this fragment as the root fragment, and unlock those process models.
555         List<ProcessModelVersion> usedProcessModels = processModelVersionRepo.getUsedProcessModelVersions(originalFragment);
556         for (ProcessModelVersion pmv : usedProcessModels) {
557             createNewProcessModelVersion(pmv, updatedFragment, composingFragments, newVersionNumber);
558         }
559 
560         if (originalFragment != null) {
561             // unlock the fragment
562             LOGGER.debug("Unlocking the original fragment: " + originalFragment);
563             lService.unlockFragment(originalFragment);
564 
565             // release locks of all descendant fragments of the original fragment
566             lService.unlockDescendantFragments(originalFragment);
567 
568             // create new version for all ascendant fragments
569             LOGGER.debug("Propagating to parent fragments of fragment: " + originalFragment);
570             List<FragmentVersion> lockedParents = fragmentVersionRepo.getLockedParentFragments(originalFragment);
571 
572             for (FragmentVersion parent : lockedParents) {
573                 propagateToParentsWithLockRelease(parent, originalFragment, updatedFragment, composingFragments, newVersionNumber);
574             }
575         }
576     }
