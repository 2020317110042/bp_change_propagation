85      /**
86       * @see org.apromore.service.ContentService#addNode(org.apromore.graph.canonical.INode, String, java.util.Set, java.util.Set)
87       * {@inheritDoc}
88       */
89      @Override
90      public Node addNode(final INode cpfNode, final String graphType, Set<org.apromore.dao.model.Object> objects, Set<Resource> resources) {
91          LOGGER.trace("Adding new Node: " + cpfNode.getId());
92  
93          try {
94              Node node = new Node();
95              node.setName(cpfNode.getName());
96              node.setUri(cpfNode.getId());
97              node.setOriginalId(cpfNode.getOriginalId());
98              node.setNetId(cpfNode.getNetId());
99              node.setGraphType(graphType);
100             node.setNodeType(cpfNode.getNodeType());
101             node.setAllocation(cpfNode.getAllocation());
102             node.setConfiguration(cpfNode.isConfigurable());
103             node.setTeamWork(cpfNode.isTeamWork());
104             if (cpfNode.getTimeDate() != null) {
105                 node.setTimeDate(cpfNode.getTimeDate().getTime());
106             }
107             if (cpfNode.getTimeDuration() != null) {
108                 node.setTimeDuration(cpfNode.getTimeDuration());
109             }
110             if (cpfNode.getDirection() != null) {
111                 node.setMessageDirection(DirectionEnum.valueOf(cpfNode.getDirection().toString()));
112             }
113 
114             addNodeExpressions(cpfNode, node);
115             addObjectReferences(node, cpfNode, objects);
116             addResourceReferences(node, cpfNode, resources);
117             addNodeAttributes(node, cpfNode);
118 
119             return nRepository.save(node);
120         } catch (Exception e) {
121             LOGGER.error("Unable to add Node(" + cpfNode.getId() + "): " + e.getMessage(), e);
122         }
123         return null;
124     }

126     /**
127      * @see org.apromore.service.ContentService#addEdge(org.apromore.graph.canonical.CPFEdge, org.apromore.dao.model.FragmentVersion, org.apromore.service.helper.OperationContext)
128      * {@inheritDoc}
129      */
130     @Override
131     public Edge addEdge(final CPFEdge cpfEdge, FragmentVersion fv, OperationContext op) {
132         LOGGER.trace("Adding new Edge " + cpfEdge.getOriginalId() + " id=" + cpfEdge.getId() + " attributes=" + cpfEdge.getAttributes());
133 
134         try {
135             Edge edge = new Edge();
136             edge.setUri(cpfEdge.getId());
137             edge.setOriginalId(cpfEdge.getOriginalId());
138 
139             edge.setSourceNode(nRepository.findNodeByUriAndFragmentVersion(cpfEdge.getSource().getId(), fv.getId()));
140             if (edge.getSourceNode() == null) {
141                 edge.setSourceNode(findNode(op.getPersistedNodes().values(), cpfEdge.getSource().getId()));
142             }
143 
144             edge.setTargetNode(nRepository.findNodeByUriAndFragmentVersion(cpfEdge.getTarget().getId(), fv.getId()));
145             if (edge.getTargetNode() == null) {
146                 edge.setTargetNode(findNode(op.getPersistedNodes().values(), cpfEdge.getTarget().getId()));
147             }
148 
149             if (cpfEdge.getConditionExpr() != null) {
150                 Expression expr = new Expression();
151                 expr.setDescription(cpfEdge.getConditionExpr().getDescription());
152                 expr.setExpression(cpfEdge.getConditionExpr().getExpression());
153                 expr.setLanguage(cpfEdge.getConditionExpr().getLanguage());
154                 expr.setReturnType(cpfEdge.getConditionExpr().getReturnType());
155                 edge.setConditionExpression(exRepository.save(expr));
156             }
157 
158             addEdgeAttributes(edge, cpfEdge);
159 
160             if (edge.getSourceNode() == null || edge.getTargetNode() == null) {
161                 LOGGER.error("Either the Source or Target nodes are null. Please check!");
162             }
163 
164             return eRepository.save(edge);
165         } catch (Exception e) {
166             LOGGER.error("Unable to add Edge(" + cpfEdge.getOriginalId() + "): " + e.getMessage());
167         }
168         return null;
169     }

172     private void addNodeAttributes(final Node node, final INode v) {
173         NodeAttribute nAtt;
174         for (Entry<String, IAttribute> e : v.getAttributes().entrySet()) {
175             nAtt = new NodeAttribute();
176             nAtt.setName(e.getKey());
177             nAtt.setValue(e.getValue().getValue());
178             java.lang.Object any = e.getValue().getAny();
179             if (any instanceof Element) {
180                 nAtt.setAny(XMLUtils.anyElementToString((Element) any));
181             }
182             else if (any instanceof ConfigurationAnnotation) {
183                 nAtt.setAny(XMLUtils.extensionElementToString(any));
184             }
185             else if (any != null) {
186                 throw new IllegalArgumentException("Parsed an unsupported extension: " + any);
187             }
188             nAtt.setNode(node);
189             node.getAttributes().add(nAtt);
190         }
191     }

194     /* Add the Object References to the Node. */
195     private void addObjectReferences(Node node, INode cpfNode, Set<Object> objects) {
196         ObjectRef objectRef;
197         if (cpfNode.getObjectReferences() != null) {
198             for (ICPFObjectReference cpfObjRef : cpfNode.getObjectReferences()) {
199                 if (cpfObjRef.getObjectId() != null) {
200                     objectRef = new ObjectRef();
201 
202                     objectRef.setNode(node);
203                     objectRef.setConsumed(cpfObjRef.isConsumed());
204                     objectRef.setOptional(cpfObjRef.isOptional());
205                     objectRef.setType(cpfObjRef.getObjectRefType());
206                     objectRef.setObject(findObject(objects, cpfObjRef.getObjectId()));
207                     if (objectRef.getObject() != null) {
208                         node.getObjectRefs().add(objectRef);
209                         addObjectAttributes(objectRef, cpfObjRef);
210                     } else {
211                         LOGGER.warn("Object Reference for Node(" + cpfNode.getId() + ") not created, Could not find Object with id: " +
212                                 cpfObjRef.getObjectId());
213                     }
214                 } else {
215                     LOGGER.warn("Object Reference for Node(" + cpfNode.getId() + ") found with a NULL Object ID. Not Adding Reference");
216                 }
217             }
218         }
219     }

255                 } else {

389     private org.apromore.dao.model.Object findObject(Collection<Object> objects, String objectId) {
390         org.apromore.dao.model.Object found = null;
391         for (Object obj : objects) {
392             if (obj.getUri().equals(objectId)) {
393                 found = obj;
394                 break;
395             }
396         }
397         if (found == null) {
398             LOGGER.warn("Could not find Object with Id: " + objectId);
399         }
400         return found;
