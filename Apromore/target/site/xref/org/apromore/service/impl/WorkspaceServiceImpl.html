120     @Override
121     @Transactional(readOnly = false)
122     public void createFolder(String userId, String folderName, Integer parentFolderId) {
123         Folder folder = new Folder();
124         folder.setName(folderName);
125         User user = userRepo.findByRowGuid(userId);
126 
127         if (parentFolderId != 0) {
128             Folder parent = folderRepo.findOne(parentFolderId);
129             if (parent != null) {
130                 folder.setParentFolder(parent);
131             }
132         }
133 
134         Workspace workspace = workspaceRepo.findOne(1);
135         folder.setWorkspace(workspace);
136         folder.setCreatedBy(user);
137         folder.setModifiedBy(user);
138         folder.setDateCreated(Calendar.getInstance().getTime());
139         folder.setDateModified(Calendar.getInstance().getTime());
140         folder.setDescription("");
141         folder = folderRepo.save(folder);
142 
143         GroupFolder gf = new GroupFolder();
144         gf.setFolder(folder);
145         gf.setGroup(user.getGroup());
146         gf.setHasOwnership(true);
147         gf.setHasWrite(true);
148         gf.setHasRead(true);
149 
150         groupFolderRepo.save(gf);
151     }

173     @Override
174     public List<Folder> getBreadcrumbs(Integer folderId) {
175         List<Folder> folders = new ArrayList<>();
176 
177         Folder folder = folderRepo.findOne(folderId);
178         if (folder != null) {
179             folders.add(folder);
180             while (folder.getParentFolder() != null && folder.getParentFolder().getId() != 0) {
181                 folder = folderRepo.findOne(folder.getParentFolder().getId());
182                 folders.add(folder);
183             }
184         }
185 
186         return folders;
187     }

189     @Override
190     public List<GroupFolder> getSubFolders(String userRowGuid, Integer folderId) {
191         User user = userRepo.findByRowGuid(userRowGuid);
192         List<GroupFolder> folderUsers = new ArrayList<>();
193         Map<Integer, GroupFolder> map = new HashMap<>();
194         for (GroupFolder gf: groupFolderRepo.findByParentFolderAndUser(folderId, userRowGuid)) {
195             GroupFolder fu = map.get(gf.getFolder().getId());
196             if (fu == null) {
197                 fu = new GroupFolder();
198                 fu.setGroup(gf.getGroup());
199                 fu.setFolder(gf.getFolder());
200                 fu.setHasRead(gf.isHasRead());
201                 fu.setHasWrite(gf.isHasWrite());
202                 fu.setHasOwnership(gf.isHasOwnership());
203                 folderUsers.add(fu);
204                 map.put(gf.getFolder().getId(), fu);
205             } else {
206                 fu.setHasRead(fu.isHasRead()           || gf.isHasRead());
207                 fu.setHasWrite(fu.isHasWrite()         || gf.isHasWrite());
208                 fu.setHasOwnership(fu.isHasOwnership() || gf.isHasOwnership());
209             }
210         }
211         return folderUsers;
212     }

214     @Override
215     @Transactional(readOnly = false)
216     public String saveFolderPermissions(Integer folderId, String groupRowGuid, boolean hasRead, boolean hasWrite, boolean hasOwnership) {
217         Folder folder = folderRepo.findOne(folderId);
218         Group group = groupRepo.findByRowGuid(groupRowGuid);
219 
220         createGroupFolder(group, folder, hasRead, hasWrite, hasOwnership);
221 
222         Folder parentFolder = folder.getParentFolder();
223         while (parentFolder != null && parentFolder.getId() > 0) {
224             parentFolder = folderRepo.findOne(parentFolder.getId());
225             createGroupFolder(group, parentFolder, true, false, false);
226             parentFolder = parentFolder.getParentFolder();
227         }
228         saveSubFolderPermissions(folder, group, hasRead, hasWrite, hasOwnership);
229 
230         return "";
231     }

298      * {@inheritDoc}
299      */
300     @Override
301     @Transactional(readOnly = false)
302     public void removePublicStatusForUsers(final Process process) {
303         Group publicGroup = groupRepo.findPublicGroup();
304         if (publicGroup == null) {
305             LOGGER.warning("No public group in repository");
306         } else {
307             Set<GroupProcess> freshGroupProcesses = new HashSet<>();
308             for (GroupProcess groupProcess: process.getGroupProcesses()) {
309                 if (!publicGroup.equals(groupProcess.getGroup())) {
310                     freshGroupProcesses.add(groupProcess);
311                 }
312             }
313             process.setGroupProcesses(freshGroupProcesses);
314         }
315     }

334     /* Save the Sub Folder Permissions. */
335     private void saveSubFolderPermissions(Folder folder, Group group, boolean hasRead, boolean hasWrite, boolean hasOwnership) {
336         for (Folder subFolder : folder.getSubFolders()) {
337             createGroupFolder(group, subFolder, hasRead, hasWrite, hasOwnership);
338             saveSubFolderPermissions(subFolder, group, hasRead, hasWrite, hasOwnership);
339         }
340         for (Process process : folder.getProcesses()) {
341             createGroupProcess(group, process, hasRead, hasWrite, hasOwnership);
342         }
343     }

345     /* Delete the sub Folder permissions. */
346     private void removeSubFolderPermissions(Folder folder, Group group) {
347         for (Folder subFolder : folder.getSubFolders()) {
348             removeGroupFolder(group, subFolder);
349             removeSubFolderPermissions(subFolder, group);
350         }
351         for (Process process : folder.getProcesses()) {
352             removeGroupProcess(group, process);
353         }
354     }

360     private void createGroupFolder(Group group, Folder folder, boolean hasRead, boolean hasWrite, boolean hasOwnership) {
361         GroupFolder groupFolder = groupFolderRepo.findByGroupAndFolder(group, folder);
362         if (groupFolder == null) {
363             groupFolder = new GroupFolder();
364             groupFolder.setGroup(group);
365             groupFolder.setFolder(folder);
366         }
367         assert groupFolder != null;
368         groupFolder.setHasRead(hasRead);
369         groupFolder.setHasWrite(hasWrite);
370         groupFolder.setHasOwnership(hasOwnership);
371 
372         groupFolderRepo.save(groupFolder);
373     }
