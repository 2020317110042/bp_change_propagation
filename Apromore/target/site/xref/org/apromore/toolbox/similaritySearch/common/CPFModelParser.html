18      public static Graph readModel(CanonicalProcessType cpf) {
19          Graph epcGraph = new Graph();
20          epcGraph.name = cpf.getName();
21  
22          NetType mainNet = null;
23          for (NetType n : cpf.getNet()) {
24              if (n.getNode().size() > 0) {
25                  mainNet = n;
26                  break;
27              }
28          }
29  
30          if (mainNet == null) {
31              return epcGraph;
32          }
33  
34          addNodes(mainNet, epcGraph);
35          addEdges(mainNet, epcGraph);
36          epcGraph.linkVertices();
37  
38          // read references and objects
39          addObjectsAndResources(cpf, epcGraph);
40  
41          return epcGraph;
42      }

111     private static void addObjectsAndResources(CanonicalProcessType cpf, Graph epcGraph) {

113             for (ObjectType o : net.getObject()) {
114                 epcGraph.addObject(
115                         new VertexObject(o.getId(), o.getName(), o.isConfigurable(),
116                                 o instanceof HardType ? VertexObject.SoftHart.Hard : (o instanceof SoftType ? VertexObject.SoftHart.Soft : VertexObject.SoftHart.Other))

118             }

122             epcGraph.addResource(new VertexResource(r.getId(), r.getName(), r.isConfigurable(),
123                     r instanceof HumanType ? VertexResource.Type.Human : (r instanceof NonhumanType ? VertexResource.Type.NonHuman : VertexResource.Type.Other)));
124         }

127     private static void addNodes(NetType mainNet, Graph epcGraph) {
128         for (NodeType n : mainNet.getNode()) {
129             HashMap<String, String> annotationMap = parseAnnotationForEventsAndFunctions(getFromAnnotations("annotation", n.getAttribute()));
130             // gateways
131             if (n instanceof RoutingType && !(n instanceof StateType)) {
132                 boolean initialGW = true;
133                 String initial = getFromAnnotations("added", n.getAttribute());
134                 if (initial != null && "true".equals(initial)) {
135                     initialGW = false;
136                 }
137                 Vertex v = new Vertex(
138                         (n instanceof XORJoinType || n instanceof XORSplitType) ? "xor" :
139                                 ((n instanceof ANDJoinType || n instanceof ANDSplitType) ? "and" :
140                                         "or"),
141                         n.getId());
142 
143                 // add annotations
144                 if (annotationMap != null) {
145                     v.addAnnotations(annotationMap);
146                 }
147                 // this is initial gateway
148                 // TODO if we don't have graph that is already merged, then
149                 // set all gateways to initials
150                 if (initialGW) {
151                     v.setInitialGW();
152                 }
153                 if (n.isConfigurable() != null && n.isConfigurable()) {
154                     v.setConfigurable(true);
155                 }
156 
157                 epcGraph.addVertex(v);
158             } else if (n instanceof TaskType) {
159                 Vertex v = new Vertex(Type.function, n.getName(), n.getId());/*pass id of nodes of net to nodes in graph*/
160                 // add annotations
161                 if (annotationMap != null) {
162                     v.addAnnotations(annotationMap);
163                 }
164 
165                 addResourcesAndObjects((WorkType) n, v);
166 
167                 epcGraph.addVertex(v);
168             } else if (n instanceof EventType) {
169                 Vertex v = new Vertex(Type.event, n.getName(), n.getId());
170                 // add annotations
171                 if (annotationMap != null) {
172                     v.addAnnotations(annotationMap);
173                 }
174 
175                 addResourcesAndObjects((WorkType) n, v);
176 
177                 epcGraph.addVertex(v);
178             } else if (n instanceof StateType) {
179                 Vertex v = new Vertex(Type.state, n.getName(), n.getId());
180                 // add annotations
181                 if (annotationMap != null) {
182                     v.addAnnotations(annotationMap);
183                 }
184                 epcGraph.addVertex(v);
185             }
186         }

252     private static void addResourcesAndObjects(WorkType n, Vertex v) {
253         // add resources
254         for (ResourceTypeRefType r : n.getResourceTypeRef()) {
255             v.resourceRefs.add(new VertexResourceRef(
256                     r.getResourceTypeId(),
257                     r.getQualifier(),
258                     parseModelsFromAnnotations(getFromAnnotations("annotation", r.getAttribute()))));
259         }
260 
261         // add objects
262         for (ObjectRefType o : n.getObjectRef()) {
263             v.objectRefs.add(new VertexObjectRef(
264                     o.isOptional(),
265                     o.getObjectId(),
266                     o.isConsumed(),
267                     (o.getType().equals(InputOutputType.INPUT) ?
268                             VertexObjectRef.InputOutput.Input :
269                             VertexObjectRef.InputOutput.Output),
270                     parseModelsFromAnnotations(getFromAnnotations("annotation", o.getAttribute()))));
271         }

283     // the annotation attribute for events and functions are in format
284     // model1:name in model1;model2:name in modelass 2;
285     // the names must not contain ';', otherwise the name parsing fails
286     private static HashMap<String, String> parseAnnotationForEventsAndFunctions(String nodeValue) {
287         HashMap<String, String> annotationMap = new HashMap<String, String>();
288         if (nodeValue == null) {
289             return annotationMap;
290         }
291 
292         StringTokenizer st = new StringTokenizer(nodeValue, ";");
293         while (st.hasMoreTokens()) {
294             String nextToken = st.nextToken();
295             int colon = nextToken.indexOf(':');
296             String modelName = nextToken.substring(0, colon);
297             String value = nextToken.substring(colon + 1, nextToken.length());
298             annotationMap.put(modelName, value);
299         }
300 
301         return annotationMap;
302     }

304     private static void addEdges(NetType mainNet, Graph epcGraph) {
305 
306         List<EdgeType> edges = mainNet.getEdge();
307         HashSet<String> graphLabels = new HashSet<String>();
308 
309         // add elements
310         for (EdgeType e : edges) {
311             Edge toAdd = new Edge(e.getSourceId(), e.getTargetId(), e.getId());
312             String annotations = getFromAnnotations("name", e.getAttribute());
313 
314             graphLabels = parseModelsFromAnnotations(annotations);
315             toAdd.addLabels(graphLabels);
316             epcGraph.addEdge(toAdd);
317         }
318 
319         if (graphLabels.size() > 0) {
320             epcGraph.setGraphConfigurable();
321             epcGraph.addGraphLabels(graphLabels);
322         }
323     }

348     private static HashSet<String> parseModelsFromAnnotations(String annotations) {
349         HashSet<String> graphLabels = new HashSet<String>();
350         if (annotations != null && annotations.length() > 0) {
351             StringTokenizer st = new StringTokenizer(annotations, ";");
352             while (st.hasMoreTokens()) {
353                 String l = st.nextToken();
354                 graphLabels.add(l);
355             }
356         }
357         return graphLabels;
358     }

360     public static CanonicalProcessType writeModel(Graph g, IdGeneratorHelper idGenerator) {
361         CanonicalProcessType toReturn = new CanonicalProcessType();
362 
363         // objects and resources
364         for (VertexObject o : g.getObjects().values()) {
365             ObjectType ot = new ObjectType();
366             if (o.getSofthard().equals(VertexObject.SoftHart.Soft)) {
367                 ot = new SoftType();
368             } else if (o.getSofthard().equals(VertexObject.SoftHart.Hard)) {
369                 ot = new HardType();
370             }
371 
372             ot.setConfigurable(o.isConfigurable());
373             ot.setId(o.getId());
374             ot.setName(o.getName());
375             TypeAttribute a = new TypeAttribute();
376             a.setName("annotation");
377             a.setValue(parseAnnotationFromSet(o.getModels()));
378             ot.getAttribute().add(a);
379             toReturn.getNet().get(0).getObject().add(ot);
380         }
381 
382         // objects and resources
383         for (VertexResource r : g.getResources().values()) {
384             ResourceTypeType rt = new ResourceTypeType();
385             if (r.getType().equals(VertexResource.Type.Human)) {
386                 rt = new HumanType();
387             } else if (r.getType().equals(VertexResource.Type.NonHuman)) {
388                 rt = new NonhumanType();
389             }
390 
391             rt.setConfigurable(r.isConfigurable());
392             rt.setId(r.getId());
393             rt.setName(r.getName());
394             TypeAttribute a = new TypeAttribute();
395             a.setName("annotation");
396             a.setValue(parseAnnotationFromSet(r.getModels()));
397             rt.getAttribute().add(a);
398             toReturn.getResourceType().add(rt);
399         }
400 
401         NetType net = new NetType();
402         net.setId(idGenerator.getNextId());
403         toReturn.getNet().add(net);
404 
405         for (Vertex v : g.getVertices()) {
406             NodeType n = new NodeType();
407             if (v.getType().equals(Vertex.Type.state)) {
408                 n = new StateType();
409             } else if (v.getType().equals(Vertex.Type.event)) {
410                 n = new EventType();
411             } else if (v.getType().equals(Vertex.Type.function)) {
412                 n = new TaskType();
413             } else if (v.getType().equals(Vertex.Type.gateway)) {
414                 if (v.getParents().size() > 1) {
415                     if (v.getGWType().equals(GWType.and)) {
416                         n = new ANDJoinType();
417                     } else if (v.getGWType().equals(GWType.or)) {
418                         n = new ORJoinType();
419                     } else if (v.getGWType().equals(GWType.xor)) {
420                         n = new XORJoinType();
421                     }
422                 } else if (v.getChildren().size() > 1) {
423                     if (v.getGWType().equals(GWType.and)) {
424                         n = new ANDSplitType();
425                     } else if (v.getGWType().equals(GWType.or)) {
426                         n = new ORSplitType();
427                     } else if (v.getGWType().equals(GWType.xor)) {
428                         n = new XORSplitType();
429                     }
430                 }
431                 if (v.isAddedGW()) {
432                     TypeAttribute a = new TypeAttribute();
433                     a.setName("added");
434                     a.setValue("true");
435                     n.getAttribute().add(a);
436 
437                     /*added by Zaiwen on 9 June 2016*/
438                     if (v.getGWType().equals(GWType.and)) {
439                         n = new ANDSplitType();
440                     } else if (v.getGWType().equals(GWType.or)) {
441                         n = new ORSplitType();
442                     } else if (v.getGWType().equals(GWType.xor)) {
443                         n = new XORSplitType();
444                     }
445                 }
446             }
447 
448             n.setConfigurable(v.isConfigurable());
449             n.setId(v.getID());
450             n.setName(v.getLabel());
451             TypeAttribute a = new TypeAttribute();
452             a.setName("annotation");
453             a.setValue(parseAnnotationFromMap(v.getAnnotationMap()));
454             n.getAttribute().add(a);
455 
456             if (v.getType().equals(Vertex.Type.event) || v.getType().equals(Vertex.Type.function)) {
457                 // object ref
458                 for (VertexObjectRef o : v.objectRefs) {
459                     ObjectRefType oRef = new ObjectRefType();
460                     if (o.getInputOutput().equals(VertexObjectRef.InputOutput.Input)) {
461                         oRef.setType(InputOutputType.INPUT);
462                     } else if (o.getInputOutput().equals(VertexObjectRef.InputOutput.Output)) {
463                         oRef.setType(InputOutputType.OUTPUT);
464                     }
465                     oRef.setConsumed(o.getConsumed());
466                     oRef.setObjectId(o.getObjectID());
467                     oRef.setOptional(o.isOptional());
468                     // attributes
469                     TypeAttribute a1 = new TypeAttribute();
470                     a1.setName("annotation");
471                     a1.setValue(parseAnnotationFromSet(o.getModels()));
472                     oRef.getAttribute().add(a1);
473 
474                     ((WorkType) n).getObjectRef().add(oRef);
475                 }
476                 // resource ref
477                 for (VertexResourceRef r : v.resourceRefs) {
478                     ResourceTypeRefType rRef = new ResourceTypeRefType();
479                     rRef.setResourceTypeId(r.getResourceID());
480                     rRef.setQualifier(r.getQualifier());
481                     // attrubutes
482                     TypeAttribute a1 = new TypeAttribute();
483                     a1.setName("annotation");
484                     a1.setValue(parseAnnotationFromSet(r.getModels()));
485                     rRef.getAttribute().add(a1);
486 
487                     ((WorkType) n).getResourceTypeRef().add(rRef);
488                 }
489             }
490             net.getNode().add(n);
491         }
492 
493         for (Edge e : g.getEdges()) {
494             EdgeType et = new EdgeType();
495             et.setId(e.getId());
496             // from and to vertex
497             et.setSourceId(e.getFromVertex());
498             et.setTargetId(e.getToVertex());
499             // attributes
500             TypeAttribute a = new TypeAttribute();
501             a.setName("name");
502             a.setValue(parseAnnotationFromSet(e.getLabels()));
503             et.getAttribute().add(a);
504             net.getEdge().add(et);
505         }
506 
507         return toReturn;
508     }

510     private static String parseAnnotationFromMap(
511             HashMap<String, String> annotationMap) {
512         String toReturn = "";
513         for (Map.Entry<String, String> a : annotationMap.entrySet()) {
514             toReturn += a.getKey() + ":" + a.getValue() + ";";

520         String toReturn = "";
