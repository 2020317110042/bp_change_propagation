18      public static void AppendNodePropagation2CG (Graph mergedModel, Graph variant, Vertex v_p, Vertex new_v, String label, String edge_id, List<VertexPairFromG2CG> vertexPairFromG2CGs, List<EdgePairFromG2CG> edgePairFromG2CGs){
19  
20          //首先将新节点追加到p后, 改变变体
21          ChangeOperationOnProcessVariant.AppendNodeToG(variant,new_v,v_p,label,edge_id);
22  
23          //得到cg中p对应的点的id
24          String right_CG_v_id = findCorrespondingNodeIdInCG(v_p.getID(), vertexPairFromG2CGs);
25  
26          //得到cg中s对应的点cg_p
27          Vertex right_CG_v = new Vertex(v_p.getType(),new String(""), new String(""));
28          List<Vertex> vertexes = mergedModel.getVertices();
29          Iterator<Vertex> it = vertexes.iterator();
30          while (it.hasNext()){
31              Vertex v = it.next();
32              if(v.getID().equals(right_CG_v_id)){
33  
34                      right_CG_v = v;
35                  break;
36              }
37          }
38  
39  
40          //判断p是事件还是函数，判断p的前继边是否有多个注释
41  
42              /*g获得cg_p的前置节点集*/
43          List<Vertex> preset_of_right_CG_v = mergedModel.getPreset(right_CG_v_id);
44          Vertex pre_of_right_CG_v = preset_of_right_CG_v.get(0);
45  
46              /*得到cg_p和其前继节点集的边注释*/
47          Edge e = mergedModel.containsEdge(pre_of_right_CG_v.getID(), right_CG_v_id);
48          HashSet<String> label_of_e = e.getLabels();
49  
50  
51  
52          if((v_p.getType().toString().equals("function") || v_p.getType().toString().equals("event")) && label_of_e.size() > 1){
53  
54              //**create a new auxiliary node 'v'''***//
55              Vertex newGw = new Vertex(Vertex.GWType.xor, mergedModel.getIdGenerator().getNextId());
56              newGw.setConfigurable(true);
57  
58              //**insert the new auxiliary node between 'right_CG_v' and its preceding vertex***********//
59              ChangeOperationOnMergedModel.InsertNodeToCG(mergedModel, pre_of_right_CG_v, right_CG_v, newGw);
60  
61                   /*get the 'pre-set' of 'v_p*/
62              List<Vertex> preset_of_v_p = variant.getPreset(v_p.getID());
63              Vertex pre_of_v_p = preset_of_v_p.get(0);
64  
65                  /*get the label on the edge between 'v_p' and its 'pre-set' in the variant before changing*/
66              Edge edge = variant.containsEdge(pre_of_v_p.getID(), v_p.getID());
67              HashSet<String> label_of_edge = edge.getLabels();
68  
69              /*split the common edge between the newly created auxiliary node and 'right_CG_v'*/
70              ChangeOperationOnMergedModel.SplitCommonRegion(mergedModel, newGw, right_CG_v, right_CG_v, label_of_edge);
71  
72          }
73  
74          //cg中追加节点
75          Vertex new_v_in_CG = new Vertex(new_v.getType(), new_v.getLabel(), mergedModel.getIdGenerator().getNextId());
76          HashSet<String> label_on_new_added_edge = new HashSet<String>();
77          label_on_new_added_edge.add(label);
78          ChangeOperationOnMergedModel.AppendNodeToCG(mergedModel, right_CG_v, new_v_in_CG, label_on_new_added_edge,mergedModel.getIdGenerator().getNextId());
79  
80  
81              /*****************************************************************************************/
82          ///**update the edge mapping between G and CG*,very important*//
83  
84          VertexPairFromG2CG vpg2cg = new VertexPairFromG2CG(new_v.getID(), new_v_in_CG.getID(), variant.ID, mergedModel.ID);
85          vertexPairFromG2CGs.add(vpg2cg);
86  
87          ///**update the edge mapping between G and CG*,very important*//
88  
89          Edge right_CG_e = mergedModel.containsEdge(right_CG_v.getID(), new_v_in_CG.getID());
90          List<String> right = new ArrayList<String>();
91          right.add(right_CG_e.getId());
92          Edge left_G_e = variant.containsEdge(v_p.getID(), new_v.getID());
93          EdgePairFromG2CG epg2cg = new EdgePairFromG2CG(left_G_e.getId(), right, variant.ID, mergedModel.ID, new Boolean(false) );
94          edgePairFromG2CGs.add(epg2cg);
95      }

98      public static void PrependNodePropagation2CG (Graph mergedModel, Graph variant, Vertex new_v, Vertex v_s, String label, String edge_id, List<VertexPairFromG2CG> vertexPairFromG2CGs, List<EdgePairFromG2CG> edgePairFromG2CGs){
99  
100         //首先将新节点预加到s上, 改变变体
101         ChangeOperationOnProcessVariant.PrependNodeToG(variant,new_v,v_s,label, edge_id);
102 
103         //得到cg中s对应的点的id
104         String right_CG_v_id = findCorrespondingNodeIdInCG(v_s.getID(), vertexPairFromG2CGs);
105 
106         //得到cg中s对应的点cg_s
107         Vertex right_CG_v = new Vertex(v_s.getType(),new String(""), new String(""));
108         List<Vertex> vertexes = mergedModel.getVertices();
109         Iterator<Vertex> it = vertexes.iterator();
110         while (it.hasNext()){
111             Vertex v = it.next();
112             if(v.getID().equals(right_CG_v_id)){
113 
114                 right_CG_v = v;
115                 break;
116             }
117         }
118 
119         //得到cg_s的后续节点集
120         List<Vertex> postset_of_right_CG_v = mergedModel.getPostset(right_CG_v_id);
121         Vertex post_of_right_CG_v = postset_of_right_CG_v.get(0);
122 
123         //得到cg_s和其后续节点集的边注释
124         Edge e = mergedModel.containsEdge(right_CG_v_id, post_of_right_CG_v.getID());
125         HashSet<String> label_of_e = e.getLabels();
126 
127         /*判断s是事件还是函数，判断s的后续边是否有多个注释*/
128         if((v_s.getType().toString().equals("function") || v_s.getType().toString().equals("event")) && label_of_e.size() > 1){
129 
130             //创建一个新的辅助节点v''
131             Vertex newGw = new Vertex(Vertex.GWType.xor, mergedModel.getIdGenerator().getNextId());
132             newGw.setConfigurable(true);
133 
134             //将辅助节点插入cg_s与其后续节点之间
135             ChangeOperationOnMergedModel.InsertNodeToCG(mergedModel, right_CG_v, post_of_right_CG_v, newGw);
136 
137             //得到s的后续节点集
138             List<Vertex> postset_of_v_s = variant.getPostset(v_s.getID());
139             Vertex post_of_v_s = postset_of_v_s.get(0);
140 
141             //得到s和其后续节点集的边注释
142             Edge edge = variant.containsEdge(v_s.getID(),post_of_v_s.getID());
143             HashSet<String> label_of_edge = edge.getLabels();
144 
145             //在辅助节点和cg_s间划分公共边
146             ChangeOperationOnMergedModel.SplitCommonRegion(mergedModel, right_CG_v, newGw, right_CG_v, label_of_edge);
147         }
148 
149         //cg中预加节点
150         Vertex new_v_in_CG = new Vertex(new_v.getType(), new_v.getLabel(), mergedModel.getIdGenerator().getNextId());
151         HashSet<String> label_on_new_added_edge = new HashSet<String>();
152         label_on_new_added_edge.add(label);
153         ChangeOperationOnMergedModel.PrependNodeToCG(mergedModel, new_v_in_CG, right_CG_v, label_on_new_added_edge,mergedModel.getIdGenerator().getNextId());
154 
155         //更新g和cg的节点映射
156         VertexPairFromG2CG vpg2g = new VertexPairFromG2CG(new_v.getID(), new_v_in_CG.getID(), variant.ID, mergedModel.ID);
157         vertexPairFromG2CGs.add(vpg2g);
158 
159         //更新g和cg的边映射
160         Edge right_CG_e = mergedModel.containsEdge(new_v_in_CG.getID(),right_CG_v.getID());
161         List<String> right = new ArrayList<String>();
162         right.add(right_CG_e.getId());
163         Edge left_G_e = variant.containsEdge(new_v.getID(),v_s.getID());
164         EdgePairFromG2CG epg2cg = new EdgePairFromG2CG(left_G_e.getId(), right, variant.ID, mergedModel.ID, new Boolean(false) );
165         edgePairFromG2CGs.add(epg2cg);
166     }

168     //插入边的传播(G2CG)
169     public static void InsertEdgePropagation2CG (Graph originalMergedModel, Graph variantToBeChanged, Vertex v_p, Vertex v_s, String label, String edge_id, List<VertexPairFromG2CG> vertexPairFromG2CGs,List<EdgePairFromG2CG> edgePairFromG2CGs){
170 
171         //首先对变体进行插入边操作
172         ChangeOperationOnProcessVariant.InsertEdgeToG(variantToBeChanged,v_p,v_s,label,edge_id);
173 
174         //找到p,s在cg中对应点的id
175         String id_cg_p = findCorrespondingNodeIdInCG(v_p.getID(),vertexPairFromG2CGs);
176         String id_cg_s = findCorrespondingNodeIdInCG(v_s.getID(),vertexPairFromG2CGs);
177 
178         //在cg中找到上述id对应的点，得到cg_p
179         Vertex cg_p = new Vertex(v_p.getType(),new String(""), new String(""));
180         List<Vertex> vertexes = originalMergedModel.getVertices();
181         Iterator<Vertex> it = vertexes.iterator();
182         while (it.hasNext()){
183             Vertex v = it.next();
184             if(v.getID().equals(id_cg_p)){
185                 cg_p = v;
186                 break;
187             }
188         }
189 
190         //得到cg_s
191         Vertex cg_s = new Vertex(v_s.getType(),new String(""), new String(""));
192         List<Vertex> vertexes1 = originalMergedModel.getVertices();
193         Iterator<Vertex> it1 = vertexes1.iterator();
194         while (it1.hasNext()){
195             Vertex v1 = it1.next();
196             if(v1.getID().equals(id_cg_s)){
197                 cg_s = v1;
198                 break;
199             }
200         }
201 
202         //得到变体的边的注释并插入cg的对应边,若没有对应的边则需要插入边
203         HashSet<String> edge_label = variantToBeChanged.getGraphLabels();//得到g的标签
204         ChangeOperationOnMergedModel.InsertEdgeToCG(originalMergedModel, cg_p, cg_s, originalMergedModel.getIdGenerator().getNextId());
205         ChangeOperationOnMergedModel.InsertEdgeAnnotationToCG (originalMergedModel, cg_p, cg_s, edge_label);//在边上插入标签
206 
207         //更新边的映射
208         Edge right_CG_e = originalMergedModel.containsEdge(cg_p.getID(), cg_s.getID());
209         List<String> right = new ArrayList<String>();
210         right.add(right_CG_e.getId());
211         Edge left_G_e = variantToBeChanged.containsEdge(v_p.getID(), v_s.getID());
212         EdgePairFromG2CG epg2cg = new EdgePairFromG2CG(left_G_e.getId(), right, variantToBeChanged.ID, originalMergedModel.ID, new Boolean(false) );
213         edgePairFromG2CGs.add(epg2cg);
214 
215     }

217     //删除边的传播(G2CG)
218     public static void DeleteEdgePropagation2CG (Graph originalMergedModel, Graph variantToBeChanged, Vertex v_p, Vertex v_s, List<VertexPairFromG2CG> vertexPairFromG2CGs,List<EdgePairFromG2CG> edgePairFromG2CGs){
219 
220         //e为变体中对应的边
221         Edge e = variantToBeChanged.containsEdge(v_p.getID(), v_s.getID());
222 
223         //对变体执行删除边操作
224         ChangeOperationOnProcessVariant.DeleteEdgeFromG(variantToBeChanged, v_p, v_s);
225 
226         //用right_CG_e_ids存放cg中与e对应的边的id
227         List<String> right_CG_e_ids = findCorrespondingEdgeSetIdInCG(e.getId(), edgePairFromG2CGs);
228 
229         //得到cg中所有边的id并用right_CG_p存放cg中与e相对应的边(不唯一)
230         List<Edge> right_CG_all_edge = originalMergedModel.getEdges();
231         List<Edge> right_CG_p = new ArrayList<Edge>();
232         Iterator<Edge> it = right_CG_all_edge.iterator();
233         while (it.hasNext()){
234             Edge edge = it.next();
235             if(right_CG_e_ids.contains(edge.getId())){
236                 right_CG_p.add(edge);
237             }
238         }
239 
240         // 用edge_annotation_hash_set存放该变体的标签
241         HashSet<String> edge_annotation_hash_set = variantToBeChanged.getGraphLabels();
242         //删除cg中对应边的相关标签
243         Iterator<Edge> ite = right_CG_p.iterator();
244         while(ite.hasNext()){//循环对right_CG_p中存放的边进行删除注释操作，删除其对应g中（vp，vs）的注释
245             Edge edge = ite.next();
246             //对于每个边缘，删除其上的边缘注释“ pid（variant）”
247             ChangeOperationOnMergedModel.DeleteEdgeAnnotationFromCG(originalMergedModel, edge.getFromVertex(), edge.getToVertex(), edge_annotation_hash_set);
248             if(edge.getLabels().isEmpty()){
249                 //在cg中找到cg_p
250                 Vertex cg_p = new Vertex(v_p.getType(),new String(""), new String(""));
251                 List<Vertex> vertexes = originalMergedModel.getVertices();
252                 Iterator<Vertex> it1 = vertexes.iterator();
253                 while (it1.hasNext()){
254                     Vertex v = it1.next();
255                     if(v.getID().equals(edge.getFromVertex())){
256                         cg_p = v;
257                         break;
258                     }
259                 }
260                 //在cg中找到cg_p
261                 Vertex cg_s = new Vertex(v_s.getType(),new String(""), new String(""));
262                 List<Vertex> vertexes1 = originalMergedModel.getVertices();
263                 Iterator<Vertex> it11 = vertexes1.iterator();
264                 while (it11.hasNext()){
265                     Vertex v1 = it11.next();
266                     if(v1.getID().equals(edge.getToVertex())){
267                         cg_s = v1;
268                         break;
269                     }
270                 }
271                 ChangeOperationOnMergedModel.DeleteEdgeFromCG(originalMergedModel,cg_p,cg_s);
272             }
273         }
274         //更新映射
275         Iterator<EdgePairFromG2CG> iterator = edgePairFromG2CGs.iterator();
276         while(iterator.hasNext()){
277             EdgePairFromG2CG edgePairFromG2CG = iterator.next();
278             //删除部分g与cg的边缘映射，因为变体中的某些边已被删除
279             if (edgePairFromG2CG.getLeft_G_e_id().equals(e.getId())){
280                 iterator.remove();
281             }
282         }
283     }

285     //更新节点标签的传播(G2CG)
286     public static void UpdateLabelPropagation2CG (Graph originalMergedModel, Graph variantToBeChanged, String nlabel, Vertex nv, List<VertexPairFromG2CG> vertexPairFromG2CGs,List<EdgePairFromG2CG> edgePairFromG2CGs){
287         //首先对变体的节点更新标签
288         nv.setLabel(nlabel);
289 
290         //找到cg中nv对应的点cg_nv
291         String id_cg_nv = findCorrespondingNodeIdInCG(nv.getID(),vertexPairFromG2CGs);
292         Vertex cg_nv = new Vertex(nv.getType(),new String(""), new String(""));
293         List<Vertex> vertexes = originalMergedModel.getVertices();
294         Iterator<Vertex> it = vertexes.iterator();
295         while (it.hasNext()){
296             Vertex v = it.next();
297             if(v.getID().equals(id_cg_nv)){
298                 cg_nv = v;
299                 break;
300             }
301         }
302 
303         //得到cg_nv的后续节点集
304         List<Vertex> postset_of_id_cg_nv = originalMergedModel.getPostset(id_cg_nv);
305         HashSet<String> label_of_e = new HashSet<>();
306         String slabel_of_e="";
307         Vertex post_of_id_cg_nv=null;
308         if(!postset_of_id_cg_nv.isEmpty()){
309             post_of_id_cg_nv = postset_of_id_cg_nv.get(0);
310             //得到cg_nv和其后续节点集的边(出边)注释
311             Edge e = originalMergedModel.containsEdge(id_cg_nv, post_of_id_cg_nv.getID());
312             label_of_e = e.getLabels();
313             slabel_of_e = label_of_e.toString();
314         }
315 
316 
317         //得到cg_nv的前继节点集
318         List<Vertex> preset_of_id_cg_nv = originalMergedModel.getPreset(id_cg_nv);
319         HashSet<String> label_of_e1 = new HashSet<>();
320         String slabel_of_e1 ="";
321         Vertex pre_of_id_cg_nv = null;
322         if(!preset_of_id_cg_nv.isEmpty()){
323             pre_of_id_cg_nv = preset_of_id_cg_nv.get(0);
324             //得到cg_nv和其前继节点集的边(入边)注释
325             Edge e1 = originalMergedModel.containsEdge(pre_of_id_cg_nv.getID(), id_cg_nv);
326             label_of_e1 = e1.getLabels();
327             slabel_of_e1 = label_of_e1.toString();
328         }
329 
330 
331         //根据cg_nv的出边和入边的注释数判断如何更新cg_nv的注释-------------------------------------------------------------------------------------
332         if(!slabel_of_e.contains(",") && !slabel_of_e1.contains(",")){//若出边和入边的注释有一个大小为1
333             cg_nv.setLabel(nlabel);//直接更新点注释即可
334             //无需更新点边映射
335         }
336         else if(slabel_of_e1.contains(",") && !slabel_of_e.contains(",") ){//若入边的注释大于1且无出边
337             Vertex m = new Vertex("xor",originalMergedModel.getIdGenerator().getNextId());//定义一个新的xor连接器m
338             ChangeOperationOnMergedModel.InsertNodeToCG(originalMergedModel,pre_of_id_cg_nv,cg_nv,m);//插入到cg_nv及其前继节点之间
339             Vertex v2 = new Vertex(nv.getType(),nlabel,originalMergedModel.getIdGenerator().getNextId());//生成新节点v2
340             //在m后面追加v2
341             ChangeOperationOnMergedModel.AppendNodeToCG(originalMergedModel,m,v2,variantToBeChanged.getGraphLabels(),originalMergedModel.getIdGenerator().getNextId());
342             //删除边(m,cg_nv)上的相关图的注释
343             ChangeOperationOnMergedModel.DeleteEdgeAnnotationFromCG(originalMergedModel,m.getID(),cg_nv.getID(),variantToBeChanged.getGraphLabels());
344 
345             //更新点映射
346             Iterator<VertexPairFromG2CG> iterator = vertexPairFromG2CGs.iterator();//首先删除原本的nv的映射
347             while(iterator.hasNext()){
348                 VertexPairFromG2CG vertexPairFromG2CG = iterator.next();
349                 if (vertexPairFromG2CG.getLeft_G_v_id().equals(nv.getID())){
350                     iterator.remove();
351                 }
352             }
353             VertexPairFromG2CG vpg2g = new VertexPairFromG2CG(nv.getID(), v2.getID(), variantToBeChanged.ID, originalMergedModel.ID);//添加新的nv和v2的映射
354             vertexPairFromG2CGs.add(vpg2g);
355 
356             //更新边映射
357             List<Vertex> preset_of_nv = variantToBeChanged.getPreset(nv.getID());
358             Vertex pre_of_nv = preset_of_nv.get(0);
359             Edge ee = variantToBeChanged.containsEdge(pre_of_nv.getID(), nv.getID());//得到nv的入边ee

361             while(iterator1.hasNext()){
362                 EdgePairFromG2CG edgePairFromG2CG = iterator1.next();
363                 if (edgePairFromG2CG.getLeft_G_e_id().equals(ee.getId())){
364                     iterator1.remove();
365                 }
366             }
367             Edge right_CG_e = originalMergedModel.containsEdge(pre_of_id_cg_nv.getID(),m.getID());
368             Edge right_CG_e1 = originalMergedModel.containsEdge(m.getID(),v2.getID());
369             List<String> right = new ArrayList<String>();
370             right.add(right_CG_e.getId());
371             right.add(right_CG_e1.getId());
372             EdgePairFromG2CG epg2cg = new EdgePairFromG2CG(ee.getId(), right, variantToBeChanged.ID, originalMergedModel.ID, new Boolean(false) );
373             edgePairFromG2CGs.add(epg2cg);//将经过m的一段路径与边ee相映射
374         }
375         else if(slabel_of_e.contains(",") && !slabel_of_e1.contains(",") ){//若出边的注释大于1且无入边
376             Vertex n = new Vertex("xor",originalMergedModel.getIdGenerator().getNextId());//定义一个新的xor连接器n
377             ChangeOperationOnMergedModel.InsertNodeToCG(originalMergedModel,cg_nv,post_of_id_cg_nv,n);//插入到cg_nv及其后续节点之间
378             Vertex v22 = new Vertex(nv.getType(),nlabel,originalMergedModel.getIdGenerator().getNextId());//生成新节点v22
379             //在n前面预加v22
380             ChangeOperationOnMergedModel.PrependNodeToCG(originalMergedModel,v22,n,variantToBeChanged.getGraphLabels(),originalMergedModel.getIdGenerator().getNextId());
381             //删除边(cg_nv,n)上的相关图的注释
382             ChangeOperationOnMergedModel.DeleteEdgeAnnotationFromCG(originalMergedModel,cg_nv.getID(),n.getID(),variantToBeChanged.getGraphLabels());
383 
384             //更新点映射
385             Iterator<VertexPairFromG2CG> iterator = vertexPairFromG2CGs.iterator();//首先删除原本的nv的映射
386             while(iterator.hasNext()){
387                 VertexPairFromG2CG vertexPairFromG2CG = iterator.next();
388                 if (vertexPairFromG2CG.getLeft_G_v_id().equals(nv.getID())){
389                     iterator.remove();
390                 }
391             }
392             VertexPairFromG2CG vpg2g = new VertexPairFromG2CG(nv.getID(), v22.getID(), variantToBeChanged.ID, originalMergedModel.ID);//添加新的nv和v2的映射
393             vertexPairFromG2CGs.add(vpg2g);
394

396             List<Vertex> postset_of_nv = variantToBeChanged.getPostset(nv.getID());
397             Vertex post_of_nv = postset_of_nv.get(0);
398             Edge ee = variantToBeChanged.containsEdge(nv.getID(),post_of_nv.getID());//得到nv的出边ee
399             Iterator<EdgePairFromG2CG> iterator1 = edgePairFromG2CGs.iterator();//删除ee原本的边映射
400             while(iterator1.hasNext()){
401                 EdgePairFromG2CG edgePairFromG2CG = iterator1.next();
402                 if (edgePairFromG2CG.getLeft_G_e_id().equals(ee.getId())){
403                     iterator1.remove();
404                 }
405             }
406             Edge right_CG_e = originalMergedModel.containsEdge(v22.getID(),n.getID());
407             Edge right_CG_e1 = originalMergedModel.containsEdge(n.getID(),post_of_id_cg_nv.getID());
408             List<String> right = new ArrayList<String>();
